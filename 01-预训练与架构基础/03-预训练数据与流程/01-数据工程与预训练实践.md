## 1. 预训练数据收集

### 1.1 主要数据源

大规模语言模型的预训练依赖于海量、多样化的数据源。理解不同数据源的特点、获取方式和质量控制方法是构建高质量训练数据集的基础。

#### Common Crawl 网页数据

Common Crawl 是目前最大的公开网页爬虫数据集，每月抓取数十亿网页，数据量达到数百 TB。

**数据特点**：
- 覆盖全球多语言网页
- 包含 HTML 结构、元数据和原始文本
- 数据噪声较大，需要深度清洗
- 存在大量重复和低质量内容

**获取方式**：

```python
import boto3
from warcio.archiveiterator import ArchiveIterator

def download_common_crawl(bucket='commoncrawl', prefix='crawl-data/CC-MAIN-2024-10/'):
    """下载 Common Crawl WARC 文件"""
    s3 = boto3.client('s3', region_name='us-east-1')
    
    # 列出可用的 WARC 文件
    response = s3.list_objects_v2(
        Bucket=bucket,
        Prefix=prefix + 'warc/',
        MaxKeys=100
    )
    
    warc_files = []
    for obj in response.get('Contents', []):
        if obj['Key'].endswith('.warc.gz'):
            warc_files.append(obj['Key'])
    
    return warc_files

def extract_text_from_warc(warc_file_path):
    """从 WARC 文件中提取文本内容"""
    texts = []
    
    with open(warc_file_path, 'rb') as stream:
        for record in ArchiveIterator(stream):
            if record.rec_type == 'response':
                content_type = record.http_headers.get_header('Content-Type', '')
                if 'text/html' in content_type:
                    html_content = record.content_stream().read()
                    # 提取纯文本（使用 BeautifulSoup 或 trafilatura）
                    text = extract_text(html_content)
                    texts.append(text)
    
    return texts

def extract_text(html_content):
    """使用 trafilatura 提取高质量文本"""
    import trafilatura
    
    html_str = html_content.decode('utf-8', errors='ignore')
    extracted = trafilatura.extract(
        html_str,
        include_comments=False,
        include_tables=False,
        include_links=False,
        favor_precision=True
    )
    
    return extracted if extracted else ""
```

```python
        return papers

    from Bio import Entrez, Medline

        """获取 PubMed 文章摘要"""

            db="pubmed",
            term=query,
            retmax=retmax,
            sort="relevance"
```

```python
def collect_multilingual_data(languages=["en", "zh", "es", "fr", "de", "ja", "ko"]):

    # 2. OSCAR 多语言网页语料
    oscar_datasets = {}
    for lang in languages:
        try:
            oscar = load_dataset("oscar-corpus/OSCAR-2301", lang)
            oscar_datasets[lang] = oscar
        except Exception as e:
            print(f"Failed to load OSCAR for {lang}: {e}")
    
    # 3. Common Crawl 按语言过滤
    cc_datasets = filter_common_crawl_by_language(languages)
    
    return {
        "wikipedia": wiki_datasets,
        "oscar": oscar_datasets,
        "common_crawl": cc_datasets
    }

def filter_common_crawl_by_language(target_languages):
    """按语言过滤 Common Crawl 数据"""
    import fasttext
    
    # 加载语言识别模型
    lid_model = fasttext.load_model('lid.176.bin')
    
    filtered_data = {lang: [] for lang in target_languages}
    
    def process_warc_batch(warc_files):
        for warc_file in warc_files:
            texts = extract_text_from_warc(warc_file)
            
            for text in texts:
                if len(text) < 100:
                    continue
                
                # 语言识别
                predictions = lid_model.predict(text[:1000])
                detected_lang = predictions[0][0].replace('__label__', '')
                
                if detected_lang in target_languages:
                    confidence = predictions[1][0]
                    if confidence > 0.8:  # 高置信度
                        filtered_data[detected_lang].append(text)
    
    return filtered_data
```

```python
def calculate_storage_requirements(
    num_tokens,
    avg_bytes_per_token=4.5,
    compression_ratio=0.3,
    redundancy_factor=3
):
    """计算数据存储需求"""
    
    # 原始数据大小
    raw_size_tb = (num_tokens * avg_bytes_per_token) / (1024**4)
    
    # 压缩后大小
    compressed_size_tb = raw_size_tb * compression_ratio
    
    # 考虑冗余（原始、清洗后、处理中）
    total_storage_tb = compressed_size_tb * redundancy_factor
    
    return {
        "raw_size_tb": round(raw_size_tb, 2),
        "compressed_size_tb": round(compressed_size_tb, 2),
        "total_storage_tb": round(total_storage_tb, 2),
        "recommendation": f"建议准备至少 {total_storage_tb * 1.5:.0f} TB 存储空间（含 50% 余量）"
    }
```

#### 质量标准定义

数据质量直接影响模型性能。需要建立多维度的质量评估体系。

**质量维度与指标**：

```python
class TextQualityMetrics:
    """文本质量评估指标"""
    
    @staticmethod
    def readability_score(text):
        """可读性评分"""
        import textstat
        
        try:
            # Flesch Reading Ease
            flesch = textstat.flesch_reading_ease(text)
            
            # Flesch-Kincaid Grade
            grade = textstat.flesch_kincaid_grade(text)
            
            # Gunning Fog Index
            fog = textstat.gunning_fog(text)
            
            return {
                "flesch_reading_ease": round(flesch, 2),
                "flesch_kincaid_grade": round(grade, 2),
                "gunning_fog": round(fog, 2),
                "is_readable": flesch > 30 and grade < 15
            }
        except:
            return None
    
    @staticmethod
    def information_density(text):
        """信息密度评估"""
        import nltk
        from nltk.tokenize import word_tokenize, sent_tokenize
        
        # 分词和分句
        words = word_tokenize(text)
        sentences = sent_tokenize(text)
        
        # 计算指标
        avg_sentence_length = len(words) / max(len(sentences), 1)
        
        # 词汇丰富度（type-token ratio）
        unique_words = set(w.lower() for w in words if w.isalpha())
        lexical_diversity = len(unique_words) / max(len(words), 1)
        
        # 实体密度
        entities = extract_entities(text)
        entity_density = len(entities) / max(len(sentences), 1)
        
        return {
            "avg_sentence_length": round(avg_sentence_length, 2),
            "lexical_diversity": round(lexical_diversity, 3),
            "entity_density": round(entity_density, 2),
            "info_density_score": round(
                (lexical_diversity * 0.4 + 
                 min(entity_density / 5, 1) * 0.3 +
                 min(avg_sentence_length / 25, 1) * 0.3),
                3
            )
        }
    
    @staticmethod
    def toxicity_detection(text):
        """毒性检测"""
        from transformers import pipeline
        
        # 使用预训练的毒性检测模型
        toxicity_classifier = pipeline(
            "text-classification",
            model="unitary/toxic-bert"
        )
        
        result = toxicity_classifier(text[:512])  # 限制长度
        
        return {
            "is_toxic": result[0]["label"] == "toxic",
            "toxicity_score": result[0]["score"] if result[0]["label"] == "toxic" else 1 - result[0]["score"]
        }
    
    @staticmethod
    def bias_detection(text):
        """偏见检测"""
        # 简化版：检测敏感词汇和刻板印象
        bias_indicators = {
            "gender_bias": ["he always", "she never", "women are", "men are"],
            "racial_bias": ["all [group] are", "they always"],
            "age_bias": ["young people", "old people", "boomers", "millennials"]
        }
        
        text_lower = text.lower()
        detected_biases = []
        
        for bias_type, indicators in bias_indicators.items():
            for indicator in indicators:
                if indicator in text_lower:
                    detected_biases.append({
                        "type": bias_type,
                        "indicator": indicator
                    })
        
        return {
            "has_bias": len(detected_biases) > 0,
            "biases": detected_biases
        }

def comprehensive_quality_score(text, weights=None):
    """综合质量评分"""
    if weights is None:
        weights = {
            "readability": 0.2,
            "information_density": 0.3,
            "low_toxicity": 0.25,
            "low_bias": 0.15,
            "length_appropriateness": 0.1
        }
    
    metrics = TextQualityMetrics()
    
    # 可读性
    readability = metrics.readability_score(text)
    readability_score = 0
    if readability and readability["is_readable"]:
        readability_score = min(readability["flesch_reading_ease"] / 100, 1.0)
    
    # 信息密度
    info_density = metrics.information_density(text)
    info_score = info_density["info_density_score"]
    
    # 毒性（越低越好）
    toxicity = metrics.toxicity_detection(text)
    toxicity_score = 1 - toxicity["toxicity_score"]
    
    # 偏见（越低越好）
    bias = metrics.bias_detection(text)
    bias_score = 0 if bias["has_bias"] else 1.0
    
    # 长度适当性
    length = len(text)
    if 100 <= length <= 10000:
        length_score = 1.0
    elif length < 100:
        length_score = length / 100
    else:
        length_score = max(0, 1 - (length - 10000) / 90000)
    
    # 加权总分
    total_score = (
        readability_score * weights["readability"] +
        info_score * weights["information_density"] +
        toxicity_score * weights["low_toxicity"] +
        bias_score * weights["low_bias"] +
        length_score * weights["length_appropriateness"]
    )
    
    return {
        "total_score": round(total_score, 3),
        "readability_score": round(readability_score, 3),
        "information_density_score": round(info_score, 3),
        "toxicity_score": round(toxicity_score, 3),
        "bias_score": round(bias_score, 3),
        "length_score": round(length_score, 3)
    }
```

```python
    import fasttext

    lid_model = fasttext.load_model('lid.176.bin')
    
    language_counts = Counter()
    
    # 采样分析
    samples = random.sample(dataset, min(sample_size, len(dataset)))
    
    for text in samples:
        if len(text) < 50:
            continue
        
        # 预测语言
        predictions = lid_model.predict(text[:1000])
        lang = predictions[0][0].replace('__label__', '')
        confidence = predictions[1][0]
        
        if confidence > 0.8:
            language_counts[lang] += 1
    
    # 计算分布
    total = sum(language_counts.values())
    distribution = {
        lang: {
            "count": count,
            "percentage": round(count / total * 100, 2)
        }
        for lang, count in language_counts.most_common()
    }
    
    return distribution
```

```python
class NoiseFilter:
    """噪声过滤器"""
    
    def __init__(self):
        # 定义噪声模式
        self.noise_patterns = [
            # 重复字符（如 "aaaaaa"）
            re.compile(r'(.)\1{10,}'),
            # URL 列表
            re.compile(r'(https?://\S+\s*){5,}'),
            # 导航菜单模式
            re.compile(r'(Home|Menu|Login|Register|Contact).*\1'),
            # 广告模式
            re.compile(r'(Buy now|Subscribe|Click here|Sign up).*', re.IGNORECASE),
            # 大量数字（可能是表格或数据）
            re.compile(r'(\d+\s+){20,}'),
            # 特殊字符过多
            re.compile(r'[^\w\s]{50,}'),
        ]
        
        # 低质量指标
        self.min_length = 50
        self.max_length = 100000
        self.max_special_char_ratio = 0.3
        self.min_word_ratio = 0.5
    
    def contains_noise(self, text):
        """检查文本是否包含噪声"""
        for pattern in self.noise_patterns:
            if pattern.search(text):
                return True
        return False
    
    def quality_check(self, text):
        """质量检查"""
        # 长度检查
        if len(text) < self.min_length or len(text) > self.max_length:
            return False
        
        # 特殊字符比例
        special_chars = len(re.findall(r'[^\w\s]', text))
        special_ratio = special_chars / max(len(text), 1)
        if special_ratio > self.max_special_char_ratio:
            return False
        
        # 单词比例（英文）
        words = re.findall(r'\b\w+\b', text)
        word_ratio = len(words) / max(len(text.split()), 1)
        if word_ratio < self.min_word_ratio:
            return False
        
        return True
    
    def clean_noise(self, text):
        """清理噪声"""
        # 移除多余空白
        text = re.sub(r'\n{3,}', '\n\n', text)
        text = re.sub(r' {2,}', ' ', text)
        
        # 移除孤立字符
        text = re.sub(r'\b\w\b', '', text)
        
        # 移除空行
        lines = [line.strip() for line in text.split('\n')]
        lines = [line for line in lines if line]
        text = '\n'.join(lines)
        
        return text.strip()
    
    def filter_document(self, text):
        """过滤文档"""
        # 检查噪声
        if self.contains_noise(text):
            return None
        
        # 质量检查
        if not self.quality_check(text):
            return None
        
        # 清理噪声
        cleaned = self.clean_noise(text)
        
        # 最终检查
        if len(cleaned) < self.min_length:
            return None
        
        return cleaned

# 使用示例
```

```python
        if len(paragraphs) >= 3:

        }
        
        final_score = sum(scores[k] * weights[k] for k in weights)
```

### 2.2 去重技术

数据去重是预训练数据准备的关键步骤。重复数据会导致模型过拟合、训练效率低下和评估偏差。

#### MinHash + LSH 去重

MinHash（最小哈希）结合 LSH（局部敏感哈希）是大规模文本去重的经典方法，能够在近似线性时间内找出相似文档。

**核心原理**：

```python
import hashlib
import numpy as np
from datasketch import MinHash, MinHashLSH

class MinHashDeduplicator:
    """基于 MinHash + LSH 的去重器"""
    
    def __init__(self, num_perm=128, threshold=0.8):
        """
        初始化去重器
        
        Args:
            num_perm: 哈希置换数量（越多越准确，但计算成本更高）
            threshold: Jaccard 相似度阈值
        """
        self.num_perm = num_perm
        self.threshold = threshold
        self.lsh = MinHashLSH(threshold=threshold, num_perm=num_perm)
        self.seen_hashes = set()
    
    def _create_minhash(self, text):
        """创建 MinHash 对象"""
        m = MinHash(num_perm=self.num_perm)
        
        # 分 shingle（n-gram）
        shingles = self._get_shingles(text, n=5)
        
        for shingle in shingles:
            m.update(shingle.encode('utf-8'))
        
        return m
    
    def _get_shingles(self, text, n=5):
        """生成 n-gram shingles"""
        words = text.lower().split()
        shingles = set()
        
        for i in range(len(words) - n + 1):
            shingle = ' '.join(words[i:i+n])
            shingles.add(shingle)
        
        return shingles
    
    def add_document(self, doc_id, text):
        """添加文档到 LSH 索引"""
        minhash = self._create_minhash(text)
        
        # 检查是否重复
        duplicates = self.lsh.query(minhash)
        
        if duplicates:
            # 找到重复文档
            return {
                'is_duplicate': True,
                'duplicate_of': duplicates[0],
                'similarity': self._estimate_similarity(minhash, duplicates[0])
            }
        else:
            # 新文档，添加到索引
            self.lsh.insert(doc_id, minhash)
            return {
                'is_duplicate': False,
                'duplicate_of': None
            }
    
    def _estimate_similarity(self, m1, m2):
        """估计 Jaccard 相似度"""
        # MinHash 的交集比例就是 Jaccard 相似度的估计
        if hasattr(m1, 'jaccard'):
            return m1.jaccard(m2)
        return None
    
    def deduplicate_batch(self, documents, batch_size=10000):
        """批量去重"""
        unique_docs = []
        duplicate_count = 0
        duplicate_pairs = []
        
        for i, doc in enumerate(documents):
            doc_id = f"doc_{i}"
            result = self.add_document(doc_id, doc)
            
            if not result['is_duplicate']:
                unique_docs.append(doc)
            else:
                duplicate_count += 1
                duplicate_pairs.append({
                    'doc_id': doc_id,
                    'duplicate_of': result['duplicate_of'],
                    'similarity': result['similarity']
                })
            
            if (i + 1) % batch_size == 0:
                print(f"Processed {i+1}/{len(documents)} documents, "
                      f"found {duplicate_count} duplicates")
        
        print(f"\nDeduplication complete:")
        print(f"  Total: {len(documents)}")
        print(f"  Unique: {len(unique_docs)}")
        print(f"  Duplicates: {duplicate_count}")
        print(f"  Duplicate rate: {duplicate_count/len(documents)*100:.2f}%")
        
        return unique_docs, duplicate_pairs

# 使用示例
def minhash_deduplication_pipeline(documents):
    """MinHash 去重管道"""
    deduplicator = MinHashDeduplicator(
        num_perm=128,
        threshold=0.8
    )
    
    unique_docs, duplicates = deduplicator.deduplicate_batch(documents)
    
    return unique_docs, duplicates
```

#### SimHash 去重

SimHash 是另一种高效的近似去重算法，特别适合海量数据的去重场景。

```python
class SimHashDeduplicator:
    """基于 SimHash 的去重器"""
    
    def __init__(self, hash_size=64, similarity_threshold=3):
        """
        Args:
            hash_size: 哈希位数
            similarity_threshold: 汉明距离阈值
        """
        self.hash_size = hash_size
        self.similarity_threshold = similarity_threshold
        self.hashes = {}  # {doc_id: simhash_value}
    
    def _simhash(self, text):
        """计算 SimHash 值"""
        import jieba
        
        # 分词
        words = jieba.lcut(text.lower())
        
        # 计算每个词的哈希
        v = [0] * self.hash_size
        
        for word in words:
            # 计算词的哈希
            h = int(hashlib.md5(word.encode('utf-8')).hexdigest(), 16)
            
            for i in range(self.hash_size):
                bitmask = 1 << i
                if h & bitmask:
                    v[i] += 1
                else:
                    v[i] -= 1
        
        # 生成 SimHash
        fingerprint = 0
        for i in range(self.hash_size):
            if v[i] > 0:
                fingerprint |= (1 << i)
        
        return fingerprint
    
    def _hamming_distance(self, h1, h2):
        """计算汉明距离"""
        x = h1 ^ h2
        return bin(x).count('1')
    
    def add_document(self, doc_id, text):
        """添加文档并检查重复"""
        simhash = self._simhash(text)
        
        # 检查是否与已有文档重复
        for existing_id, existing_hash in self.hashes.items():
            distance = self._hamming_distance(simhash, existing_hash)
            
            if distance <= self.similarity_threshold:
                return {
                    'is_duplicate': True,
                    'duplicate_of': existing_id,
                    'hamming_distance': distance
                }
        
        # 新文档
        self.hashes[doc_id] = simhash
        return {
            'is_duplicate': False,
            'duplicate_of': None
        }
    
    def deduplicate_dataset(self, documents):
        """对整个数据集去重"""
        unique_docs = []
        duplicate_count = 0
        
        for i, doc in enumerate(documents):
            doc_id = f"doc_{i}"
            result = self.add_document(doc_id, doc)
            
            if not result['is_duplicate']:
                unique_docs.append(doc)
            else:
                duplicate_count += 1
            
            if (i + 1) % 10000 == 0:
                print(f"Processed {i+1} documents, "
                      f"duplicates: {duplicate_count}")
        
        return unique_docs, duplicate_count
```

#### 语义去重

基于嵌入向量的语义去重能够识别内容相似但表述不同的文档。

```python
class SemanticDeduplicator:
    """基于语义相似度的去重器"""
    
    def __init__(self, model_name='sentence-transformers/all-MiniLM-L6-v2'):
        from sentence_transformers import SentenceTransformer
        import faiss
        
        self.model = SentenceTransformer(model_name)
        self.embeddings = []
        self.faiss_index = None
    
    def _build_index(self, embeddings):
        """构建 FAISS 索引"""
        import faiss
        import numpy as np
        
        dim = embeddings.shape[1]
        
        # 使用内积索引（余弦相似度）
        self.faiss_index = faiss.IndexFlatIP(dim)
        
        # 归一化
        faiss.normalize_L2(embeddings)
        
        # 添加到索引
        self.faiss_index.add(embeddings)
    
    def compute_embeddings(self, texts, batch_size=256):
        """计算文本嵌入"""
        embeddings = []
        
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            batch_embeddings = self.model.encode(
                batch,
                batch_size=batch_size,
                show_progress_bar=True,
                convert_to_numpy=True
            )
            embeddings.append(batch_embeddings)
        
        return np.vstack(embeddings)
    
    def find_duplicates(self, embeddings, threshold=0.9):
        """查找重复文档"""
        import numpy as np
        
        # 归一化
        faiss.normalize_L2(embeddings)
        
        # 批量搜索
        batch_size = 1000
        duplicates = set()
        
        for i in range(0, len(embeddings), batch_size):
            batch = embeddings[i:i+batch_size]
            
            # 搜索相似向量（k=2，排除自身）
            scores, indices = self.faiss_index.search(batch, k=2)
            
            for j in range(len(batch)):
                # 第二个结果是最相似的其它文档
                sim_score = scores[j][1]
                neighbor_idx = indices[j][1]
                
                if sim_score >= threshold and neighbor_idx != -1:
                    # 避免重复计数
                    pair = tuple(sorted([i+j, neighbor_idx]))
                    duplicates.add(pair)
        
        return duplicates
    
    def deduplicate_semantic(self, texts, threshold=0.9):
        """语义去重"""
        print("Computing embeddings...")
        embeddings = self.compute_embeddings(texts)
        
        print("Building FAISS index...")
        self._build_index(embeddings)
        
        print("Finding duplicates...")
        duplicates = self.find_duplicates(embeddings, threshold)
        
        # 标记重复文档
        duplicate_indices = set()
        for i, j in duplicates:
            duplicate_indices.add(j)  # 保留第一个，标记第二个为重复
        
        unique_texts = [
            text for i, text in enumerate(texts)
            if i not in duplicate_indices
        ]
        
        print(f"Total: {len(texts)}")
        print(f"Unique: {len(unique_texts)}")
        print(f"Duplicates: {len(duplicate_indices)}")
        
        return unique_texts, list(duplicate_indices)
```

```python
class CrossDocumentDeduplicator:
    """跨文档片段去重"""
    
    def __init__(self, chunk_size=200, overlap=50):
        """
        Args:
            chunk_size: 文本块大小（词数）
            overlap: 重叠词数
        """
        self.chunk_size = chunk_size
        self.overlap = overlap
        self.chunk_index = {}  # {chunk_hash: [doc_ids]}
    
    def _split_into_chunks(self, text):
        """将文本分割为重叠块"""
        words = text.split()
        chunks = []
        
        for i in range(0, len(words), self.chunk_size - self.overlap):
            chunk = ' '.join(words[i:i+self.chunk_size])
            if len(chunk.split()) >= self.chunk_size * 0.8:  # 至少 80% 长度
                chunks.append(chunk)
        
        return chunks
    
    def find_duplicate_chunks(self, documents):
        """查找跨文档重复片段"""
        from collections import defaultdict
        
        chunk_to_docs = defaultdict(list)
        
        for doc_id, doc in enumerate(documents):
            chunks = self._split_into_chunks(doc)
            
            for chunk in chunks:
                # 使用哈希快速索引
                chunk_hash = hash(chunk)
                chunk_to_docs[chunk_hash].append({
                    'doc_id': doc_id,
                    'chunk': chunk
                })
        
        # 找出在多个文档中出现的块
        duplicate_chunks = {
            chunk_hash: docs
            for chunk_hash, docs in chunk_to_docs.items()
            if len(docs) > 1
        }
        
        return duplicate_chunks
    
    def remove_duplicate_content(self, documents, duplicate_chunks):
        """移除重复内容"""
        cleaned_docs = []
        
        for doc_id, doc in enumerate(documents):
            cleaned_doc = doc
            
            # 查找并移除重复块
            for chunk_hash, occurrences in duplicate_chunks.items():
                if any(occ['doc_id'] == doc_id for occ in occurrences):
                    # 这个文档包含重复块
                    for occ in occurrences:
                        if occ['doc_id'] == doc_id:
                            chunk = occ['chunk']
                            # 简单替换（实际应更智能）
                            if chunk in cleaned_doc:
                                cleaned_doc = cleaned_doc.replace(chunk, '', 1)
            
            cleaned_docs.append(cleaned_doc.strip())
        
        return cleaned_docs
```

```python
class InformativenessScorer:
    """信息量评分器"""
    
    def score(self, text):
        from collections import Counter
        import numpy as np
        
        words = text.lower().split()

            lexical_diversity * 0.3 +
            min(entity_density / 5, 1) * 0.3 +

        return {

            'details': {
                'lexical_diversity': round(lexical_diversity, 3),

class CoherenceScorer:
    """连贯性评分器"""
    
    def score(self, text):
        paragraphs = text.split('\n\n')
        
        if len(paragraphs) < 2:
            return {'score': 0.5, 'details': {'reason': 'too_few_paragraphs'}}
        
        # 1. 段落长度一致性
        para_lengths = [len(p.split()) for p in paragraphs]
        length_std = np.std(para_lengths)
        length_cv = length_std / max(np.mean(para_lengths), 1)
        length_score = max(0, 1 - length_cv)
        
        # 2. 主题一致性（使用 TF-IDF）
        from sklearn.feature_extraction.text import TfidfVectorizer
        from sklearn.metrics.pairwise import cosine_similarity
        
        vectorizer = TfidfVectorizer(max_features=100)
        tfidf_matrix = vectorizer.fit_transform(paragraphs)
        
        # 计算相邻段落相似度
        similarities = []
        for i in range(len(paragraphs) - 1):
            sim = cosine_similarity(
                tfidf_matrix[i:i+1],
                tfidf_matrix[i+1:i+2]
            )[0][0]
            similarities.append(sim)
        
        coherence_score = np.mean(similarities) if similarities else 0
        
        # 综合评分
        final_score = length_score * 0.4 + coherence_score * 0.6
        
        return {
            'score': round(min(final_score, 1.0), 3),
            'details': {
                'length_consistency': round(length_score, 3),
                'topic_coherence': round(coherence_score, 3),
                'num_paragraphs': len(paragraphs)
            }
        }
```

```python
        if output_file:
            self._save_results(final_docs, stats, output_file)
        
        return final_docs, stats
    
    def _filter_by_language(self, documents):
        """按语言过滤"""
        filtered = []
        rejected = 0
        
        for doc in documents:
            langs = self.language_identifier.identify(doc)
            
            if langs:
                top_lang = langs[0]
                if (top_lang['language'] in self.config['target_languages'] and
                    top_lang['confidence'] >= 0.85):
                    filtered.append(doc)
                else:
                    rejected += 1
            else:
                rejected += 1
        
        print(f"Language filter: {len(filtered)} accepted, {rejected} rejected")
        return filtered
    
    def _deduplicate(self, documents):
        """去重"""
        unique_docs, _ = self.deduplicator.deduplicate_batch(documents)
        return unique_docs
    
    def _filter_by_quality(self, documents):
        """按质量过滤"""
        filtered = []
        quality_scores = []
        rejection_reasons = {
            'low_score': 0,
            'too_toxic': 0,
            'has_bias': 0,
            'length_invalid': 0

            )

            if eval_result['total_score'] < self.config['min_quality_score']:
                rejection_reasons['low_score'] += 1
                continue
            
            # 毒性检查
            if eval_result['dimension_scores']['toxicity'] < (1 - self.config['max_toxicity']):
                rejection_reasons['too_toxic'] += 1
                continue

               eval_result['details']['bias']['bias_count'] > self.config['max_bias_count']:
                rejection_reasons['has_bias'] += 1
                continue

                print(f"Quality filter progress: {i+1}/{len(documents)}, "
                      f"accepted: {len(filtered)}")
        
        # 统计
        stats = {
            'mean_quality_score': np.mean(quality_scores) if quality_scores else 0,
            'median_quality_score': np.median(quality_scores) if quality_scores else 0,
            'rejection_reasons': rejection_reasons
        }
        
        print(f"Quality filter: {len(filtered)} accepted")
        print(f"Mean quality score: {stats['mean_quality_score']:.3f}")
        
        return filtered, stats
    
    def _final_validation(self, documents):
        """最终验证"""
        validated = []
        
        for doc in documents:
            # 基本验证
            if doc and len(doc.strip()) >= self.config['min_length']:
                validated.append(doc.strip())
        
        print(f"Final validation: {len(validated)} documents passed")
        return validated
    
    def _save_results(self, documents, stats, output_file):

        with open(output_file, 'w', encoding='utf-8') as f:
            json.dump({
                'documents': documents,
                'statistics': stats
            }, f, ensure_ascii=False, indent=2)
        
        print(f"Results saved to {output_file}")
```

#### 分片与打包

将词元化的数据组织成固定长度的序列，以便高效训练。

```python
class SequencePacker:
    """序列打包器"""
    
    def __init__(self, max_seq_length=2048, eos_token_id=2):
        self.max_seq_length = max_seq_length
        self.eos_token_id = eos_token_id
    
    def pack_sequences(self, token_ids_list, strategy='concat'):
        """
        打包序列
        
        Args:
            token_ids_list: 词元 ID 列表
            strategy: 打包策略 ('concat', 'pad', 'truncate')
        """
        packed_sequences = []
        
        if strategy == 'concat':
            # 连接多个文档到固定长度
            packed_sequences = self._concat_pack(token_ids_list)
        elif strategy == 'pad':
            # 填充到固定长度
            packed_sequences = self._pad_pack(token_ids_list)
        elif strategy == 'truncate':
            # 截断到固定长度
            packed_sequences = self._truncate_pack(token_ids_list)
        
        return packed_sequences
    
    def _concat_pack(self, token_ids_list):
        """连接打包（文档包）"""
        packed = []
        current_sequence = []
        
        for token_ids in token_ids_list:
            # 添加 EOS token
            sequence_with_eos = token_ids + [self.eos_token_id]
            
            # 如果当前序列加上这个文档超过最大长度
            if len(current_sequence) + len(sequence_with_eos) > self.max_seq_length:
                # 保存当前序列
                if current_sequence:
                    # 填充到最大长度
                    current_sequence = current_sequence[:self.max_seq_length]
                    if len(current_sequence) < self.max_seq_length:
                        current_sequence.extend(
                            [self.eos_token_id] * (self.max_seq_length - len(current_sequence))
                        )
                    packed.append(current_sequence)
                
                # 开始新序列
                current_sequence = sequence_with_eos[:self.max_seq_length]
            else:
                current_sequence.extend(sequence_with_eos)
        
        # 处理最后一个序列
        if current_sequence:
            current_sequence = current_sequence[:self.max_seq_length]
            if len(current_sequence) < self.max_seq_length:
                current_sequence.extend(
                    [self.eos_token_id] * (self.max_seq_length - len(current_sequence))
                )
            packed.append(current_sequence)
        
        return packed
    
    def _pad_pack(self, token_ids_list):
        """填充打包"""
        packed = []
        
        for token_ids in token_ids_list:
            if len(token_ids) >= self.max_seq_length:
                # 截断
                packed.append(token_ids[:self.max_seq_length])
            else:
                # 填充
                padded = token_ids + [self.eos_token_id] * (self.max_seq_length - len(token_ids))
                packed.append(padded)
        
        return packed
    
    def _truncate_pack(self, token_ids_list):
        """截断打包"""
        return [
            token_ids[:self.max_seq_length]
            for token_ids in token_ids_list
        ]
    
    def create_training_data(self, packed_sequences):
        """创建训练数据（输入-目标对）"""
        import torch
        
        input_ids = []
        labels = []
        
        for sequence in packed_sequences:
            # 输入：sequence[:-1]
            # 目标：sequence[1:]
            input_ids.append(sequence[:-1])
            labels.append(sequence[1:])
        
        # 转换为张量
        input_tensor = torch.tensor(input_ids, dtype=torch.long)
        labels_tensor = torch.tensor(labels, dtype=torch.long)
        
        return input_tensor, labels_tensor
```

```python
class BufferedShuffler:
    """缓冲洗牌器（用于流式数据）"""
    
    def __init__(self, buffer_size=100000, seed=42):
        self.buffer_size = buffer_size
        self.rng = np.random.RandomState(seed)
        self.buffer = []
    
    def shuffle_stream(self, data_stream):
        """流式洗牌"""
        for item in data_stream:
            self.buffer.append(item)
            
            if len(self.buffer) >= self.buffer_size:
                # 洗牌缓冲区
                self.rng.shuffle(self.buffer)
                
                # 输出一半
                output = self.buffer[:self.buffer_size//2]
                self.buffer = self.buffer[self.buffer_size//2:]
                
                yield from output
        
        # 输出剩余数据
        if self.buffer:
            self.rng.shuffle(self.buffer)
            yield from self.buffer
```

```python
import os
import json
from datetime import datetime
import torch

class CheckpointManager:
    """检查点管理器"""
    
    def __init__(self, checkpoint_dir, max_checkpoints=5):
        self.checkpoint_dir = checkpoint_dir
        self.max_checkpoints = max_checkpoints
        self.checkpoints = []
        
        os.makedirs(checkpoint_dir, exist_ok=True)
    
    def save_checkpoint(
        self,
        model,
        optimizer,
        scheduler,
        step,
        epoch,
        loss,
        metadata=None
    ):
        """保存检查点"""
        checkpoint = {
            'step': step,
            'epoch': epoch,
            'loss': loss,
            'model_state_dict': model.state_dict(),
            'optimizer_state_dict': optimizer.state_dict(),
            'scheduler_state_dict': scheduler.state_dict(),
            'timestamp': datetime.now().isoformat(),
            'metadata': metadata or {}
        }
        
        # 保存文件
        checkpoint_file = os.path.join(
            self.checkpoint_dir,
            f'checkpoint_step_{step}.pt'
        )
        torch.save(checkpoint, checkpoint_file)
        
        # 记录检查点
        self.checkpoints.append({
            'file': checkpoint_file,
            'step': step,
            'epoch': epoch,
            'loss': loss
        })
        
        # 清理旧检查点
        self._cleanup_old_checkpoints()
        
        # 保存最新检查点（便于恢复）
        latest_file = os.path.join(self.checkpoint_dir, 'latest_checkpoint.pt')
        torch.save(checkpoint, latest_file)
        
        # 保存检查点历史
        history_file = os.path.join(self.checkpoint_dir, 'checkpoint_history.json')
        with open(history_file, 'w') as f:
            json.dump(self.checkpoints, f, indent=2)
        
        print(f"Saved checkpoint at step {step}")
        return checkpoint_file
    
    def load_checkpoint(self, checkpoint_file, model, optimizer=None, scheduler=None):
        """加载检查点"""
        print(f"Loading checkpoint: {checkpoint_file}")
        
        checkpoint = torch.load(checkpoint_file, map_location='cpu')
        
        # 加载模型状态
        model.load_state_dict(checkpoint['model_state_dict'])
        
        # 加载优化器状态
        if optimizer:
            optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
        
        # 加载调度器状态
        if scheduler:
            scheduler.load_state_dict(checkpoint['scheduler_state_dict'])
        
        print(f"Loaded checkpoint: step={checkpoint['step']}, "
              f"epoch={checkpoint['epoch']}, loss={checkpoint['loss']:.4f}")
        
        return checkpoint
    
    def get_latest_checkpoint(self):
        """获取最新检查点"""
        latest_file = os.path.join(self.checkpoint_dir, 'latest_checkpoint.pt')
        
        if os.path.exists(latest_file):
            return latest_file
        return None
    
    def _cleanup_old_checkpoints(self):
        """清理旧检查点"""
        if len(self.checkpoints) > self.max_checkpoints:
            # 保留最新的 max_checkpoints 个
            to_remove = self.checkpoints[:-self.max_checkpoints]
            
            for cp in to_remove:
                if os.path.exists(cp['file']):
                    os.remove(cp['file'])
            
            self.checkpoints = self.checkpoints[-self.max_checkpoints:]
            print(f"Cleaned up old checkpoints, kept {len(self.checkpoints)}")
```

```python
        # 计算速度
        time_since_last = current_time - self.last_log_time
        steps_since_last = step - self.last_log_step
        
        if time_since_last > 0:
            steps_per_second = steps_since_last / time_since_last
            tokens_per_second = steps_per_second * batch_size * 2048  # 假设序列长度 2048
        else:
            steps_per_second = 0
            tokens_per_second = 0

        self.loss_history.append(loss)
        self.lr_history.append(lr)
        self.speed_history.append(steps_per_second)
        
        # 定期输出日志
        if step % self.log_interval == 0:
            avg_loss = sum(self.loss_history) / len(self.loss_history)
            avg_speed = sum(self.speed_history) / len(self.speed_history)
            
            elapsed_time = current_time - self.start_time
            eta = self._estimate_eta(step, elapsed_time)
            
            log_entry = {
                'step': step,
                'epoch': epoch,
                'loss': round(loss, 4),
                'avg_loss': round(avg_loss, 4),
                'learning_rate': lr,
                'steps_per_second': round(steps_per_second, 2),

        remaining_steps = total_steps - current_step
        eta_seconds = (elapsed_time / current_step) * remaining_steps

        eta_hours = eta_seconds / 3600
```

#### 故障恢复

```python
class FaultTolerantTrainer:
    """容错训练器"""
    
    def __init__(
        self,
        model,
        optimizer,
        scheduler,
        data_loader,
        checkpoint_manager,
        monitor
    ):
        self.model = model
        self.optimizer = optimizer
        self.scheduler = scheduler
        self.data_loader = data_loader
        self.checkpoint_manager = checkpoint_manager
        self.monitor = monitor
        
        self.global_step = 0
        self.current_epoch = 0
        self.best_loss = float('inf')
    
    def train(
        self,
        num_epochs,
        gradient_accumulation_steps=1,
        max_grad_norm=1.0,
        resume_from=None
    ):
        """训练循环"""
        # 恢复训练
        if resume_from:
            self._resume(resume_from)
        else:
            self.monitor.start()
        
        for epoch in range(self.current_epoch, num_epochs):
            self.current_epoch = epoch
            self.model.train()
            
            epoch_loss = 0
            num_batches = 0
            
            for batch_idx, batch in enumerate(self.data_loader):
                try:
                    # 前向传播
                    outputs = self.model(
                        input_ids=batch['input_ids'],
                        labels=batch['labels']
                    )
                    loss = outputs.loss
                    
                    # 梯度累积
                    loss = loss / gradient_accumulation_steps
                    loss.backward()
                    
                    # 更新权重
                    if (batch_idx + 1) % gradient_accumulation_steps == 0:
                        # 梯度裁剪
                        if max_grad_norm > 0:
                            torch.nn.utils.clip_grad_norm_(
                                self.model.parameters(),
                                max_grad_norm
                            )
                        
                        self.optimizer.step()
                        self.scheduler.step()
                        self.optimizer.zero_grad()
                        
                        self.global_step += 1
                        
                        # 记录日志
                        current_lr = self.scheduler.get_last_lr()[0]
                        batch_size = batch['input_ids'].size(0)
                        self.monitor.log_step(
                            self.global_step,
                            epoch,
                            loss.item() * gradient_accumulation_steps,
                            current_lr,
                            batch_size
                        )
                    
                    epoch_loss += loss.item() * gradient_accumulation_steps
                    num_batches += 1
                    
                    # 定期保存检查点
                    if self.global_step % 1000 == 0:
                        self._save_checkpoint(epoch, loss.item())
                    
                except Exception as e:
                    print(f"Error at step {self.global_step}: {e}")
                    print("Attempting to recover...")
                    self._recover_from_error()
                    continue
            
            # Epoch 结束
            avg_epoch_loss = epoch_loss / max(num_batches, 1)
            print(f"Epoch {epoch} completed, avg loss: {avg_epoch_loss:.4f}")
            
            # 保存 Epoch 检查点
            self._save_checkpoint(epoch, avg_epoch_loss, is_epoch_end=True)
            
            # 更新最佳损失
            if avg_epoch_loss < self.best_loss:
                self.best_loss = avg_epoch_loss
                self._save_best_model()
    
    def _resume(self, checkpoint_file):
        """恢复训练"""
        checkpoint = self.checkpoint_manager.load_checkpoint(
            checkpoint_file,
            self.model,
            self.optimizer,
            self.scheduler
        )
        
        self.global_step = checkpoint['step']
        self.current_epoch = checkpoint['epoch']
        
        print(f"Resumed training from step {self.global_step}, "
              f"epoch {self.current_epoch}")
    
    def _save_checkpoint(self, epoch, loss, is_epoch_end=False):
        """保存检查点"""
        self.checkpoint_manager.save_checkpoint(
            self.model,
            self.optimizer,
            self.scheduler,
            self.global_step,
            epoch,
            loss,
            metadata={
                'is_epoch_end': is_epoch_end,
                'best_loss': self.best_loss
            }
        )
    
    def _save_best_model(self):
        """保存最佳模型"""
        best_model_file = os.path.join(
            self.checkpoint_manager.checkpoint_dir,
            'best_model.pt'
        )
        torch.save(self.model.state_dict(), best_model_file)
        print(f"Saved best model (loss: {self.best_loss:.4f})")
    
    def _recover_from_error(self):
        """从错误中恢复"""
        # 清空梯度
        self.optimizer.zero_grad()
        
        # 跳过当前 batch
        print("Skipped problematic batch, continuing training")
```

```python
class DataMixtureOptimizer:
    """数据配比优化器"""
    
    def __init__(self, target_tokens, domain_configs):
        """
        Args:
            target_tokens: 目标总 token 数
            domain_configs: 领域配置 {domain: {'ratio': 0.3, 'sources': [...]}}
        """
        self.target_tokens = target_tokens
        self.domain_configs = domain_configs
    
    def calculate_domain_tokens(self):
        """计算各领域 token 数量"""
        domain_tokens = {}
        
        for domain, config in self.domain_configs.items():
            ratio = config['ratio']
            tokens = int(self.target_tokens * ratio)
            domain_tokens[domain] = {
                'tokens': tokens,
                'ratio': ratio,
                'sources': config.get('sources', []),
                'estimated_documents': tokens // 500  # 假设平均每文档 500 tokens
            }
        
        return domain_tokens
    
    def optimize_mixture(self, available_data):
        """
        优化数据配比
        
        Args:
            available_data: {domain: available_tokens}
        """
        optimized = {}
        total_available = sum(available_data.values())
        
        for domain, config in self.domain_configs.items():
            target_tokens = int(self.target_tokens * config['ratio'])
            available = available_data.get(domain, 0)
            
            if available >= target_tokens:
                # 数据充足，按比例采样
                optimized[domain] = {
                    'use_tokens': target_tokens,
                    'available_tokens': available,
                    'sampling_ratio': 1.0,
                    'status': 'sufficient'
                }
            else:
                # 数据不足
                optimized[domain] = {
                    'use_tokens': available,
                    'available_tokens': available,
                    'sampling_ratio': 1.0,
                    'status': 'insufficient',
                    'deficit': target_tokens - available
                }
        
        # 重新分配不足的部分
        self._redistribute_deficit(optimized)
        
        return optimized
    
    def _redistribute_deficit(self, optimized):
        """重新分配不足的部分"""
        total_deficit = sum(
            v.get('deficit', 0)
            for v in optimized.values()
        )
        
        if total_deficit == 0:
            return
        
        # 从数据充足的领域按比例补充
        surplus_domains = [
            d for d, v in optimized.items()
            if v['status'] == 'sufficient'
        ]
        
        total_surplus = sum(
            optimized[d]['available_tokens'] - optimized[d]['use_tokens']
            for d in surplus_domains
        )
        
        for domain in surplus_domains:
            surplus = optimized[domain]['available_tokens'] - optimized[domain]['use_tokens']
            additional = int(total_deficit * (surplus / total_surplus))
            optimized[domain]['use_tokens'] += additional
    
    def visualize_mixture(self, optimized):
        """可视化数据配比"""
        import matplotlib.pyplot as plt
        
        domains = list(optimized.keys())
        tokens = [v['use_tokens'] for v in optimized.values()]
        ratios = [v['use_tokens'] / self.target_tokens * 100 for v in optimized.values()]
        
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
        
        # 饼图
        ax1.pie(tokens, labels=domains, autopct='%1.1f%%', startangle=90)
        ax1.set_title('Data Mixture Distribution')
        
        # 柱状图
        ax2.bar(domains, ratios)
        ax2.set_xlabel('Domain')
        ax2.set_ylabel('Percentage (%)')
        ax2.set_title('Domain Ratio')
        ax2.tick_params(axis='x', rotation=45)
        
        plt.tight_layout()
        plt.savefig('data_mixture.png', dpi=150)
        plt.show()
```

### 5.2 质量 vs 数量

在预训练数据准备中,质量与数量的平衡是一个关键决策。

#### 数据策略对比

| 策略 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| 高质量精选 | 训练效率高、模型质量更好 | 数据收集成本高、可能丢失多样性 | 小规模模型、专业领域 |
| 大规模原始 | 覆盖广泛、成本低 | 需要更强过滤、训练效率低 | 超大规模模型、通用场景 |
| 混合策略 | 平衡质量和多样性 | 需要精心设计配比 | 大多数场景 |

```python
class DataQualityQuantityTradeoff:
    """质量与数量权衡分析器"""
```

```python
from transformers import MarianMTModel, MarianTokenizer

class BackTranslationAugmentor:
    """回译数据增强器"""
    
    def __init__(self, pivot_languages=['de', 'fr', 'es']):
        """
        Args:
            pivot_languages: 中间语言列表
        """
        self.pivot_languages = pivot_languages
        self.models = {}
        self.tokenizers = {}
        
        # 加载翻译模型
        self._load_models()
    
    def _load_models(self):
        """加载翻译模型"""
        for lang in self.pivot_languages:
            # 英语 -> 中间语言
            en_to_lang = f'Helsinki-NLP/opus-mt-en-{lang}'
            self.models[f'en_{lang}'] = MarianMTModel.from_pretrained(en_to_lang)
            self.tokenizers[f'en_{lang}'] = MarianTokenizer.from_pretrained(en_to_lang)
            
            # 中间语言 -> 英语
            lang_to_en = f'Helsinki-NLP/opus-mt-{lang}-en'
            self.models[f'{lang}_en'] = MarianMTModel.from_pretrained(lang_to_en)
            self.tokenizers[f'{lang}_en'] = MarianTokenizer.from_pretrained(lang_to_en)
    
    def translate(self, text, src_lang, tgt_lang):
        """翻译文本"""
        model_key = f'{src_lang}_{tgt_lang}'
        model = self.models[model_key]
        tokenizer = self.tokenizers[model_key]
        
        # 编码
        inputs = tokenizer(text, return_tensors='pt', padding=True, truncation=True)
        
        # 翻译
        translated = model.generate(**inputs)
        
        # 解码
        return tokenizer.decode(translated[0], skip_special_tokens=True)
    
    def augment(self, text, num_variants=3):
        """生成增强变体"""
        variants = []
        
        for pivot_lang in self.pivot_languages[:num_variants]:
            try:
                # 原文 -> 中间语言 -> 原文
                pivot_text = self.translate(text, 'en', pivot_lang)
                back_translated = self.translate(pivot_text, pivot_lang, 'en')
                
                if back_translated and back_translated != text:
                    variants.append({
                        'text': back_translated,
                        'pivot_language': pivot_lang,
                        'method': 'back_translation'
                    })
            except Exception as e:
                print(f"Back translation failed for {pivot_lang}: {e}")
        
        return variants
    
    def augment_batch(self, texts, num_variants=3):
        """批量增强"""
        all_variants = []
        
        for text in texts:
            variants = self.augment(text, num_variants)
            all_variants.extend(variants)
        
        return all_variants

# 使用示例
```

```python
class TextRewritingAugmentor:
    """文本改写增强器"""
    
    def __init__(self, paraphrase_model='tuner007/pegasus_paraphrase'):
        from transformers import PegasusForConditionalGeneration, PegasusTokenizer
        
        self.model = PegasusForConditionalGeneration.from_pretrained(paraphrase_model)
        self.tokenizer = PegasusTokenizer.from_pretrained(paraphrase_model)
    
    def paraphrase(self, text, num_return_sequences=3):
        """改写文本"""
        # Pegasus 期望输入格式
        batch = self.tokenizer(
            [text],
            truncation=True,
            padding='longest',
            max_length=60,
            return_tensors="pt"
        )
        
        # 生成
        translated = self.model.generate(
            **batch,
            max_length=60,
            num_beams=10,
            num_return_sequences=num_return_sequences,
            temperature=1.5
        )
        
        # 解码
        paraphrases = [
            self.tokenizer.decode(t, skip_special_tokens=True)
            for t in translated
        ]
        
        return paraphrases
    
    def synonym_replacement(self, text, replace_ratio=0.1):
        """同义词替换"""
        import nltk
        from nltk.corpus import wordnet
        
        words = nltk.word_tokenize(text)
        replaced_words = []
        
        for word in words:
            if nltk.pos_tag([word])[0][1].startswith('JJ') or \
               nltk.pos_tag([word])[0][1].startswith('RB'):
                # 形容词或副词
                if random.random() < replace_ratio:
                    synonyms = wordnet.synsets(word)
                    if synonyms:
                        synonym = synonyms[0].lemmas()[0].name()
                        replaced_words.append(synonym)
                        continue
            
            replaced_words.append(word)
        
        return ' '.join(replaced_words)
    
    def random_insertion(self, text, insert_ratio=0.05):
        """随机插入同义词"""
        import nltk
        from nltk.corpus import wordnet
        
        words = text.split()
        num_insertions = int(len(words) * insert_ratio)
        
        for _ in range(num_insertions):
            idx = random.randint(0, len(words) - 1)
            word = words[idx]
            
            synonyms = wordnet.synsets(word)
            if synonyms:
                synonym = synonyms[0].lemmas()[0].name()
                words.insert(idx, synonym)
        
        return ' '.join(words)
    
    def random_swap(self, text, swap_ratio=0.1):
        """随机交换词序"""
        words = text.split()
        num_swaps = int(len(words) * swap_ratio)
        
        for _ in range(num_swaps):
            idx1, idx2 = random.sample(range(len(words)), 2)
            words[idx1], words[idx2] = words[idx2], words[idx1]
        
        return ' '.join(words)
    
    def random_deletion(self, text, delete_ratio=0.1):
        """随机删除词"""
        words = text.split()
        
        if len(words) <= 2:
            return text
        
        num_deletions = int(len(words) * delete_ratio)
        
        for _ in range(num_deletions):
            if len(words) > 2:
                idx = random.randint(0, len(words) - 1)
                words.pop(idx)
        
        return ' '.join(words)

# 使用示例
def augment_with_eda(texts, num_augmented_per_text=5):
    """使用 EDA（Easy Data Augmentation）增强数据"""
    augmentor = TextRewritingAugmentor()
    
    augmented_data = []
    
    for text in texts:
        variants = []
        
        # 改写
        if random.random() < 0.5:
            paraphrases = augmentor.paraphrase(text, num_return_sequences=2)
            variants.extend(paraphrases)
        
        # 同义词替换
        if random.random() < 0.3:
            variant = augmentor.synonym_replacement(text, replace_ratio=0.1)
            variants.append(variant)
        
        # 随机插入
        if random.random() < 0.2:
            variant = augmentor.random_insertion(text, insert_ratio=0.05)
            variants.append(variant)
        
        # 随机交换
        if random.random() < 0.2:
            variant = augmentor.random_swap(text, swap_ratio=0.1)
            variants.append(variant)
        
        # 随机删除
        if random.random() < 0.2:

            variants.append(variant)
        
        # 选择前 N 个变体
        augmented_data.extend(variants[:num_augmented_per_text])
```

```python
class SelfInstructDataGenerator:

        with open(seed_file, 'r') as f:
            for line in f:
                tasks.append(json.loads(line))

    def generate_instructions(
        self,
        num_instructions=1000,
        batch_size=10
    ):
        """生成指令"""
        generated_instructions = []
        
        prompt_template = """
Based on the following example tasks, generate {num} new tasks.
The tasks should be diverse and cover different domains.

Example tasks:
{examples}

New tasks (one per line):
"""
        
        num_batches = num_instructions // batch_size
        
        for i in range(num_batches):
            # 随机选择示例任务
            examples = random.sample(self.seed_tasks, min(5, len(self.seed_tasks)))
            examples_text = '\n'.join([t['instruction'] for t in examples])
            
            # 生成新指令
            prompt = prompt_template.format(
                num=batch_size,
                examples=examples_text
            )
            
            response = self.llm_client.generate(prompt, max_tokens=500)
            new_instructions = response.strip().split('\n')
            
            for instruction in new_instructions:
                if instruction.strip():
                    generated_instructions.append({
                        'instruction': instruction.strip(),
                        'input': '',
                        'output': '',
                        'source': 'self_instruct'
                    })
            
            if (i + 1) % 10 == 0:
                print(f"Generated {len(generated_instructions)} instructions")
        
        return generated_instructions

        """生成输入输出对"""
        if num_pairs:
            instructions = random.sample(instructions, min(num_pairs, len(instructions)))
        
        instruction_output_pairs = []
        
        for instruction in instructions:
            prompt = f"""
Given the following instruction, generate a realistic input and output.

Instruction: {instruction['instruction']}

Input: (if applicable, otherwise leave empty)
"""
            
            input_response = self.llm_client.generate(prompt, max_tokens=200)
            input_text = input_response.strip()
            
            # 生成输出

Instruction: {instruction['instruction']}
Input: {input_text}

Output:
"""
            
            output_response = self.llm_client.generate(prompt2, max_tokens=500)
            output_text = output_response.strip()
            
            instruction_output_pairs.append({
                'instruction': instruction['instruction'],
                'input': input_text if input_text else '',
                'output': output_text,
                'source': 'self_instruct'
            })
        
        return instruction_output_pairs

        instructions = self.generate_instructions(num_instructions)
        
        print("Generating input-output pairs...")
        pairs = self.generate_input_output_pairs(instructions, num_pairs)
        
        return pairs
```

```python
                prompt = self._create_concretizing_prompt(current_instruction)
            elif evolution_type == 'reasoning':
                prompt = self._create_reasoning_prompt(current_instruction)
            else:
                prompt = self._create_breadth_prompt(current_instruction)
            
            response = self.llm_client.generate(prompt, max_tokens=300)
            evolved_instruction = response.strip()
            
            if evolved_instruction:
                evolved.append(evolved_instruction)
        
        return evolved
    
    def _create_breadth_prompt(self, instruction):
        """广度演化：增加新维度"""
        return f"""
Please make the following instruction more complex by adding new requirements or constraints.

Original instruction: {instruction}

Evolved instruction:
"""
    
    def _create_depth_prompt(self, instruction):
        """深度演化：增加深度"""
        return f"""
Please make the following instruction more in-depth and challenging.

Original instruction: {instruction}

Evolved instruction:
"""
    
    def _create_concretizing_prompt(self, instruction):
        """具体化：添加具体场景"""
        return f"""
Please make the following instruction more concrete by adding a specific scenario or context.

Original instruction: {instruction}

Evolved instruction:
"""
    
    def _create_reasoning_prompt(self, instruction):
        """推理化：增加推理要求"""
        return f"""
Please modify the following instruction to require more logical reasoning or multi-step thinking.

Original instruction: {instruction}

Evolved instruction:
"""
    
    def generate_evoluted_dataset(
        self,
        seed_instructions,
        evolutions_per_instruction=5,
        evolution_types=['breadth', 'depth', 'concretizing', 'reasoning']
    ):
        """生成演化数据集"""
        all_instructions = []
        
        for seed in seed_instructions:
            # 对每种演化类型进行演化
            for evo_type in evolution_types:
                evolved = self.evolve_instruction(
                    seed,
                    evolution_type=evo_type,
                    num_evolutions=evolutions_per_instruction
                )
                
                for instruction in evolved:
                    all_instructions.append({
                        'instruction': instruction,
                        'evolution_type': evo_type,
                        'source': 'evol_instruct'
                    })
        
        return all_instructions
    
    def generate_responses(self, instructions):
        """为指令生成响应"""
        instruction_output_pairs = []
        
        for item in instructions:
            prompt = f"""
{item['instruction']}

Your response:
"""
            
            response = self.llm_client.generate(prompt, max_tokens=1000)
            
            instruction_output_pairs.append({
                'instruction': item['instruction'],
                'input': '',
                'output': response.strip(),
                'evolution_type': item['evolution_type'],
                'source': 'evol_instruct'
            })
        
        return instruction_output_pairs
```

```python
class PreferenceDataGenerator:
    """偏好数据生成器（用于 RLHF/DPO）"""
    
    def __init__(self, llm_client):
        self.llm_client = llm_client
    
    def generate_preference_pairs(
        self,
        prompts,
        num_responses_per_prompt=4
    ):
        """生成偏好数据对"""
        preference_data = []
        
        for prompt in prompts:
            # 生成多个响应
            responses = []
            for i in range(num_responses_per_prompt):
                response = self.llm_client.generate(
                    prompt,
                    max_tokens=500,
                    temperature=0.7 + i * 0.1  # 递增温度以获得多样性
                )
                responses.append(response.strip())
            
            # 对响应排序
            ranked_responses = self._rank_responses(prompt, responses)
            
            # 构建偏好对
            for i in range(len(ranked_responses)):
                for j in range(i + 1, len(ranked_responses)):
                    preference_data.append({
                        'prompt': prompt,
                        'chosen': ranked_responses[i]['response'],
                        'rejected': ranked_responses[j]['response'],
                        'chosen_rank': ranked_responses[i]['rank'],
                        'rejected_rank': ranked_responses[j]['rank']
                    })
        
        return preference_data
    
    def _rank_responses(self, prompt, responses):
        """对响应排序"""
        ranking_prompt = f"""
Given the following prompt and multiple responses, rank them from best to worst.
Consider: helpfulness, accuracy, completeness, and clarity.

Prompt: {prompt}

Responses:
"""
        
        for i, response in enumerate(responses):
            ranking_prompt += f"\nResponse {i+1}:\n{response}\n"
        
        ranking_prompt += "\nRanking (best to worst, e.g., 2, 1, 4, 3):\n"
        
        ranking_response = self.llm_client.generate(ranking_prompt, max_tokens=100)
        
        # 解析排名
        try:
            ranking = [int(x) for x in ranking_response.strip().split(',')]
            ranked = []
            for rank, idx in enumerate(ranking, 1):
                ranked.append({
                    'response': responses[idx - 1],
                    'rank': rank
                })
            return sorted(ranked, key=lambda x: x['rank'])
        except:
            # 如果解析失败，随机排序
            return [
                {'response': r, 'rank': i}
                for i, r in enumerate(responses, 1)
            ]
    
    def generate_with_critic_model(
        self,
        prompts,
        critic_model,
        num_responses_per_prompt=2
    ):
        """使用评判模型生成偏好数据"""
        preference_data = []
        
        for prompt in prompts:
            responses = []
            
            # 生成响应
            for _ in range(num_responses_per_prompt):
                response = self.llm_client.generate(
                    prompt,
                    max_tokens=500,
                    temperature=0.7
                )
                responses.append(response.strip())
            
            # 评判模型评分
            scores = []
            for response in responses:
                score = critic_model.evaluate(prompt, response)
                scores.append(score)
            
            # 排序
            ranked = sorted(
                zip(responses, scores),
                key=lambda x: x[1],
                reverse=True
            )
            
            # 构建偏好对
            for i in range(len(ranked)):
                for j in range(i + 1, len(ranked)):
                    preference_data.append({
                        'prompt': prompt,
                        'chosen': ranked[i][0],
                        'rejected': ranked[j][0],
                        'chosen_score': ranked[i][1],
                        'rejected_score': ranked[j][1]
                    })
        
        return preference_data
```

```python
class CompletePretrainingPipeline:
    """完整的预训练数据管道"""
    
    def __init__(self, config):
        self.config = config
        
        # 初始化各个组件
        self.data_collector = DataCollector(config['data_sources'])
        self.cleaner = DataCleaner(config['cleaning_config'])
        self.deduplicator = Deduplicator(config['dedup_config'])
        self.quality_evaluator = QualityEvaluator(config['quality_config'])
        self.tokenizer = PretrainingTokenizer(config['tokenizer'])
        self.packer = SequencePacker(config['max_seq_length'])
    
    def run(self, output_dir):
        """运行完整管道"""
        print("=" * 60)
        print("STARTING PRETRAINING DATA PIPELINE")
        print("=" * 60)
        
        # 阶段 1：数据收集
        print("\n[1/7] Data Collection...")
        raw_data = self.data_collector.collect()
        print(f"Collected {len(raw_data)} documents")
        
        # 阶段 2：数据清洗
        print("\n[2/7] Data Cleaning...")
        cleaned_data = self.cleaner.clean(raw_data)
        print(f"Cleaned: {len(cleaned_data)} documents")
        
        # 阶段 3：去重
        print("\n[3/7] Deduplication...")
        deduplicated_data = self.deduplicator.deduplicate(cleaned_data)
        print(f"After dedup: {len(deduplicated_data)} documents")
        
        # 阶段 4：质量评估
        print("\n[4/7] Quality Evaluation...")
        quality_filtered, quality_stats = self.quality_evaluator.filter(deduplicated_data)
        print(f"High quality: {len(quality_filtered)} documents")
        
        # 阶段 5：词元化
        print("\n[5/7] Tokenization...")
        tokenized_data, token_stats = self.tokenizer.tokenize(quality_filtered)
        print(f"Tokenized: {len(tokenized_data)} sequences")
        
        # 阶段 6：打包
        print("\n[6/7] Sequence Packing...")
        packed_data = self.packer.pack(tokenized_data)
        print(f"Packed: {len(packed_data)} sequences")
        
        # 阶段 7：保存
        print("\n[7/7] Saving...")
        self._save(packed_data, output_dir)
        
        print("\n" + "=" * 60)
        print("PIPELINE COMPLETED SUCCESSFULLY")
        print("=" * 60)
        
        return {
            'raw_count': len(raw_data),
            'final_count': len(packed_data),
            'retention_rate': len(packed_data) / len(raw_data)
        }
    
    def _save(self, data, output_dir):
        """保存数据"""
        import json
        
        # 保存处理后的数据
        output_file = os.path.join(output_dir, 'pretraining_data.json')
        with open(output_file, 'w') as f:
            json.dump(data, f)
        
        # 保存统计信息
        stats_file = os.path.join(output_dir, 'pipeline_stats.json')
        with open(stats_file, 'w') as f:
            json.dump(self.get_statistics(), f, indent=2)
        
        print(f"Data saved to {output_file}")
        print(f"Statistics saved to {stats_file}")
```
