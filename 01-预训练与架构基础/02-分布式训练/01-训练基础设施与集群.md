```python
        # 大规模预训练
        if budget > 500000:
            return "H100 集群 (8+ 卡)"
        else:
            return "A100 集群 (8+ 卡)"

        if budget > 50000:
            return "H100 (单卡或多卡)"
        else:
            return "A100 80GB (单卡或多卡)"
    
    elif task == "finetuning" and scale == "small":
        # 小规模微调（QLoRA）
```

### 2.3 自动化集群管理脚本

```python
"""
GPU 集群自动化管理工具
"""

import subprocess
import json
from typing import List, Dict

class ClusterManager:
    """集群管理器"""
    
    def __init__(self, cluster_type="slurm"):
        self.cluster_type = cluster_type
    
    def get_available_gpus(self) -> Dict:
        """获取可用 GPU 信息"""
        if self.cluster_type == "slurm":
            return self._slurm_get_gpus()
        elif self.cluster_type == "kubernetes":
            return self._k8s_get_gpus()
    
    def _slurm_get_gpus(self):
        """Slurm 集群 GPU 查询"""
        result = subprocess.run(
            ["sinfo", "--json"],
            capture_output=True,
            text=True
        )
        info = json.loads(result.stdout)
        
        gpus_info = {}
        for node in info['nodes']:
            gpus_info[node['name']] = {
                "gpus_total": node['gres']['gpu'],
                "gpus_used": node['alloc_cpus'],
                "state": node['state']
            }
        
        return gpus_info
    
    def submit_job(self, config: Dict) -> str:
        """提交训练作业"""
        if self.cluster_type == "slurm":
            return self._slurm_submit(config)
    
    def _slurm_submit(self, config: Dict) -> str:
        """Slurm 作业提交"""
        
        # 生成 Slurm 脚本
        script = self._generate_slurm_script(config)
        
        # 提交作业
        result = subprocess.run(
            ["sbatch", script],
            capture_output=True,
            text=True
        )
        
        job_id = result.stdout.strip().split()[-1]
        return job_id
    
    def _generate_slurm_script(self, config: Dict) -> str:
        """生成 Slurm 脚本"""
        
        script = f"""#!/bin/bash
#SBATCH --job-name={config['job_name']}
#SBATCH --nodes={config['nodes']}
#SBATCH --ntasks-per-node=1
#SBATCH --gres=gpu:{config['gpus_per_node']}
#SBATCH --cpus-per-task={config['cpus_per_task']}
#SBATCH --mem={config['memory']}
#SBATCH --time={config['time']}
#SBATCH --partition={config.get('partition', 'gpu')}
#SBATCH --output={config.get('output', 'training_%j.log')}

module load cuda/12.1
module load python/3.10

source {config['venv']}/bin/activate

export MASTER_ADDR=$(scontrol show hostname "$SLURM_NODEID" | head -n 1)
export MASTER_PORT=29500

srun python -m torch.distributed.run \\
    --nnodes=$SLURM_NNODES \\
    --nproc_per_node={config['gpus_per_node']} \\
    --rdzv_id=$SLURM_JOB_ID \\
    --rdzv_backend=c10d \\
    --rdzv_endpoint=$MASTER_ADDR:$MASTER_PORT \\
    {config['script']} {config.get('args', '')}
"""
        
        # 写入文件
        script_path = f"/tmp/slurm_{config['job_name']}.sh"
        with open(script_path, 'w') as f:
            f.write(script)
        
        return script_path
    
    def monitor_job(self, job_id: str):
        """监控作业状态"""
        result = subprocess.run(
            ["squeue", "--job", job_id, "--json"],
            capture_output=True,
            text=True
        )
        
        info = json.loads(result.stdout)
        return info
    
    def cancel_job(self, job_id: str):
        """取消作业"""
        subprocess.run(["scancel", job_id])
    
    def get_gpu_utilization(self) -> Dict:
        """获取 GPU 利用率"""
        result = subprocess.run(
            ["nvidia-smi", "--query-gpu=index,utilization.gpu,memory.used,memory.total",
             "--format=csv,noheader,nounits"],
            capture_output=True,
            text=True
        )
        
        utilization = {}
        for line in result.stdout.strip().split('\n'):
            gpu_id, util, mem_used, mem_total = map(float, line.split(','))
            utilization[int(gpu_id)] = {
                "utilization": util,
                "memory_used_gb": mem_used / 1024,
                "memory_total_gb": mem_total / 1024,
                "memory_usage": mem_used / mem_total
            }
        
        return utilization

# 使用示例
def manage_cluster():
    manager = ClusterManager("slurm")
    
    # 查看可用 GPU
    gpus = manager.get_available_gpus()
    print(f"可用 GPU: {gpus}")

    }
    
    job_id = manager.submit_job(config)

    # 监控作业

    print(f"作业状态: {status}")
    
    # 监控 GPU 利用率
    utilization = manager.get_gpu_utilization()
    print(f"GPU 利用率: {utilization}")
```

```python
        """
        计算训练成本
        
        config: dict
            - model_size: 模型参数量
            - training_tokens: 训练 token 数
            - gpu_type: GPU 类型
            - num_gpus: GPU 数量
            - mfu: 模型 FLOPS 利用率 (0.0-1.0)
        """

        training_time_hours = training_time_seconds / 3600

        total_cost = training_time_hours * config['num_gpus'] * gpu_price
```

```python
            repeat=1
        ),
        on_trace_ready=torch.profiler.tensorboard_trace_handler('./logs'),
        record_shapes=True,
        profile_memory=True,
        with_stack=True
    ) as prof:
        
        for step in range(100):
            with record_function("train_step"):
                loss = train_step()
                loss.backward()
                optimizer.step()
            
            prof.step()
    
    # 输出分析报告
    print(prof.key_averages().table(
        sort_by="cuda_time_total",

# 瓶颈分析
def analyze_bottleneck():
    """分析训练瓶颈"""
    
    import time
    
    # 时间分解
    timings = {
        "data_loading": 0,
        "forward_pass": 0,
        "backward_pass": 0,
        "optimizer_step": 0,
        "communication": 0
    }
    
    for batch in data_loader:
        # 数据加载
        start = time.time()
        batch = load_batch()
        timings["data_loading"] += time.time() - start
        
        # 前向传播
        start = time.time()
        loss = model(batch)
        timings["forward_pass"] += time.time() - start
        
        # 反向传播
        start = time.time()
        loss.backward()
        timings["backward_pass"] += time.time() - start
        
        # 优化器更新
        start = time.time()
        optimizer.step()
        timings["optimizer_step"] += time.time() - start
    
    # 分析
    total_time = sum(timings.values())
    for key, value in timings.items():
        percentage = value / total_time * 100
        print(f"{key}: {percentage:.1f}%")
    
    # 识别瓶颈
    bottleneck = max(timings, key=timings.get)
    print(f"\n主要瓶颈: {bottleneck}")
```
