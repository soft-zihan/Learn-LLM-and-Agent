## 1. 训练框架概览与选型

### 1.1 主流框架对比

| 框架 | 定位 | 核心优势 | 适用场景 | 学习曲线 |
|------|------|----------|----------|----------|
| **Axolotl** | 配置驱动微调 | YAML 配置、多框架支持、生产级 | 快速微调原型 | 中 |
| **LLaMA-Factory** | 统一训练平台 | Web UI、零代码、多模型 | 可视化训练、教学 | 低 |
| **OpenRLHF** | 大规模 RLHF | Ray 分布式、70B+ 支持 | 大规模对齐训练 | 高 |
| **veRL (verl)** | HybridFlow RL | 混合架构、GRPO、数学推理 | RL 后训练 | 高 |
| **Unsloth** | 速度优化 | 2-5 倍加速、低显存 | 单卡/小集群微调 | 低 |
| **TRL** | HuggingFace RLHF | 官方支持、文档完善 | PPO/DPO 训练 | 中 |
| **DeepSpeed-Chat** | 微软 RLHF | DeepSpeed 优化 | 企业级训练 | 高 |

### 1.2 选型决策树

```python
def choose_training_framework(requirements):
    """
    根据需求选择训练框架
    
    Args:
        requirements: 需求字典
            - task: 任务类型 (sft, rlhf, dpo, continued_pretrain)
            - model_size: 模型参数量 (7B, 13B, 70B, etc.)
            - gpu_count: GPU 数量
            - gpu_memory: 单卡显存 (GB)
            - ease_of_use: 易用性要求 (low, medium, high)
            - need_ui: 是否需要 Web UI
    """
    task = requirements['task']
    model_size = requirements['model_size']
    gpu_count = requirements['gpu_count']
    need_ui = requirements.get('need_ui', False)
    
    # 决策逻辑
    if need_ui:
        return "LLaMA-Factory"
    
    if task in ['rlhf', 'ppo'] and model_size >= '70B':
        if gpu_count >= 8:
            return "OpenRLHF"
        else:
            return "veRL"
    
    if task == 'dpo':
        if model_size <= '13B':
            return "Unsloth"  # 速度最快
        else:
            return "Axolotl"
    
    if task == 'sft':
        if gpu_count == 1:
            return "Unsloth"  # 单卡最优
        elif gpu_count <= 4:
            return "Axolotl"  # 多卡方便
        else:
            return "LLaMA-Factory"  # 大规模训练
    
    if task == 'continued_pretrain':
        return "Axolotl"
    
    return "TRL"  # 默认选择
```

## 2. Axolotl 实战

### 2.1 框架简介

**Axolotl** 是一个配置驱动的 LLM 微调框架，核心特点：

- **YAML 配置驱动**：所有训练参数通过 YAML 文件配置
- **多框架后端**：支持 DeepSpeed、FSDP、Accelerate
- **生产级特性**：支持 WANDB 监控、梯度检查点、多种 PEFT 方法
- **社区活跃**：GitHub 10K+ Stars，持续更新

GitHub: https://github.com/OpenAccess-AI-Collective/axolotl

### 2.5 数据准备脚本

```python
# scripts/prepare-data.py
import json
import pandas as pd
from datasets import Dataset

def prepare_alpaca_data(input_file, output_file):
    """准备 Alpaca 格式数据"""
    # 1. 读取原始数据
    df = pd.read_csv(input_file)
    
    # 2. 转换为 Alpaca 格式
    alpaca_data = []
    for _, row in df.iterrows():
        alpaca_data.append({
            "instruction": row['question'],
            "input": row.get('context', ''),
            "output": row['answer']
        })
    
    # 3. 保存为 JSONL
    with open(output_file, 'w', encoding='utf-8') as f:
        for item in alpaca_data:
            f.write(json.dumps(item, ensure_ascii=False) + '\n')
    
    print(f"已保存 {len(alpaca_data)} 条数据到 {output_file}")

def prepare_chatml_data(input_file, output_file):
    """准备 ChatML 格式数据"""
    df = pd.read_csv(input_file)
    
    chatml_data = []
    for _, row in df.iterrows():
        chatml_data.append({
            "messages": [
                {"role": "system", "content": row.get('system_prompt', '你是有用的助手')},
                {"role": "user", "content": row['question']},
                {"role": "assistant", "content": row['answer']}
            ]
        })
    
    with open(output_file, 'w', encoding='utf-8') as f:
        for item in chatml_data:
            f.write(json.dumps(item, ensure_ascii=False) + '\n')

def prepare_dpo_data(input_file, output_file):
    """准备 DPO 偏好对数据"""
    df = pd.read_csv(input_file)
    
    dpo_data = []
    for _, row in df.iterrows():
        dpo_data.append({
            "prompt": row['prompt'],
            "chosen": row['chosen_response'],
            "rejected": row['rejected_response']
        })
    
    with open(output_file, 'w', encoding='utf-8') as f:
        for item in dpo_data:
            f.write(json.dumps(item, ensure_ascii=False) + '\n')

# 使用示例
if __name__ == "__main__":
    prepare_alpaca_data("data/raw/train.csv", "data/train.jsonl")
    prepare_alpaca_data("data/raw/test.csv", "data/test.jsonl")
```

#### 模型测试脚本

```python
# scripts/test-model.py
from transformers import AutoModelForCausalLM, AutoTokenizer
import torch

def test_model(model_path, test_prompts):
    """测试微调后的模型"""
    # 1. 加载模型
    tokenizer = AutoTokenizer.from_pretrained(model_path)
    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        device_map="auto",
        torch_dtype=torch.bfloat16
    )
    
    # 2. 测试提示词
    results = []
    for prompt in test_prompts:
        # 3. 编码
        inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
        
        # 4. 生成
        with torch.no_grad():
            outputs = model.generate(
                **inputs,
                max_new_tokens=512,
                temperature=0.7,
                top_p=0.9,
                do_sample=True
            )
        
        # 5. 解码
        response = tokenizer.decode(outputs[0], skip_special_tokens=True)
        results.append({
            "prompt": prompt,
            "response": response
        })
        
        print(f"\n{'='*80}")
        print(f"提示词：{prompt}")
        print(f"{'='*80}")
        print(f"回答：{response}")
        print(f"{'='*80}\n")
    
    return results

# 测试示例
test_prompts = [
    "解释量子计算的基本原理",
    "用 Python 实现快速排序算法",
    "写一首关于春天的诗"
]

if __name__ == "__main__":
    import sys
    model_path = sys.argv[1]
    test_model(model_path, test_prompts)
```

### 4.7 实战案例：RLHF 对齐训练

#### 完整流程

```python
# scripts/rlhf_pipeline.py
import subprocess
import json

class RLHFPipeline:
    """RLHF 训练流水线"""
    
    def __init__(self, config):
        self.config = config
    
    def step1_generate_samples(self):
        """步骤 1：使用 SFT 模型生成样本"""
        cmd = f"""
        openrlhf batch_inference \
          --model_path {self.config['sft_model']} \
          --prompt_data {self.config['prompt_file']} \
          --output_data {self.config['generated_samples']} \
          --max_len 2048 \
          --temperature 0.7 \
          --num_return_sequences 2
        """
        subprocess.run(cmd, shell=True)
    
    def step2_score_samples(self):
        """步骤 2：奖励模型打分"""
        cmd = f"""
        openrlhf batch_inference \
          --model_path {self.config['reward_model']} \
          --input_data {self.config['generated_samples']} \
          --output_data {self.config['scored_samples']} \
          --task reward
        """
        subprocess.run(cmd, shell=True)
    
    def step3_create_dpo_pairs(self):
        """步骤 3：创建 DPO 偏好对"""
        # 根据分数选择 chosen/rejected
        with open(self.config['scored_samples'], 'r') as f:
            samples = json.load(f)
        
        dpo_pairs = []
        for sample in samples:
            if len(sample['responses']) >= 2:
                # 选择高分作为 chosen，低分作为 rejected
                sorted_responses = sorted(
                    sample['responses'],
                    key=lambda x: x['score'],
                    reverse=True
                )
                dpo_pairs.append({
                    "prompt": sample['prompt'],
                    "chosen": sorted_responses[0]['text'],
                    "rejected": sorted_responses[-1]['text']
                })
        
        with open(self.config['dpo_pairs'], 'w') as f:
            json.dump(dpo_pairs, f, ensure_ascii=False, indent=2)
    
    def step4_train_dpo(self):
        """步骤 4：DPO 训练"""
        cmd = f"""
        openrlhf train_dpo_ray \
          --ref_num_nodes 1 \
          --ref_num_gpus_per_node {self.config['gpu_count']} \
          --actor_num_nodes 1 \
          --actor_num_gpus_per_node {self.config['gpu_count']} \
          --ref_pretrain {self.config['sft_model']} \
          --actor_pretrain {self.config['sft_model']} \
          --save_path {self.config['output_dir']} \
          --dataset {self.config['dpo_pairs']} \
          --bf16 \
          --learning_rate 5e-7
        """
        subprocess.run(cmd, shell=True)
    
    def run(self):
        """运行完整流水线"""
        print("步骤 1：生成样本...")
        self.step1_generate_samples()
        
        print("步骤 2：奖励模型打分...")
        self.step2_score_samples()
        
        print("步骤 3：创建 DPO 偏好对...")
        self.step3_create_dpo_pairs()
        
        print("步骤 4：DPO 训练...")
        self.step4_train_dpo()
        
        print("RLHF 训练完成！")

# 使用示例
config = {
    'sft_model': 'output/llama3-sft',
    'reward_model': 'OpenAssistant/reward-model',
    'prompt_file': 'data/prompts.jsonl',
    'generated_samples': 'data/generated.json',
    'scored_samples': 'data/scored.json',
    'dpo_pairs': 'data/dpo_pairs.json',
    'output_dir': 'output/llama3-rlhf',
    'gpu_count': 4
}

pipeline = RLHFPipeline(config)
pipeline.run()
```

### 5.3 GRPO 算法实现

#### GRPO 原理

Group Relative Policy Optimization (GRPO) 是 2025 年在数学推理任务上表现优异的 RL 算法：

```python
def grpo_loss(policy_logits, ref_logits, rewards, group_size=4):
    """
    GRPO 损失计算
    
    Args:
        policy_logits: 策略模型 logits
        ref_logits: 参考模型 logits
        rewards: 奖励信号
        group_size: 组大小
    """
    # 1. 计算优势（组内相对优势）
    rewards_grouped = rewards.reshape(-1, group_size)
    mean_reward = rewards_grouped.mean(dim=1, keepdim=True)
    std_reward = rewards_grouped.std(dim=1, keepdim=True)
    advantages = (rewards - mean_reward) / (std_reward + 1e-8)
    
    # 2. 计算策略比率
    policy_log_probs = log_probs_from_logits(policy_logits)
    ref_log_probs = log_probs_from_logits(ref_logits)
    ratio = torch.exp(policy_log_probs - ref_log_probs)
    
    # 3. 裁剪
    eps = 0.2
    ratio_clipped = torch.clamp(ratio, 1 - eps, 1 + eps)
    
    # 4. 损失
    loss_surrogate = -torch.min(
        ratio * advantages,
        ratio_clipped * advantages
    )
    
    # 5. KL 惩罚
    kl_penalty = kl_divergence(policy_logits, ref_logits)
    
    # 6. 总损失
    beta = 0.01
    total_loss = loss_surrogate.mean() + beta * kl_penalty.mean()
    
    return total_loss
```

```python
        train_dataset=dataset,
        dataset_text_field="text",
        max_seq_length=4096,
        args=training_args,
    )
    
    trainer.train()
    
    # 6. 保存
    print("保存模型...")
    model.save_pretrained("output/qwen2-sft-lora")
    tokenizer.save_pretrained("output/qwen2-sft-lora")
    
    print("训练完成！")

if __name__ == "__main__":
    main()
```

### 7.3 选型决策树

```python
def final_framework_selection(task, resources, experience):
    """
    最终框架选择决策
    
    Args:
        task: 任务类型
        resources: 资源情况
        experience: 经验水平
    """
    # 新手或需要可视化
    if experience == "beginner" or resources.get('need_ui'):
        return "LLaMA-Factory"
    
    # 单卡或显存受限
    if resources['gpu_count'] == 1:
        return "Unsloth"
    
    # 大规模 RLHF
    if task in ['ppo', 'rlhf'] and resources['gpu_count'] >= 4:
        if resources['model_size'] >= 70:
            return "OpenRLHF"
        else:
            return "veRL"
    
    # DPO 对齐
    if task == 'dpo':
        if resources['gpu_count'] == 1:
            return "Unsloth"
        else:
            return "Axolotl"
    
    # 标准 SFT
    if task == 'sft':
        if resources['gpu_count'] <= 2:
            return "Unsloth"
        else:
            return "Axolotl"
    
    # 默认
    return "Axolotl"
```
