#### 数据质量过滤

```python
def filter_high_quality(fineweb_dataset, min_score=0.7):
    """过滤高质量数据"""
    high_quality = []
    
    for sample in fineweb_dataset:
        # 1. 基于启发式分数
        if sample['score'] < min_score:
            continue
        
        # 2. 文本长度过滤
        if len(sample['text']) < 200 or len(sample['text']) > 100000:
            continue
        
        # 3. 语言过滤（仅英文）
        if sample['language'] != 'en':
            continue
        
        # 4. 重复度过滤
        if calculate_repetition(sample['text']) > 0.3:
            continue
        
        high_quality.append(sample)
    
    return high_quality

def calculate_repetition(text):
    """计算文本重复度"""
    words = text.split()
    if len(words) < 10:
        return 0
    
    # 4-gram 重复
    n = 4
    ngrams = [tuple(words[i:i+n]) for i in range(len(words)-n+1)]
    unique_ngrams = set(ngrams)
    
    if len(ngrams) == 0:
        return 0
    
    return 1 - len(unique_ngrams) / len(ngrams)
```

```python
    """Evol-Instruct 数据生成管道"""
```

```python
    # 1. 生成多个候选响应
    candidates = generate_multiple_responses(prompt, n=5)
    
    # 2. 专家审核评分
    expert_scores = []
    for candidate in candidates:
        score = expert_review(
            response=candidate,
            criteria=[
                "准确性",
                "完整性",
                "清晰度",
                "安全性"
            ]
        )
        expert_scores.append(score)
    
    # 3. 选择最佳和最差
    best_idx = argmax(expert_scores)
    worst_idx = argmin(expert_scores)
    
    return {
        "prompt": prompt,
        "chosen": candidates[best_idx],
        "rejected": candidates[worst_idx],
        "scores": expert_scores
    }
```

```python
def deduplicate_dataset(dataset, method="minhash"):
    """
    数据集去重
    
    Args:
        dataset: 原始数据集
        method: 去重方法 (minhash, exact, fuzzy)
    """
    if method == "minhash":
        return minhash_deduplication(dataset)
    elif method == "exact":
        return exact_deduplication(dataset)
    elif method == "fuzzy":
        return fuzzy_deduplication(dataset)

def minhash_deduplication(dataset, threshold=0.8):
    """MinHash 近似去重"""
    from datasketch import MinHash, MinHashLSH
    
    # 1. 构建 MinHash
    minhashes = []
    for doc in dataset:
        m = MinHash(num_perm=128)
        for word in doc['text'].split():
            m.update(word.encode('utf8'))
        minhashes.append(m)
    
    # 2. 构建 LSH 索引
    lsh = MinHashLSH(threshold=threshold, num_perm=128)
    unique_docs = []
    
    for i, minhash in enumerate(minhashes):
        # 3. 查询是否重复
        results = lsh.query(minhash)
        if not results:
            # 不重复，添加
            lsh.insert(str(i), minhash)
            unique_docs.append(dataset[i])
    
    return unique_docs

def exact_deduplication(dataset):
    """精确去重"""
    seen = set()
    unique_docs = []
    
    for doc in dataset:
        text_hash = hash(doc['text'])
        if text_hash not in seen:
            seen.add(text_hash)
            unique_docs.append(doc)
    
    return unique_docs
```

#### 去污染（Decontamination）

```python
def decontaminate_training_data(train_dataset, test_datasets):
    """
    训练集去污染
    
    移除训练集中与测试集重叠的样本
    """
    # 1. 构建测试集 n-gram 索引
    test_ngrams = set()
    for test_ds in test_datasets:
        for sample in test_ds:
            ngrams = extract_ngrams(sample['text'], n=13)
            test_ngrams.update(ngrams)
    
    # 2. 过滤训练集
    clean_train = []
    contaminated_count = 0
    
    for sample in train_dataset:
        train_ngrams = extract_ngrams(sample['text'], n=13)
        
        # 检查是否有重叠
        overlap = train_ngrams.intersection(test_ngrams)
        if len(overlap) > 0:
            contaminated_count += 1
            continue  # 移除污染样本
        
        clean_train.append(sample)
    
    print(f"移除污染样本：{contaminated_count}")
    print(f"保留训练样本：{len(clean_train)}")
    
    return clean_train

def extract_ngrams(text, n=13):
    """提取 n-gram"""
    words = text.split()
    ngrams = set()
    
    for i in range(len(words) - n + 1):
        ngram = ' '.join(words[i:i+n])
        ngrams.add(ngram)
    
    return ngrams
```

### 9.2 质量评分标准

#### 启发式质量指标

```python
def calculate_quality_score(document):
    """
    计算文档质量分数
    
    Returns:
        0-1 之间的质量分数
    """
    text = document['text']
    
    scores = {
        # 1. 长度分数
        'length': length_score(text),
        
        # 2. 词汇多样性
        'lexical_diversity': lexical_diversity_score(text),
        
        # 3. 标点符号使用
        'punctuation': punctuation_score(text),
        
        # 4. 句子结构
        'sentence_structure': sentence_structure_score(text),
        
        # 5. 信息密度
        'information_density': information_density_score(text),
        
        # 6. 可读性
        'readability': readability_score(text)
    }
    
    # 加权平均
    weights = {
        'length': 0.15,
        'lexical_diversity': 0.20,
        'punctuation': 0.10,
        'sentence_structure': 0.15,
        'information_density': 0.20,
        'readability': 0.20
    }
    
    total_score = sum(scores[k] * weights[k] for k in scores)
    
    return {
        'total_score': total_score,
        'subscores': scores
    }

def length_score(text):
    """长度分数"""
    length = len(text)
    if length < 100:
        return 0.2
    elif length < 500:
        return 0.5
    elif length < 5000:
        return 1.0
    else:
        return 0.7

def lexical_diversity_score(text):
    """词汇多样性"""
    words = text.split()
    if len(words) == 0:
        return 0
    
    unique_words = set(words)
    return len(unique_words) / len(words)

def information_density_score(text):
    """信息密度"""
    # 计算实体、数字、专有名词密度
    import re
    
    entities = len(re.findall(r'[A-Z][a-z]+', text))
    numbers = len(re.findall(r'\d+', text))
    
    words = len(text.split())
    if words == 0:
        return 0
    
    return (entities + numbers) / words
```

```python
            }
        elif domain == "code":
            return {
                "the_stack_v2": 0.70,
                "fineweb_edu": 0.15,
                "stackoverflow": 0.10,
                "github_docs": 0.05
            }
    
    elif task == "sft":
        return {
            "ultrachat": 0.40,
            "openorca": 0.20,
            "evol_instruct": 0.20,
            "domain_specific": 0.20
        }
    
    elif task == "dpo":
        return {
            "ultrafeedback": 0.40,
            "hh_rlhf": 0.20,
            "nectar": 0.20,
            "domain_preference": 0.20
        }

def mix_datasets(sources, proportions, sample_size=100000):
    """
    混合多个数据集
    
    Args:
        sources: 数据集字典
        proportions: 混合比例
        sample_size: 总样本数
    """
    mixed_data = []
    
    for source_name, proportion in proportions.items():
        n_samples = int(sample_size * proportion)
        dataset = sources[source_name]
        
        # 随机采样
        sampled = dataset.shuffle().select(range(n_samples))
        mixed_data.extend(sampled)
    
    # 打乱混合数据
    random.shuffle(mixed_data)
    
    return mixed_data
```

```python
def process_dataset(dataset):
    """数据处理示例"""

    filtered = dataset.filter(lambda x: len(x['text']) > 200)
    
    # 2. 映射
    processed = filtered.map(

            'length': len(x['text'])
        }
    )
    
    # 3. 分块
    chunked = processed.flat_map(

    )
```
