## 1. API 设计最佳实践

### 1.1 RESTful API

设计良好的 RESTful API 是模型服务化的基础。合理的端点设计、请求/响应格式和错误处理机制能够显著提升用户体验和系统可维护性。

```python
@app.post("/v1/chat/completions", response_model=CompletionResponse)
async def create_chat_completion(request: CompletionRequest):
    """
    创建聊天补全
    
    根据输入的对话消息生成模型响应。支持流式和非流式输出。
    
    **性能指标**:
    - P50 延迟: < 500ms (首 token)
    - P99 延迟: < 2000ms (首 token)
    - 吞吐量: 100+ requests/second
    """
    try:
        # 验证请求
        if not request.messages:
            raise HTTPException(status_code=400, detail="Messages array is required")
        
        # 生成请求 ID
        request_id = f"chatcmpl-{uuid.uuid4().hex[:12]}"
        
        # 记录开始时间
        start_time = time.time()
        
        # 调用模型推理
        if request.stream:
            # 流式输出使用 Server-Sent Events
            return StreamingResponse(
                generate_stream(request, request_id),
                media_type="text/event-stream"
            )
        else:
            # 非流式输出
            result = await generate_completion(request, request_id)
            
            # 计算使用量
            compute_time = time.time() - start_time
            result.usage = CompletionUsage(
                prompt_tokens=result.usage.prompt_tokens,
                completion_tokens=result.usage.completion_tokens,
                total_tokens=result.usage.prompt_tokens + result.usage.completion_tokens,
                tokens_per_second=result.usage.completion_tokens / compute_time
            )
            
            return result
            
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Internal error: {str(e)}")
```

#### 错误处理机制

```python
from enum import Enum
from fastapi.responses import JSONResponse

class ErrorCode(str, Enum):
    """错误代码枚举"""
    INVALID_REQUEST = "invalid_request_error"
    AUTHENTICATION_ERROR = "authentication_error"
    PERMISSION_ERROR = "permission_error"
    NOT_FOUND = "not_found_error"
    RATE_LIMIT_ERROR = "rate_limit_error"
    INTERNAL_ERROR = "internal_error"
    MODEL_ERROR = "model_error"
    VALIDATION_ERROR = "validation_error"
    TIMEOUT_ERROR = "timeout_error"
    OUT_OF_MEMORY = "out_of_memory_error"

class APIError(BaseModel):
    """API 错误响应"""
    error: Dict[str, Any]

def create_error_response(
    status_code: int,
    message: str,
    code: ErrorCode,
    param: Optional[str] = None,
    type: str = "invalid_request_error"
) -> JSONResponse:
    """创建错误响应"""
    error = APIError(
        error={
            "message": message,
            "type": type,
            "param": param,
            "code": code.value
        }
    )
    return JSONResponse(
        status_code=status_code,
        content=error.dict()
    )

# 错误处理中间件
@app.exception_handler(HTTPException)
async def http_exception_handler(request, exc):
    """HTTP 异常处理"""
    return create_error_response(
        status_code=exc.status_code,
        message=exc.detail,
        code=ErrorCode.INTERNAL_ERROR if exc.status_code >= 500 else ErrorCode.INVALID_REQUEST
    )

@app.exception_handler(ValueError)
async def value_error_handler(request, exc):
    """值错误处理"""
    return create_error_response(
        status_code=400,
        message=str(exc),
        code=ErrorCode.VALIDATION_ERROR
    )

@app.exception_handler(TimeoutError)
async def timeout_error_handler(request, exc):
    """超时错误处理"""
    return create_error_response(
        status_code=504,
        message="Request timeout",
        code=ErrorCode.TIMEOUT_ERROR
    )

@app.exception_handler(MemoryError)
async def memory_error_handler(request, exc):
    """内存错误处理"""

        status_code=507,
        message="Out of memory",
        code=ErrorCode.OUT_OF_MEMORY

# 请求验证中间件
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware

class RequestValidationMiddleware(BaseHTTPMiddleware):
    """请求验证中间件"""
    
    async def dispatch(self, request: Request, call_next):
        # 检查 API Key
        api_key = request.headers.get("Authorization")
        if not api_key or not validate_api_key(api_key):
            return create_error_response(
                status_code=401,
                message="Invalid API key",
                code=ErrorCode.AUTHENTICATION_ERROR
            )
        
        # 检查速率限制
        client_ip = request.client.host
        if is_rate_limited(client_ip):
            return create_error_response(
                status_code=429,
                message="Rate limit exceeded",
                code=ErrorCode.RATE_LIMIT_ERROR
            )
        
        # 检查请求大小
        content_length = request.headers.get("content-length")
        if content_length and int(content_length) > 10 * 1024 * 1024:  # 10MB
            return create_error_response(
                status_code=413,
                message="Request too large",
                code=ErrorCode.INVALID_REQUEST
            )
        
        response = await call_next(request)
        return response

app.add_middleware(RequestValidationMiddleware)
```

#### gRPC 服务实现

```python
import grpc
from concurrent import futures
import asyncio

# 生成的 gRPC 代码
import llm_service_pb2
import llm_service_pb2_grpc

class ChatCompletionServicer(llm_service_pb2_grpc.ChatCompletionServiceServicer):
    """gRPC 聊天补全服务"""
    
    def __init__(self, model_engine):
        self.model_engine = model_engine
    
    async def CreateCompletion(self, request, context):
        """创建聊天补全"""
        try:
            # 转换请求
            messages = [
                {"role": msg.role, "content": msg.content}
                for msg in request.messages
            ]
            
            # 调用模型
            result = await self.model_engine.generate(
                model=request.model,
                messages=messages,
                temperature=request.temperature,
                top_p=request.top_p,
                max_tokens=request.max_tokens,
                stop=request.stop if request.stop else None
            )
            
            # 构建响应
            response = llm_service_pb2.CompletionResponse(
                id=result.id,
                object="chat.completion",
                created=result.created,
                model=result.model,
                choices=[
                    llm_service_pb2.Choice(
                        index=choice.index,
                        message=llm_service_pb2.Message(
                            role=choice.message.role,
                            content=choice.message.content
                        ),
                        finish_reason=choice.finish_reason
                    )
                    for choice in result.choices
                ],
                usage=llm_service_pb2.Usage(
                    prompt_tokens=result.usage.prompt_tokens,
                    completion_tokens=result.usage.completion_tokens,
                    total_tokens=result.usage.total_tokens,
                    tokens_per_second=result.usage.tokens_per_second
                )
            )
            
            return response
            
        except Exception as e:
            context.set_code(grpc.StatusCode.INTERNAL)
            context.set_details(str(e))
            return llm_service_pb2.CompletionResponse()
    
    async def StreamCompletion(self, request, context):
        """流式聊天补全"""
        try:
            messages = [
                {"role": msg.role, "content": msg.content}
                for msg in request.messages
            ]
            
            # 流式生成
            async for chunk in self.model_engine.generate_stream(
                model=request.model,
                messages=messages,
                temperature=request.temperature,
                top_p=request.top_p,
                max_tokens=request.max_tokens
            ):
                yield llm_service_pb2.CompletionResponseChunk(
                    id=chunk.id,
                    object="chat.completion.chunk",
                    created=chunk.created,
                    model=chunk.model,
                    choices=[
                        llm_service_pb2.Choice(
                            index=choice.index,
                            delta=llm_service_pb2.Message(
                                content=choice.delta.content
                            ),
                            finish_reason=choice.finish_reason or ""
                        )
                        for choice in chunk.choices
                    ]
                )
        
        except Exception as e:
            context.set_code(grpc.StatusCode.INTERNAL)
            context.set_details(str(e))
    
    async def CreateEmbedding(self, request, context):
        """创建嵌入"""
        try:
            embeddings = await self.model_engine.embed(
                model=request.model,
                input=request.input
            )
            
            return llm_service_pb2.EmbeddingResponse(
                object="list",
                data=[
                    llm_service_pb2.EmbeddingData(
                        object="embedding",
                        embedding=emb,
                        index=i
                    )
                    for i, emb in enumerate(embeddings)
                ],
                model=request.model,
                usage=llm_service_pb2.Usage(
                    prompt_tokens=sum(len(t.split()) for t in request.input),
                    total_tokens=sum(len(t.split()) for t in request.input)
                )
            )
        
        except Exception as e:
            context.set_code(grpc.StatusCode.INTERNAL)
            context.set_details(str(e))
            return llm_service_pb2.EmbeddingResponse()
    
    async def HealthCheck(self, request, context):
        """健康检查"""
        return llm_service_pb2.HealthCheckResponse(
            status="healthy",
            version="2.0.0",
            models_loaded=["llama-3-70b", "mistral-7b"],
            gpu_memory_usage=0.75,
            active_requests=42
        )

def serve():
    """启动 gRPC 服务"""
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    
    # 添加服务
    model_engine = ModelEngine()
    llm_service_pb2_grpc.add_ChatCompletionServiceServicer_to_server(
        ChatCompletionServicer(model_engine),
        server
    )
    
    # 监听端口
    server.add_insecure_port('[::]:50051')
    server.start()
    server.wait_for_termination()

if __name__ == '__main__':
    serve()
```

#### 性能对比：REST vs gRPC

| 指标 | REST (HTTP/JSON) | gRPC (HTTP/2 + Protobuf) | 说明 |
|------|------------------|-------------------------|------|
| 延迟 | 5-20ms | 1-5ms | gRPC 延迟更低 |
| 吞吐量 | 1000-5000 req/s | 5000-20000 req/s | gRPC 吞吐量更高 |
| 序列化大小 | 1-10KB | 0.5-5KB | Protobuf 更紧凑 |
| 流式支持 | SSE/WebSocket | 原生双向流 | gRPC 流式更强大 |
| 浏览器支持 | ✅ 原生 | ❌ 需要 grpc-web | REST 浏览器友好 |
| 调试便利性 | ✅ 易调试 | ❌ 需要工具 | REST 更易调试 |
| 适用场景 | 外部 API、Web | 内部服务、微服务 | 根据场景选择 |

**推荐架构**：

```
外部客户端 → REST API (HTTP/JSON) → API Gateway → gRPC → 内部微服务 → GPU Workers
```

```python
class SSEFormatter:
    """SSE 格式化工具"""
    
    @staticmethod
    def format_event(event_type: str, data: dict, event_id: str = None) -> str:
        """
        格式化 SSE 事件
        
        格式：
        id: <event_id>
        event: <event_type>
        data: <json_data>
        
        """
        lines = []
        
        if event_id:
            lines.append(f"id: {event_id}")
        
        lines.append(f"event: {event_type}")
        lines.append(f"data: {json.dumps(data, ensure_ascii=False)}")
        lines.append("")  # 空行结束
        lines.append("")  # 双空行
        
        return "\n".join(lines)
    
    @staticmethod
    def format_comment(comment: str) -> str:
        """格式化注释"""
        return f": {comment}\n\n"
    
    @staticmethod
    def format_retry(ms: int) -> str:
        """格式化重试间隔"""
        return f"retry: {ms}\n\n"
```

```python
class ReconnectionHandler:
    """重连处理器"""
    
    def __init__(self, max_retries=5, base_delay=1):
        self.max_retries = max_retries
        self.base_delay = base_delay
    
    async def connect_with_retry(self, url, event_handler):
        """带重试的连接"""
        for attempt in range(self.max_retries):
            try:
                # 建立连接
                async with aiohttp.ClientSession() as session:
                    async with session.get(url) as response:
                        if response.status == 200:
                            # 处理事件流
                            async for line in response.content:
                                event = self.parse_sse_event(line)
                                if event:
                                    await event_handler(event)
                        else:
                            raise Exception(f"HTTP {response.status}")
                
                # 连接成功，退出重试循环
                break
            
            except Exception as e:
                print(f"Connection failed (attempt {attempt + 1}): {e}")
                
                if attempt < self.max_retries - 1:
                    # 指数退避
                    delay = self.base_delay * (2 ** attempt)
                    print(f"Retrying in {delay} seconds...")
                    await asyncio.sleep(delay)
                else:
                    print("Max retries reached")
                    raise
    
    def parse_sse_event(self, line):
        """解析 SSE 事件"""
        # 简化版解析
        if line.startswith('data: '):
            data = line[6:].strip()
            return json.loads(data)
        return None
```

```python
class TokenStreamer:
    """Token 级流式输出器"""
    
    def __init__(self):
        self.token_buffer = []
        self.detokenizer = None
    
    async def stream_tokens(
        self,
        token_generator,
        output_callback,
        flush_interval=0.05
    ):
        """
        流式输出 token
        
        Args:
            token_generator: token 生成器
            output_callback: 输出回调函数
            flush_interval: 刷新间隔（秒）
        """
        last_flush_time = time.time()
        
        async for token_info in token_generator:
            token_id = token_info['token_id']
            token_text = token_info['text']
            logprob = token_info.get('logprob')
            
            # 解码 token
            decoded_text = self.decode_token(token_id)
            
            # 添加到缓冲
            self.token_buffer.append(decoded_text)
            
            # 检查是否需要刷新
            current_time = time.time()
            if current_time - last_flush_time >= flush_interval:
                await self.flush_buffer(output_callback)
                last_flush_time = current_time
            
            # 发送 token 信息
            await output_callback({
                'type': 'token',
                'token_id': token_id,
                'text': decoded_text,
                'logprob': logprob,
                'timestamp': current_time
            })
        
        # 刷新剩余缓冲
        await self.flush_buffer(output_callback)
    
    async def flush_buffer(self, output_callback):
        """刷新缓冲"""
        if self.token_buffer:
            combined_text = ''.join(self.token_buffer)
            await output_callback({
                'type': 'text',
                'text': combined_text,
                'timestamp': time.time()
            })
            self.token_buffer = []
    
    def decode_token(self, token_id):
        """解码 token"""
        # 实现依赖具体的 tokenizer
        return self.detokenizer.decode([token_id])
```

## 3. 负载均衡策略

### 3.1 负载均衡算法

不同的负载均衡算法适用于不同的场景。

#### 负载均衡算法实现

```python
from typing import List, Dict, Optional
import random
import time
from collections import deque

class BackendServer:
    """后端服务器"""
    
    def __init__(self, host: str, port: int, weight: int = 1):
        self.host = host
        self.port = port
        self.weight = weight
        self.active_connections = 0
        self.total_requests = 0
        self.failed_requests = 0
        self.response_times = deque(maxlen=100)
        self.health_status = "healthy"
        self.last_health_check = time.time()
    
    @property
    def address(self) -> str:
        return f"{self.host}:{self.port}"
    
    @property
    def avg_response_time(self) -> float:
        if not self.response_times:
            return 0.0
        return sum(self.response_times) / len(self.response_times)
    
    @property
    def error_rate(self) -> float:
        if self.total_requests == 0:
            return 0.0
        return self.failed_requests / self.total_requests
    
    def record_request(self, response_time: float, success: bool = True):
        """记录请求"""
        self.total_requests += 1
        self.response_times.append(response_time)
        if not success:
            self.failed_requests += 1
    
    def is_healthy(self) -> bool:
        """检查健康状态"""
        return (
            self.health_status == "healthy" and
            self.error_rate < 0.1 and  # 错误率 < 10%
            time.time() - self.last_health_check < 30  # 最近 30 秒内检查过
        )

class LoadBalancer:
    """负载均衡器基类"""
    
    def __init__(self, backends: List[BackendServer]):
        self.backends = backends
        self.current_index = 0
    
    def get_healthy_backends(self) -> List[BackendServer]:
        """获取健康的后端"""
        return [b for b in self.backends if b.is_healthy()]
    
    def select_backend(self) -> Optional[BackendServer]:
        """选择后端（子类实现）"""
        raise NotImplementedError

class RoundRobinLoadBalancer(LoadBalancer):
    """轮询负载均衡"""
    
    def select_backend(self) -> Optional[BackendServer]:
        healthy_backends = self.get_healthy_backends()
        if not healthy_backends:
            return None
        
        backend = healthy_backends[self.current_index % len(healthy_backends)]
        self.current_index += 1
        return backend

class LeastConnectionsLoadBalancer(LoadBalancer):
    """最少连接负载均衡"""
    
    def select_backend(self) -> Optional[BackendServer]:
        healthy_backends = self.get_healthy_backends()
        if not healthy_backends:
            return None
        
        # 选择连接数最少的后端
        return min(healthy_backends, key=lambda b: b.active_connections)

class WeightedRoundRobinLoadBalancer(LoadBalancer):
    """加权轮询负载均衡"""
    
    def __init__(self, backends: List[BackendServer]):
        super().__init__(backends)
        self.current_weight = 0
        self.max_weight = max(b.weight for b in backends)
    
    def select_backend(self) -> Optional[BackendServer]:
        healthy_backends = self.get_healthy_backends()
        if not healthy_backends:
            return None
        
        # 过滤出当前权重的后端
        candidates = [
            b for b in healthy_backends
            if b.weight >= self.current_weight
        ]
        
        if not candidates:
            self.current_weight = 0
            return self.select_backend()
        
        backend = candidates[self.current_index % len(candidates)]
        self.current_index += 1
        self.current_weight = min(self.current_weight + 1, self.max_weight)
        
        return backend

class LatencyBasedLoadBalancer(LoadBalancer):
    """基于延迟的负载均衡"""
    
    def select_backend(self) -> Optional[BackendServer]:
        healthy_backends = self.get_healthy_backends()
        if not healthy_backends:
            return None
        
        # 选择平均延迟最低的后端
        return min(
            healthy_backends,
            key=lambda b: b.avg_response_time if b.avg_response_time > 0 else float('inf')
        )

class ConsistentHashLoadBalancer(LoadBalancer):
    """一致性哈希负载均衡"""
    
    def __init__(self, backends: List[BackendServer], virtual_nodes: int = 100):
        super().__init__(backends)
        self.virtual_nodes = virtual_nodes
        self.hash_ring: Dict[int, BackendServer] = {}
        self._build_hash_ring()
    
    def _build_hash_ring(self):
        """构建哈希环"""
        import hashlib
        
        for backend in self.backends:
            for i in range(self.virtual_nodes):
                # 计算虚拟节点的哈希值
                key = f"{backend.address}-{i}"
                hash_value = int(hashlib.md5(key.encode()).hexdigest(), 16)
                self.hash_ring[hash_value] = backend
        
        # 排序哈希环
        self.sorted_hashes = sorted(self.hash_ring.keys())
    
    def select_backend(self, key: str = None) -> Optional[BackendServer]:
        """基于键选择后端"""
        if key is None:
            key = str(time.time())  # 默认使用随机键
        
        import hashlib
        hash_value = int(hashlib.md5(key.encode()).hexdigest(), 16)
        
        # 找到第一个大于等于 hash_value 的节点
        for h in self.sorted_hashes:
            if h >= hash_value:
                return self.hash_ring[h]
        
        # 如果没找到，返回第一个
        return self.hash_ring[self.sorted_hashes[0]] if self.sorted_hashes else None
```

### 5.1 KV Cache 复用

KV Cache 是 Transformer 推理中的重要优化技术。

#### PagedAttention 实现

```python
import torch
from typing import List, Optional

class PagedAttentionCache:
    """分页注意力缓存"""
    
    def __init__(
        self,
        block_size: int = 16,
        num_blocks: int = 10000,
        num_heads: int = 32,
        head_dim: int = 128,
        dtype: torch.dtype = torch.float16
    ):
        self.block_size = block_size
        self.num_blocks = num_blocks
        
        # 分配 KV cache 内存池
        self.key_cache = torch.zeros(
            (num_blocks, num_heads, block_size, head_dim),
            dtype=dtype,
            device='cuda'
        )
        self.value_cache = torch.zeros(
            (num_blocks, num_heads, block_size, head_dim),
            dtype=dtype,
            device='cuda'
        )
        
        # 块管理
        self.free_blocks = list(range(num_blocks))
        self.block_tables = {}  # {request_id: [block_ids]}
        self.context_lengths = {}  # {request_id: context_length}
    
    def allocate_blocks(self, request_id: str, num_tokens: int) -> List[int]:
        """分配块"""
        num_blocks_needed = (num_tokens + self.block_size - 1) // self.block_size
        
        if num_blocks_needed > len(self.free_blocks):
            raise RuntimeError("Out of memory")
        
        # 分配块
        block_ids = self.free_blocks[:num_blocks_needed]
        self.free_blocks = self.free_blocks[num_blocks_needed:]
        self.block_tables[request_id] = block_ids
        self.context_lengths[request_id] = num_tokens
        
        return block_ids
    
    def update_cache(
        self,
        request_id: str,
        key_states: torch.Tensor,
        value_states: torch.Tensor,
        slot_mapping: torch.Tensor
    ):
        """更新 KV cache"""
        block_ids = self.block_tables[request_id]
        
        # 将 key/value 状态复制到对应的块中
        for i, block_id in enumerate(block_ids):
            start_idx = i * self.block_size
            end_idx = start_idx + self.block_size
            
            self.key_cache[block_id] = key_states[:, :, start_idx:end_idx]
            self.value_cache[block_id] = value_states[:, :, start_idx:end_idx]
        
        self.context_lengths[request_id] += key_states.size(2)
    
    def get_cache(self, request_id: str) -> tuple:
        """获取 KV cache"""
        block_ids = self.block_tables[request_id]
        
        # 拼接所有块
        key_cache = torch.cat([
            self.key_cache[block_id] for block_id in block_ids
        ], dim=2)
        value_cache = torch.cat([
            self.value_cache[block_id] for block_id in block_ids
        ], dim=2)
        
        return key_cache, value_cache
    
    def free_blocks(self, request_id: str):
        """释放块"""
        if request_id in self.block_tables:
            block_ids = self.block_tables[request_id]
            self.free_blocks.extend(block_ids)
            del self.block_tables[request_id]
            del self.context_lengths[request_id]
```

### 5.2 语义缓存

语义缓存基于向量相似度缓存相似的查询结果。

```python
import redis
import numpy as np
from sentence_transformers import SentenceTransformer

class SemanticCache:
    """语义缓存"""
    
    def __init__(
        self,
        redis_url: str = "redis://localhost:6379",
        similarity_threshold: float = 0.95,
        embedding_model: str = 'all-MiniLM-L6-v2',
        ttl: int = 3600
    ):
        self.redis_client = redis.from_url(redis_url)
        self.similarity_threshold = similarity_threshold
        self.ttl = ttl
        
        # 加载嵌入模型
        self.embedding_model = SentenceTransformer(embedding_model)
        
        # 创建 FAISS 索引
        import faiss
        self.embedding_dim = self.embedding_model.get_sentence_embedding_dimension()
        self.index = faiss.IndexFlatIP(self.embedding_dim)
        self.cache_mapping = {}  # {index_id: (cache_key, response)}
    
    def _compute_embedding(self, text: str) -> np.ndarray:
        """计算嵌入"""
        embedding = self.embedding_model.encode(text)
        embedding = embedding / np.linalg.norm(embedding)  # 归一化
        return embedding.reshape(1, -1)
    
    def get(self, prompt: str) -> Optional[str]:
        """查找缓存"""
        embedding = self._compute_embedding(prompt)
        
        # 搜索相似嵌入
        similarities, indices = self.index.search(embedding, k=1)
        
        if similarities[0][0] >= self.similarity_threshold:
            cache_key = self.cache_mapping[indices[0][0]]
            cached_response = self.redis_client.get(cache_key)
            
            if cached_response:
                return cached_response.decode('utf-8')
        
        return None
    
    def set(self, prompt: str, response: str):
        """设置缓存"""
        embedding = self._compute_embedding(prompt)
        
        # 添加到 FAISS 索引
        cache_key = f"cache:{len(self.cache_mapping)}"
        self.index.add(embedding)
        self.cache_mapping[len(self.cache_mapping)] = cache_key
        
        # 存储到 Redis
        self.redis_client.setex(cache_key, self.ttl, response)
    
    def clear(self):
        """清空缓存"""
        self.index = faiss.IndexFlatIP(self.embedding_dim)
        self.cache_mapping = {}
        self.redis_client.flushdb()
    
    def get_stats(self) -> dict:
        """获取缓存统计"""
        return {
            'total_entries': self.index.ntotal,
            'hit_rate': self._calculate_hit_rate()
        }
```

```python
class DynamicBatcher:
    """动态批处理器"""
    
    def __init__(
        self,
        max_batch_size: int = 32,
        max_wait_time: float = 0.1,
        max_sequence_length: int = 2048
    ):
        self.max_batch_size = max_batch_size
        self.max_wait_time = max_wait_time
        self.max_sequence_length = max_sequence_length
        
        self.request_queue = []
        self.last_batch_time = time.time()
    
    def add_request(self, request: dict) -> bool:
        """添加请求到队列"""
        if len(request['input_ids']) > self.max_sequence_length:
            return False
        
        self.request_queue.append(request)
        return True
    
    def get_batch(self) -> Optional[List[dict]]:
        """获取批次"""
        current_time = time.time()
        time_since_last_batch = current_time - self.last_batch_time
        
        # 检查是否应该创建批次
        if (len(self.request_queue) >= self.max_batch_size or
            time_since_last_batch >= self.max_wait_time):
            
            if not self.request_queue:
                return None
            
            # 创建批次
            batch = self.request_queue[:self.max_batch_size]
            self.request_queue = self.request_queue[self.max_batch_size:]
            self.last_batch_time = current_time
            
            return batch
        
        return None
    
    async def process_queue(self, process_fn):
        """处理队列"""
        while True:
            batch = self.get_batch()
            
            if batch:
                # 处理批次
                results = await process_fn(batch)
                
                # 返回结果
                for request, result in zip(batch, results):
                    request['future'].set_result(result)
            
            await asyncio.sleep(0.01)  # 10ms 检查间隔
```

```python
import torch

class ModelCompressor:
    """模型压缩器"""
    
    @staticmethod
    def quantize_8bit(model_name: str):
        """8-bit 量化"""
        quantization_config = BitsAndBytesConfig(
            load_in_8bit=True,
            llm_int8_threshold=6.0
        )
        
        model = AutoModelForCausalLM.from_pretrained(
            model_name,
            quantization_config=quantization_config,
            device_map='auto'
        )
        
        return model
    
    @staticmethod
    def quantize_4bit(model_name: str):
        """4-bit 量化"""
        quantization_config = BitsAndBytesConfig(
            load_in_4bit=True,
            bnb_4bit_compute_dtype=torch.float16,
            bnb_4bit_use_double_quant=True,
            bnb_4bit_quant_type="nf4"
        )
        
        model = AutoModelForCausalLM.from_pretrained(
            model_name,
            quantization_config=quantization_config,
            device_map='auto'
        )
        
        return model
    
    @staticmethod
    def prune_model(model, pruning_ratio: float = 0.3):
        """模型剪枝"""
        import torch.nn.utils.prune as prune
        
        for name, module in model.named_modules():
            if isinstance(module, torch.nn.Linear):
                prune.l1_unstructured(module, name='weight', amount=pruning_ratio)
                prune.remove(module, 'weight')  # 使剪枝永久化
        
        return model

# 压缩效果对比
def compare_compression_methods():
    """比较压缩方法"""

        ('Pruned 30%', 9.1, 16.9, '1.1x', '95%'),
```

```python
from prometheus_client import Counter, Histogram, Gauge, generate_latest
from fastapi import Response

# 定义指标
REQUEST_COUNT = Counter(
    'llm_requests_total',
    'Total LLM requests',
    ['model', 'endpoint', 'status']
)

REQUEST_LATENCY = Histogram(
    'llm_request_latency_seconds',
    'LLM request latency',
    ['model', 'endpoint'],
    buckets=[0.1, 0.5, 1.0, 2.0, 5.0, 10.0]
)

TOKEN_COUNT = Counter(
    'llm_tokens_total',
    'Total tokens processed',
    ['model', 'type']  # type: prompt or completion
)

GPU_MEMORY_USAGE = Gauge(
    'llm_gpu_memory_usage_bytes',
    'GPU memory usage',
    ['gpu_id']
)

ACTIVE_REQUESTS = Gauge(
    'llm_active_requests',
    'Number of active requests'
)

CACHE_HIT_RATE = Gauge(
    'llm_cache_hit_rate',
    'Cache hit rate'
)

# 中间件
@app.middleware("http")
async def metrics_middleware(request, call_next):
    """指标收集中间件"""
    start_time = time.time()
    
    ACTIVE_REQUESTS.inc()
    
    try:
        response = await call_next(request)
        
        # 记录指标
        REQUEST_COUNT.labels(
            model=request.path_params.get('model', 'unknown'),
            endpoint=request.url.path,
            status=response.status_code
        ).inc()
        
        REQUEST_LATENCY.labels(
            model=request.path_params.get('model', 'unknown'),
            endpoint=request.url.path
        ).observe(time.time() - start_time)
        
        return response
    
    finally:
        ACTIVE_REQUESTS.dec()

# /metrics 端点
@app.get("/metrics")
async def metrics():
    """Prometheus 指标"""
    return Response(content=generate_latest(), media_type="text/plain")
```
