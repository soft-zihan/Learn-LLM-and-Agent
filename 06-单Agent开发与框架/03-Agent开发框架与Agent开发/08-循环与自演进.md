## 1. Loop Engineering 概念

### 1.1 什么是 Loop

Loop 让 Agent 能够**自主持续运行**，无需人工干预每次迭代：

**Harness 模式**（单次任务）：
```
人类启动 → Agent 执行 → 人类检查 → 结束
```

**Loop 模式**（自主循环）：
```
人类设计目标 → Agent 自主循环执行 → 人类只看最终结果
```

### 1.2 核心转变

```python
# Harness 模式：人必须参与每次循环
human: "帮我修复这个 bug"
result = agent.execute(task)
human: "检查结果"  # 人必须参与

# Loop 模式：系统自主运行
human_designer: 定义目标 + 验证标准
loop_system:
    while not 目标完成:
        agent.execute()
        independent_verifier.verify()  # 独立验证者
        if 需要人类:
            escalate_to_human()
    return 最终结果
```

## 2. 生成器-评估器-规划器架构

### 2.1 三组件架构设计（完整代码）

核心思想：**分开制造者和验证者**，避免自我评估放水：

```python
from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage

# 不同能力的模型
generator_llm = ChatOpenAI(model="gpt-4o")      # 强模型：生成
evaluator_llm = ChatOpenAI(model="gpt-4o-mini")  # 便宜模型：评估
planner_llm = ChatOpenAI(model="gpt-4o-mini")    # 便宜模型：规划

class GeneratorEvaluatorState(TypedDict):
    messages: Annotated[list, add_messages]
    draft: str  # 生成的草稿
    evaluation: str  # 评估结果
    score: float  # 评分
    round: int
    max_rounds: int

def generator_node(state: GeneratorEvaluatorState) -> GeneratorEvaluatorState:
    """生成器：负责创造性工作"""
    draft_prompt = f"""
    作为资深工程师，高质量完成以下任务：
    
    任务：{state['messages'][-1].content}
    
    之前的反馈：{state.get('evaluation', '无')}
    
    请生成或改进实现方案。
    """
    
    response = generator_llm.invoke([HumanMessage(content=draft_prompt)])
    
    return {
        "draft": response.content,
        "round": state.get("round", 0) + 1
    }

def evaluator_node(state: GeneratorEvaluatorState) -> GeneratorEvaluatorState:
    """评估器：独立批判和打分"""
    eval_prompt = f"""
    作为严格的代码审查专家，评估以下实现：
    
    任务：{state['messages'][-1].content}
    实现：{state['draft']}
    
    评估维度（0-10分）：
    1. 正确性：功能是否正确
    2. 安全性：有无安全漏洞
    3. 性能：是否高效
    4. 可维护性：代码质量
    
    返回 JSON：
    ```json
    {{
        "score": 0-10,
        "issues": ["问题列表"],
        "suggestions": ["改进建议"],
        "pass": true/false
    }}
    ```
    """
    
    response = evaluator_llm.invoke([HumanMessage(content=eval_prompt)])
    
    # 解析评估结果
    import json
    try:
        eval_result = json.loads(response.content)
    except:
        eval_result = {"score": 5, "pass": False, "issues": ["解析失败"]}
    
    return {
        "evaluation": response.content,
        "score": eval_result.get("score", 0)
    }

def planner_node(state: GeneratorEvaluatorState) -> str:
    """规划器：决定下一步"""
    score = state.get("score", 0)
    round_num = state.get("round", 0)
    max_rounds = state.get("max_rounds", 5)
    
    # 决策逻辑
    if score >= 8:  # 高质量，通过
        return "finalize"
    elif round_num >= max_rounds:  # 超过最大轮数
        return "finalize"
    else:  # 继续改进
        return "improve"

def finalize_node(state: GeneratorEvaluatorState) -> GeneratorEvaluatorState:
    """完成"""
    final_message = f"""
    最终答案（经过 {state['round']} 轮改进，评分：{state.get('score', 0)}/10）：
    
    {state['draft']}
    
    评估历史：
    {state.get('evaluation', '无')}
    """
    
    return {"messages": [AIMessage(content=final_message)]}

# 构建图
workflow = StateGraph(GeneratorEvaluatorState)
workflow.add_node("generate", generator_node)
workflow.add_node("evaluate", evaluator_node)
workflow.add_node("finalize", finalize_node)

workflow.add_edge(START, "generate")
workflow.add_edge("generate", "evaluate")
workflow.add_conditional_edges(
    "evaluate",
    planner_node,
    {
        "improve": "generate",  # 循环回生成
        "finalize": "finalize"  # 完成
    }
)
workflow.add_edge("finalize", END)

agent = workflow.compile()

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="编写一个线程安全的缓存系统")],
    "max_rounds": 5
})

print(result["messages"][-1]["content"])
```

### 2.2 评估器的关键设计

**评估器应该做什么**：

```python
# ✅ 正确做法：实际运行验证

    # 只阅读代码，不实际运行
    # 容易自我欺骗
)
```

## 3. Loop 实战模式

### 3.1 完整 Loop 案例（完整代码）

自动化代码审查 Loop：

```python
from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage

llm = ChatOpenAI(model="gpt-4o")

class CodeReviewLoopState(TypedDict):
    messages: Annotated[list, add_messages]
    code: str
    review_results: List[str]
    issues_fixed: int
    max_iterations: int

def security_review(state: CodeReviewLoopState) -> CodeReviewLoopState:
    """安全审查"""
    review_prompt = f"""
    审查以下代码的安全问题：
    
    代码：
    {state['code']}
    
    检查：
    1. SQL 注入
    2. XSS 攻击
    3. 缓冲区溢出
    4. 权限绕过
    
    返回发现的安全问题。
    """
    
    response = llm.invoke([HumanMessage(content=review_prompt)])
    
    return {
        "review_results": state.get("review_results", []) + [f"安全审查：{response.content}"]
    }

def performance_review(state: CodeReviewLoopState) -> CodeReviewLoopState:
    """性能审查"""
    review_prompt = f"""
    审查以下代码的性能问题：
    
    代码：
    {state['code']}
    
    检查：
    1. 时间复杂度
    2. 内存使用
    3. 不必要的计算
    4. 缓存机会
    
    返回性能优化建议。
    """
    
    response = llm.invoke([HumanMessage(content=review_prompt)])
    
    return {
        "review_results": state["review_results"] + [f"性能审查：{response.content}"]
    }

def fix_issues(state: CodeReviewLoopState) -> CodeReviewLoopState:
    """修复问题"""
    all_reviews = "\n\n".join(state["review_results"])
    
    fix_prompt = f"""
    根据以下审查意见，修复代码中的问题：
    
    原代码：
    {state['code']}
    
    审查意见：
    {all_reviews}
    
    返回修复后的完整代码。
    """
    
    response = llm.invoke([HumanMessage(content=fix_prompt)])
    
    return {
        "code": response.content,
        "issues_fixed": state.get("issues_fixed", 0) + 1
    }

def should_continue(state: CodeReviewLoopState) -> str:
    """判断是否继续审查"""
    if state.get("issues_fixed", 0) >= state.get("max_iterations", 3):
        return "finalize"
    
    # 简单判断：如果还有问题，继续
    if "发现" in state["review_results"][-1]:
        return "continue"
    return "finalize"

def finalize_review(state: CodeReviewLoopState) -> CodeReviewLoopState:
    """完成审查"""
    final_report = f"""
    代码审查完成（经过 {state['issues_fixed']} 轮修复）
    
    审查历史：
    {state['review_results']}
    
    最终代码：
    {state['code']}
    """
    
    return {"messages": [AIMessage(content=final_report)]}

# 构建循环图
workflow = StateGraph(CodeReviewLoopState)
workflow.add_node("security", security_review)
workflow.add_node("performance", performance_review)
workflow.add_node("fix", fix_issues)
workflow.add_node("finalize", finalize_review)

workflow.add_edge(START, "security")
workflow.add_edge("security", "performance")
workflow.add_edge("performance", "fix")
workflow.add_conditional_edges(
    "fix",
    should_continue,
    {
        "continue": "security",  # 循环回去
        "finalize": "finalize"
    }
)
workflow.add_edge("finalize", END)

agent = workflow.compile()

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="审查以下代码")],
    "code": """
def login(username, password):
    query = f"SELECT * FROM users WHERE username='{username}' AND password='{password}'"
    return execute_query(query)
""",
    "max_iterations": 3
})

print(result["messages"][-1]["content"])
```

### 3.2 /loop vs /goal

| 模式 | 触发方式 | 运行时长 | 适用场景 |
|------|---------|---------|---------|
| **/loop** | 按节奏重复执行 | 持续运行 | CI 修复、代码审查 |
| **/goal** | 一直跑到条件满足 | 直到目标达成 | 复杂任务、多步骤工作 |

## 4. 风险与治理

### 4.1 理解债务

**问题**：Loop 越快，"代码库有什么"和"你真正理解什么"之间的差距越大：

```
Week 1: Loop 修复 10 个 bug → 你看了 10 个 PR → 理解 100%
Week 2: Loop 修复 50 个 bug → 你看了 20 个 PR → 理解 60%
Week 4: Loop 修复 200 个 bug → 你看了 10 个 PR → 理解 20%
```

**应对策略**：

```python
class UnderstandingDebtGuard:
    """理解债务防护"""
    
    def __init__(self):
        self.reviewed_count = 0
        self.total_count = 0
        self.min_review_ratio = 0.3  # 至少审查 30%
    
    def should_review(self, pr) -> bool:
        """判断是否应该人工审查"""
        current_ratio = self.reviewed_count / max(self.total_count, 1)
        
        if current_ratio < self.min_review_ratio:
            return True  # 强制审查
        
        import random
        return random.random() < 0.2  # 20% 随机抽查
```

### 4.2 验证体系设计

Loop 中必须包含说"不"的机制：

```python
class LoopVerificationSystem:
    """Loop 的验证体系"""
    
    def verify(self, result) -> tuple[bool, str]:
        """验证结果"""
        checks = [
            self.run_unit_tests,
            self.run_integration_tests,
            self.run_type_check,
            self.run_lint,
            self.run_security_scan,
        ]
        
        for check in checks:
            passed, error = check(result)
            if not passed:
                return False, f"验证失败：{check.__name__} - {error}"
        
        return True, "所有验证通过"
```

> 没有反馈机制的 Loop，只会让 Agent 不断重复并自我确认。

## 6. Spring AI 循环与自演进实现

### 6.1 生成器-评估器-规划器架构

对应 Python 的三组件架构设计：

```java
import lombok.Data;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

/**
 * 生成器-评估器-规划器架构（对应 Python 的 GeneratorEvaluatorState）
 * 
 * 核心思想：分开制造者和验证者，避免自我评估放水
 */
@Service
public class GeneratorEvaluatorService {

    private final ChatClient generatorClient;  // 强模型：生成
    private final ChatClient evaluatorClient;  // 便宜模型：评估
    private final ChatClient plannerClient;    // 便宜模型：规划

    public GeneratorEvaluatorService(ChatModel chatModel) {
        // 生成器：使用强模型
        this.generatorClient = ChatClient.builder(chatModel)
            .defaultSystem("作为资深工程师，高质量完成以下任务")
            .build();
        
        // 评估器：使用便宜模型
        this.evaluatorClient = ChatClient.builder(chatModel)
            .defaultSystem("作为严格的代码审查专家，评估以下实现")
            .build();
        
        // 规划器：使用便宜模型
        this.plannerClient = ChatClient.builder(chatModel)
            .build();
    }

    @Data
    public static class GeneratorEvaluatorState {
        private String task;           // 任务
        private String draft;          // 生成的草稿
        private String evaluation;     // 评估结果
        private double score;          // 评分
        private int round;             // 当前轮数
        private int maxRounds;         // 最大轮数
    }

    @Data
    public static class EvaluationResult {
        private double score;
        private boolean pass;
        private String issues;
        private String suggestions;
    }

    /**
     * 生成器节点（对应 Python 的 generator_node）
     */
    public String generate(GeneratorEvaluatorState state) {
        String prompt = """
            任务：%s
            
            之前的反馈：%s
            
            请生成或改进实现方案。
            """.formatted(
                state.getTask(),
                state.getEvaluation() != null ? state.getEvaluation() : "无"
            );
        
        return generatorClient.prompt()
            .user(prompt)
            .call()
            .content();
    }

    /**
     * 评估器节点（对应 Python 的 evaluator_node）
     */
    public EvaluationResult evaluate(GeneratorEvaluatorState state) {
        String prompt = """
            任务：%s
            实现：%s
            
            评估维度（0-10分）：
            1. 正确性：功能是否正确
            2. 安全性：有无安全漏洞
            3. 性能：是否高效
            4. 可维护性：代码质量
            
            返回 JSON 格式：{"score": 0-10, "pass": true/false}
            """.formatted(state.getTask(), state.getDraft());
        
        String response = evaluatorClient.prompt()
            .user(prompt)
            .call()
            .content();
        
        // 简化版解析，实际应使用 JSON 解析库
        EvaluationResult result = new EvaluationResult();
        result.setScore(5.0);  // 默认值
        result.setPass(false);
        return result;
    }

    /**
     * 规划器节点（对应 Python 的 planner_node）
     */
    public String plan(GeneratorEvaluatorState state) {
        double score = state.getScore();
        int roundNum = state.getRound();
        int maxRounds = state.getMaxRounds();
        
        // 决策逻辑
        if (score >= 8) {  // 高质量，通过
            return "finalize";
        } else if (roundNum >= maxRounds) {  // 超过最大轮数
            return "finalize";
        } else {  // 继续改进
            return "improve";
        }
    }

    /**
     * 执行循环（对应 Python 的 workflow.compile().invoke()）
     */
    public String executeLoop(String task, int maxRounds) {
        GeneratorEvaluatorState state = new GeneratorEvaluatorState();
        state.setTask(task);
        state.setMaxRounds(maxRounds);
        state.setRound(0);
        
        while (true) {
            // 1. 生成
            state.setRound(state.getRound() + 1);
            state.setDraft(generate(state));
            
            // 2. 评估
            EvaluationResult evalResult = evaluate(state);
            state.setScore(evalResult.getScore());
            state.setEvaluation("评分：" + evalResult.getScore());
            
            // 3. 规划
            String decision = plan(state);
            
            if ("finalize".equals(decision)) {
                break;
            }
            
            System.out.println("第 " + state.getRound() + " 轮，评分：" + state.getScore());
        }
        
        return "最终答案（经过 " + state.getRound() + " 轮改进，评分：" + state.getScore() + "/10）：\n\n" + state.getDraft();
    }
}
```

### 6.2 代码审查循环

对应 Python 的完整 Loop 案例：

```java
import lombok.Data;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;

/**
 * 自动化代码审查 Loop（对应 Python 的 CodeReviewLoopState）
 */
@Service
public class CodeReviewLoopService {

    private final ChatClient reviewClient;

    public CodeReviewLoopService(ChatModel chatModel) {
        this.reviewClient = ChatClient.builder(chatModel)
            .defaultSystem("你是严格的代码审查专家")
            .build();
    }

    @Data
    public static class CodeReviewState {
        private String code;                // 代码
        private List<String> reviewResults; // 审查结果
        private int issuesFixed;            // 已修复问题数
        private int maxIterations;          // 最大迭代次数
    }

    /**
     * 安全审查（对应 Python 的 security_review）
     */
    public String securityReview(String code) {
        String prompt = """
            审查以下代码的安全问题：
            
            代码：
            %s
            
            检查：SQL 注入、XSS 攻击、缓冲区溢出、权限绕过
            """.formatted(code);
        
        return reviewClient.prompt()
            .user(prompt)
            .call()
            .content();
    }

    /**
     * 性能审查（对应 Python 的 performance_review）
     */
    public String performanceReview(String code) {
        String prompt = """
            审查以下代码的性能问题：
            
            代码：
            %s
            
            检查：时间复杂度、内存使用、不必要的计算、缓存机会
            """.formatted(code);
        
        return reviewClient.prompt()
            .user(prompt)
            .call()
            .content();
    }

    /**
     * 修复问题（对应 Python 的 fix_issues）
     */
    public String fixIssues(String code, List<String> reviewResults) {
        String allReviews = String.join("\n\n", reviewResults);
        
        String prompt = """
            根据以下审查意见，修复代码中的问题：
            
            原代码：
            %s
            
            审查意见：
            %s
            
            返回修复后的完整代码。
            """.formatted(code, allReviews);
        
        return reviewClient.prompt()
            .user(prompt)
            .call()
            .content();
    }

    /**
     * 执行代码审查循环
     */
    public String executeCodeReview(String code, int maxIterations) {
        CodeReviewState state = new CodeReviewState();
        state.setCode(code);
        state.setReviewResults(new ArrayList<>());
        state.setIssuesFixed(0);
        state.setMaxIterations(maxIterations);
        
        while (state.getIssuesFixed() < state.getMaxIterations()) {
            // 1. 安全审查
            String securityResult = securityReview(state.getCode());
            state.getReviewResults().add("安全审查：" + securityResult);
            
            // 2. 性能审查
            String performanceResult = performanceReview(state.getCode());
            state.getReviewResults().add("性能审查：" + performanceResult);
            
            // 3. 修复问题
            String fixedCode = fixIssues(state.getCode(), state.getReviewResults());
            state.setCode(fixedCode);
            state.setIssuesFixed(state.getIssuesFixed() + 1);
            
            // 4. 判断是否继续
            if (!securityResult.contains("发现") && !performanceResult.contains("发现")) {
                break;
            }
        }
        
        return "代码审查完成（经过 " + state.getIssuesFixed() + " 轮修复）\n\n" +
               "最终代码：\n" + state.getCode();
    }
}
```
