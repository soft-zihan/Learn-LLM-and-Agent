## 1. 提示词工程基础

### 1.1 什么是好的 Prompt

好的 Prompt 遵循 **CLEAR** 原则：

- **C**lear（清晰）：意图明确，无歧义
- **L**imited（聚焦）：一次只做一件事
- **E**xplicit（显式）：明确输出格式和要求
- **A**dapted（适配）：根据模型能力调整
- **R**epeatable（可重复）：相同输入产生稳定输出

## 2. 上下文工程核心

### 2.1 什么是 Context Engineering

Context Engineering 是管理和优化 LLM 上下文窗口的艺术：

- **Prompt Engineering**：如何写好单个提示
- **Context Engineering**：如何给模型正确的上下文信息

**核心挑战**：
- 上下文窗口有限（通常 128K-200K tokens）
- 长对话历史容易溢出
- 无关信息干扰模型判断
- 关键信息可能被"遗忘"

### 2.2 上下文组装技术

Agent 的上下文通常由四部分组成：

```
完整上下文
├── System Context（系统级）
│   ├── 角色定义
│   ├── 行为规则
│   └── 工具描述
│
├── User Context（用户级）
│   ├── 用户偏好
│   ├── 历史交互
│   └── 当前任务
│
├── Session Context（会话级）
│   ├── 对话历史
│   ├── 中间结果
│   └── 工具调用记录
│
└── Tool Context（工具级）
    ├── 工具返回结果
    ├── 错误信息
    └── 外部数据
```

#### 实践：动态上下文组装

```python
from langchain_core.messages import SystemMessage, HumanMessage, AIMessage

def build_context(
    user_query: str,
    user_profile: dict,
    conversation_history: list,
    tool_results: list
) -> list:
    """动态组装上下文"""
    messages = []
    
    # 1. System Context
    system_msg = SystemMessage(content=f"""
    你是智能助手。
    用户偏好：{user_profile.get('preferences', '无')}
    语言：{user_profile.get('language', '中文')}
    """)
    messages.append(system_msg)
    
    # 2. User Context（最近 5 轮对话）
    for msg in conversation_history[-10:]:  # 保留最近 5 轮
        messages.append(msg)
    
    # 3. Tool Context（最近的工具调用结果）
    for result in tool_results[-3:]:  # 只保留最近 3 个结果
        messages.append(AIMessage(content=f"工具结果：{result}"))
    
    # 4. 当前任务
    messages.append(HumanMessage(content=user_query))
    
    return messages
```

### 2.3 上下文窗口管理策略

#### 策略 1：上下文裁剪

当接近窗口限制时，删除早期消息：

```python
def trim_context(messages: list, max_tokens: int = 100000) -> list:
    """裁剪上下文到最大 token 数"""
    total_tokens = sum(len(msg.content) for msg in messages)
    
    if total_tokens <= max_tokens:
        return messages
    
    # 保留 System 消息和最近的消息
    system_msgs = [m for m in messages if m.type == "system"]
    other_msgs = [m for m in messages if m.type != "system"]
    
    # 从最早的开始删除
    while len(other_msgs) > 2:  # 至少保留 2 条
        other_msgs.pop(0)
        total_tokens = sum(len(m.content) for m in system_msgs + other_msgs)
        if total_tokens <= max_tokens:
            break
    
    return system_msgs + other_msgs
```

#### 策略 3：上下文优先级管理

为不同类型的上下文分配优先级：

```python
class ContextPriority:
    """上下文优先级管理"""
    
    @staticmethod
    def prioritize(messages: list, max_tokens: int) -> list:
        """按优先级选择上下文"""
        # 优先级排序
        priorities = {
            "system": 100,      # 最高优先级
            "tool_result": 80,  # 工具结果
            "human": 60,        # 用户消息
            "ai": 40,           # AI 回复
        }
        
        # 按优先级排序
        sorted_msgs = sorted(
            messages,
            key=lambda m: priorities.get(m.type, 0),
            reverse=True
        )
        
        # 选择消息直到达到限制
        selected = []
        total_tokens = 0
        
        for msg in sorted_msgs:
            msg_tokens = len(msg.content)
            if total_tokens + msg_tokens <= max_tokens:
                selected.append(msg)
                total_tokens += msg_tokens
            else:
                break
        
        return selected
```

## 3. LangGraph 中的上下文管理

### 3.1 State 设计模式

在 LangGraph 中，上下文通过 State 管理：

```python
from typing import TypedDict, Annotated
from langgraph.graph.message import add_messages

class AgentState(TypedDict):
    """Agent 状态（上下文）"""
    # 完整消息历史（自动追加）
    messages: Annotated[list, add_messages]
    
    # 压缩后的摘要（可选）
    conversation_summary: str
    
    # 关键信息提取（长期记忆）
    key_facts: list[str]
    
    # 当前任务
    current_task: str
```

### 3.2 多轮对话的上下文维护

```python
def conversation_node(state: AgentState) -> AgentState:
    """对话节点：维护上下文"""
    messages = state["messages"]
    
    # 检查是否需要压缩
    total_tokens = sum(len(m.content) for m in messages)
    
    if total_tokens > 100000:  # 超过窗口限制
        # 1. 提取关键信息
        key_facts = extract_key_facts(messages)
        
        # 2. 生成摘要
        summary = summarize_conversation(messages[:-5])  # 保留最近 5 条
        
        # 3. 重建上下文
        messages = [
            SystemMessage(content=f"对话摘要：{summary}"),
            HumanMessage(content=f"关键信息：{key_facts}")
        ] + messages[-5:]  # 保留最近 5 条
    
    return {
        "messages": messages,
        "conversation_summary": state.get("conversation_summary", ""),
        "key_facts": state.get("key_facts", [])
    }
```

```python
def rag_node(state: AgentState) -> AgentState:
    """RAG 节点：注入检索到的上下文"""
    user_query = state["messages"][-1].content
```

### 3.2 上下文组装技术

对应 Python 的 `build_context()` 函数：

```java
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.messages.Message;
import org.springframework.ai.chat.messages.SystemMessage;
import org.springframework.ai.chat.messages.UserMessage;
import org.springframework.ai.chat.messages.AssistantMessage;
import org.springframework.ai.chat.prompt.Prompt;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;

@Service
public class ContextAssemblyService {

    private final ChatClient chatClient;

    public ContextAssemblyService(ChatClient.Builder builder) {
        this.chatClient = builder.build();
    }

    /**
     * 动态组装上下文（对应 Python 的 build_context）
     * 
     * 完整上下文 = System Context + User Context + Session Context + Tool Context
     */
    public String chatWithContext(
            String userQuery,
            Map<String, Object> userProfile,
            List<Message> conversationHistory,
            List<String> toolResults) {
        
        List<Message> messages = new ArrayList<>();
        
        // 1. System Context（系统级上下文）
        String systemContent = "你是智能助手。\n" +
            "用户偏好：" + userProfile.getOrDefault("preferences", "无") + "\n" +
            "语言：" + userProfile.getOrDefault("language", "中文");
        messages.add(new SystemMessage(systemContent));
        
        // 2. User Context（用户级上下文 - 最近 5 轮对话）
        int startIndex = Math.max(0, conversationHistory.size() - 10);
        messages.addAll(conversationHistory.subList(startIndex, conversationHistory.size()));
        
        // 3. Tool Context（工具级上下文 - 最近 3 个结果）
        int toolStartIndex = Math.max(0, toolResults.size() - 3);
        for (int i = toolStartIndex; i < toolResults.size(); i++) {
            messages.add(new AssistantMessage("工具结果：" + toolResults.get(i)));
        }
        
        // 4. 当前任务
        messages.add(new UserMessage(userQuery));
        
        // 调用 LLM
        Prompt prompt = new Prompt(messages);
        return chatClient.prompt(prompt)
            .call()
            .content();
    }
}
```

### 3.3 上下文窗口管理策略

#### 策略 1：上下文裁剪

```java
import org.springframework.ai.chat.messages.Message;
import org.springframework.ai.chat.messages.SystemMessage;

import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;

public class ContextTrimmer {

    /**
     * 裁剪上下文到最大 token 数（对应 Python 的 trim_context）
     */
    public static List<Message> trimContext(List<Message> messages, int maxTokens) {
        // 计算总 token 数（简化版，实际应使用 tokenizer）
        int totalTokens = messages.stream()
            .mapToInt(m -> m.getText().length() / 4)  // 粗略估算：4字符=1token
            .sum();
        
        if (totalTokens <= maxTokens) {
            return messages;
        }
        
        // 保留 System 消息和最近的消息
        List<Message> systemMessages = messages.stream()
            .filter(m -> m instanceof SystemMessage)
            .collect(Collectors.toList());
        
        List<Message> otherMessages = messages.stream()
            .filter(m -> !(m instanceof SystemMessage))
            .collect(Collectors.toList());
        
        // 从最早的开始删除
        while (otherMessages.size() > 2) {  // 至少保留 2 条
            otherMessages.remove(0);
            totalTokens = systemMessages.stream()
                .mapToInt(m -> m.getText().length() / 4)
                .sum() + otherMessages.stream()
                .mapToInt(m -> m.getText().length() / 4)
                .sum();
            
            if (totalTokens <= maxTokens) {
                break;
            }
        }
        
        List<Message> result = new ArrayList<>(systemMessages);
        result.addAll(otherMessages);
        return result;
    }
}
```

#### 策略 2：上下文压缩（Summarization）

```java
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.messages.Message;
import org.springframework.ai.chat.messages.SystemMessage;
import org.springframework.ai.chat.messages.UserMessage;

import java.util.List;
import java.util.stream.Collectors;

@Service
public class ContextCompressor {

    private final ChatClient compressorClient;  // 使用便宜模型

    public ContextCompressor(ChatClient.Builder builder) {
        // 使用便宜模型做摘要（如 gpt-4o-mini）
        this.compressorClient = builder.build();
    }

    /**
     * 压缩对话历史为摘要（对应 Python 的 summarize_conversation）
     */
    public String summarizeConversation(List<Message> messages, int maxTokens) {
        // 转换为文本
        String conversationText = messages.stream()
            .map(m -> m.getClass().getSimpleName() + ": " + m.getText())
            .collect(Collectors.joining("\n"));
        
        String summaryPrompt = """
            将以下对话压缩为摘要，保留关键信息和决策：
            
            %s
            
            摘要（200字以内）：
            """.formatted(conversationText);
        
        return compressorClient.prompt()
            .user(summaryPrompt)
            .call()
            .content();
    }

    /**
     * 智能上下文管理：超过阈值时自动压缩
     */
    public List<Message> manageContext(List<Message> messages, int threshold) {
        int totalTokens = messages.stream()
            .mapToInt(m -> m.getText().length() / 4)
            .sum();
        
        if (totalTokens > threshold) {
            // 触发压缩
            String summary = summarizeConversation(
                messages.subList(0, messages.size() - 5),  // 保留最近 5 条
                threshold
            );
            
            // 重建上下文
            List<Message> compressed = new ArrayList<>();
            compressed.add(new SystemMessage("对话摘要：" + summary));
            compressed.addAll(messages.subList(
                Math.max(0, messages.size() - 5), 
                messages.size()
            ));
            
            return compressed;
        }
        
        return messages;
    }
}
```

#### 策略 3：上下文优先级管理

```java
import org.springframework.ai.chat.messages.Message;
import org.springframework.ai.chat.messages.SystemMessage;
import org.springframework.ai.chat.messages.UserMessage;
import org.springframework.ai.chat.messages.AssistantMessage;

import java.util.*;
import java.util.stream.Collectors;

public class ContextPriorityManager {

    /**
     * 按优先级选择上下文（对应 Python 的 ContextPriority.prioritize）
     */
    public static List<Message> prioritizeMessages(List<Message> messages, int maxTokens) {
        // 优先级映射
        Map<Class<? extends Message>, Integer> priorities = Map.of(
            SystemMessage.class, 100,      // 最高优先级
            AssistantMessage.class, 80,    // 工具结果（如果是工具返回）
            UserMessage.class, 60,         // 用户消息
            AssistantMessage.class, 40     // AI 回复
        );
        
        // 按优先级排序（简化版，实际应根据内容判断）
        List<Message> sortedMessages = messages.stream()
            .sorted(Comparator.comparingInt(m -> {
                if (m instanceof SystemMessage) return -100;
                if (m instanceof UserMessage) return -60;
                return -40;
            }))
            .collect(Collectors.toList());
        
        // 选择消息直到达到限制
        List<Message> selected = new ArrayList<>();
        int totalTokens = 0;
        
        for (Message msg : sortedMessages) {
            int msgTokens = msg.getText().length() / 4;
            if (totalTokens + msgTokens <= maxTokens) {
                selected.add(msg);
                totalTokens += msgTokens;
            } else {
                break;
            }
        }
        
        return selected;
    }
}
```

### 3.4 RAG 场景的上下文注入

```java
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.messages.SystemMessage;
import org.springframework.ai.chat.messages.UserMessage;
import org.springframework.ai.document.Document;
import org.springframework.ai.vectorstore.SearchRequest;
import org.springframework.ai.vectorstore.VectorStore;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.IntStream;

@Service
public class RagContextService {

    private final ChatClient chatClient;
    private final VectorStore vectorStore;

    public RagContextService(ChatClient.Builder builder, VectorStore vectorStore) {
        this.chatClient = builder.build();
        this.vectorStore = vectorStore;
    }

    /**
     * RAG 节点：注入检索到的上下文（对应 Python 的 rag_node）
     */
    public String ragQuery(String userQuery, int topK) {
        // 1. 检索相关文档
        List<Document> relevantDocs = vectorStore.similaritySearch(
            SearchRequest.builder()
                .query(userQuery)
                .topK(topK)
                .build()
        );
        
        // 2. 组装上下文
        String contextText = IntStream.range(0, relevantDocs.size())
            .mapToObj(i -> "文档" + (i + 1) + "：" + relevantDocs.get(i).getText())
            .collect(Collectors.joining("\n\n"));
        
        // 3. 注入到提示词
        String augmentedQuery = """
            基于以下参考资料回答问题：
            
            %s
            
            问题：%s
            
            要求：
            - 仅使用提供的资料回答
            - 如果资料不足，说明"无法回答"
            - 引用资料来源
            """.formatted(contextText, userQuery);
        
        return chatClient.prompt()
            .user(augmentedQuery)
            .call()
            .content();
    }
}
```

### 4.1 长任务中的上下文保持

**场景**：研究 Agent 需要多轮搜索，保持关键发现

```python
class ResearchState(TypedDict):
    """研究状态"""
    messages: Annotated[list, add_messages]
    key_findings: list[str]  # 关键发现（长期保存）
    search_history: list[str]  # 搜索历史
    iteration: int

def research_node(state: ResearchState) -> ResearchState:
    """研究节点"""
    # 1. 执行搜索
    search_result = perform_search(state["messages"][-1].content)
    
    # 2. 提取关键信息
    key_facts = extract_facts(search_result)
    
    # 3. 更新状态
    return {
        "messages": [AIMessage(content=search_result)],
        "key_findings": state.get("key_findings", []) + key_facts,
        "search_history": state.get("search_history", []) + [state["messages"][-1].content],
        "iteration": state.get("iteration", 0) + 1
    }

def synthesize_node(state: ResearchState) -> ResearchState:
    """综合节点：使用所有关键发现"""
    # 使用所有关键发现生成最终答案
    findings_text = "\n".join(state["key_findings"])
    
    response = llm.invoke([
        HumanMessage(content=f"""
        基于以下研究发现，生成综合报告：
        
        {findings_text}
        
        搜索历史：{state["search_history"]}
        """)
    ])
    
    return {
        "messages": [AIMessage(content=response.content)]
    }
```

### 4.2 RAG 场景的上下文注入

**场景**：知识库问答，注入相关文档

```python
from langchain_core.prompts import ChatPromptTemplate

# 定义 RAG 提示词
rag_prompt = ChatPromptTemplate.from_messages([
    ("system", """你是知识问答助手。

基于提供的参考资料回答问题。

规则：
1. 仅使用参考资料中的信息
2. 如果资料不足，回答"根据现有资料无法确定"
3. 引用来源
4. 不要编造信息

参考资料：
{context}
"""),
    ("human", "{question}")
])

# 在 LangGraph 中使用
def qa_node(state: AgentState) -> AgentState:
    """问答节点"""
    question = state["messages"][-1].content
    
    # 检索相关文档
    docs = vector_store.similarity_search(question, k=5)
    context = "\n\n".join([d.page_content for d in docs])
    
    # 使用 RAG 提示词
    response = llm.invoke([
        SystemMessage(content=rag_prompt.format(context=context, question=question))
    ])
    
    return {"messages": [AIMessage(content=response.content)]}
```
