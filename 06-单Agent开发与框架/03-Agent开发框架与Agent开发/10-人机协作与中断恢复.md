## 1. LangGraph 的中断机制

### 1.1 interrupt() 函数（完整代码）

在任意节点暂停，等待人类输入：

```python
from langgraph.types import interrupt
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage
from typing import TypedDict, Annotated

llm = ChatOpenAI(model="gpt-4o")

class ReviewState(TypedDict):
    messages: Annotated[list, add_messages]
    draft: str
    human_feedback: str

def generate_draft(state: ReviewState) -> ReviewState:
    """生成草稿"""
    response = llm.invoke([
        HumanMessage(content=f"起草一份报告：{state['messages'][-1].content}")
    ])
    
    return {"draft": response.content}

def human_review_node(state: ReviewState) -> ReviewState:
    """人类审核节点"""
    # 中断执行，等待人类反馈
    feedback = interrupt({
        "question": "请审核以下草稿：",
        "draft": state["draft"],
        "options": ["approve", "revise", "reject"]
    })
    
    return {"human_feedback": feedback}

def process_feedback(state: ReviewState) -> ReviewState:
    """处理人类反馈"""
    feedback = state["human_feedback"]
    
    if feedback == "approve":
        return {"messages": [AIMessage(content=f"审核通过！\n\n{state['draft']}")]}
    elif feedback == "revise":
        # 重新生成
        response = llm.invoke([
            HumanMessage(content=f"根据审核意见修改：{state['draft']}")
        ])
        return {"draft": response.content, "messages": [AIMessage(content=response.content)]}
    else:  # reject
        return {"messages": [AIMessage(content="草稿被拒绝，需要重新起草")]}

# 构建图
workflow = StateGraph(ReviewState)
workflow.add_node("generate", generate_draft)
workflow.add_node("review", human_review_node)
workflow.add_node("process", process_feedback)

workflow.add_edge(START, "generate")
workflow.add_edge("generate", "review")
workflow.add_edge("review", "process")
workflow.add_edge("process", END)

# 编译（必须指定检查点才能使用 interrupt）
from langgraph.checkpoint.memory import MemorySaver

memory = MemorySaver()
agent = workflow.compile(checkpointer=memory)

# 第一次执行（会在 review 节点中断）
thread_config = {"configurable": {"thread_id": "review_001"}}

try:
    result = agent.invoke(
        {"messages": [HumanMessage(content="写一份 AI 发展趋势报告")]},
        config=thread_config
    )
except Exception as e:
    print(f"中断，等待人类审核：{e}")

# 人类审核后，恢复执行
human_input = "approve"  # 或 "revise" 或 "reject"

result = agent.invoke(
    {"human_feedback": human_input},
    config=thread_config
)

print(result["messages"][-1]["content"])
```

### 1.2 中断恢复（完整代码）

中断后可以从任意点恢复：

```python
from langgraph.checkpoint.sqlite import SqliteSaver
import sqlite3

# 使用 SQLite 持久化
conn = sqlite3.connect("agent_state.db", check_same_thread=False)
sqlite_memory = SqliteSaver(conn)

agent = workflow.compile(checkpointer=sqlite_memory)

# 执行（中断）
thread_config = {"configurable": {"thread_id": "session_123"}}

try:
    result = agent.invoke(
        {"messages": [HumanMessage(content="审查代码")]},
        config=thread_config
    )
except:
    print("中断")

# 查看当前状态
snapshot = agent.get_state(thread_config)
print(f"当前节点：{snapshot.metadata.get('step')}")
print(f"草稿：{snapshot.values.get('draft')}")

# 人类审核后恢复
result = agent.invoke(
    {"human_feedback": "approve"},
    config=thread_config
)

# 甚至可以在中断前修改状态
agent.update_state(
    thread_config,
    {"draft": "修改后的草稿内容"}
)

# 然后恢复
result = agent.invoke(None, config=thread_config)
```

### 1.3 Command 模式（完整代码）

使用 Command 明确指定下一步：

```python
from langgraph.types import interrupt, Command

def review_with_command(state: ReviewState) -> Command:
    """使用 Command 的审核节点"""
    feedback = interrupt({
        "question": "审核草稿",
        "draft": state["draft"]
    })
    
    # 使用 Command 决定下一步
    if feedback == "approve":
        return Command(goto="finalize")
    elif feedback == "revise":
        return Command(
            goto="generate",
            update={"messages": [HumanMessage(content="请修改草稿")]}
        )
    else:  # reject
        return Command(
            goto=END,
            update={"messages": [AIMessage(content="草稿被拒绝")]}
        )

## 2. 实战模式

### 2.1 人类审批（完整代码）

代码提交前需要人类审批：

```python
from langgraph.types import interrupt
from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage
from langchain_core.tools import tool
from langchain.agents import create_agent

llm = ChatOpenAI(model="gpt-4o")

@tool
def commit_code(code: str, message: str) -> str:
    """提交代码到仓库"""
    return f"已提交：{message}\n代码：{code[:100]}..."

class CodeCommitState(TypedDict):
    messages: Annotated[list, add_messages]
    code: str
    commit_message: str
    approved: bool

def generate_code(state: CodeCommitState) -> CodeCommitState:
    """生成代码"""
    response = llm.invoke([
        HumanMessage(content=f"编写代码：{state['messages'][-1].content}")
    ])
    
    return {"code": response.content}

def human_approval(state: CodeCommitState) -> CodeCommitState:
    """人类审批"""
    approval = interrupt({
        "question": "审批代码提交",
        "code": state["code"][:500],
        "commit_message": state.get("commit_message", "代码更新")
    })
    
    return {"approved": approval.lower() in ["approve", "yes", "同意"]}

def commit_or_reject(state: CodeCommitState) -> CodeCommitState:
    """提交或拒绝"""
    if state["approved"]:
        # 提交代码
        commit_tool = commit_code
        result = commit_tool.invoke({
            "code": state["code"],
            "message": state.get("commit_message", "代码更新")
        })
        
        return {"messages": [AIMessage(content=f"✅ 代码已提交：{result}")]}
    else:
        return {"messages": [AIMessage(content="❌ 代码提交被拒绝")]}

# 构建图
workflow = StateGraph(CodeCommitState)
workflow.add_node("generate", generate_code)
workflow.add_node("approval", human_approval)
workflow.add_node("commit", commit_or_reject)

workflow.add_edge(START, "generate")
workflow.add_edge("generate", "approval")
workflow.add_edge("approval", "commit")
workflow.add_edge("commit", END)

agent = workflow.compile(checkpointer=MemorySaver())

# 执行
result = agent.invoke(
    {"messages": [HumanMessage(content="写一个快速排序函数")]},
    config={"configurable": {"thread_id": "commit_001"}}
)
```

from langgraph.types import interrupt, Command

def tool_call_with_human_edit(state: AgentState) -> Command:

    # Agent 决定调用工具
    tool_name = "search_web"
    tool_args = {"query": state["messages"][-1].content}

## 3. 实战案例

### 3.1 代码提交前审批

完整工作流：生成代码 → 运行测试 → 人类审批 → 提交

```python
from langgraph.types import interrupt
from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage

llm = ChatOpenAI(model="gpt-4o")

class CodeReviewWorkflow(TypedDict):
    messages: Annotated[list, add_messages]
    code: str
    test_results: List[str]
    approved: bool

def write_code(state: CodeReviewWorkflow) -> CodeReviewWorkflow:
    """编写代码"""
    response = llm.invoke([
        HumanMessage(content=f"实现功能：{state['messages'][-1].content}")
    ])
    return {"code": response.content}

def run_tests(state: CodeReviewWorkflow) -> CodeReviewWorkflow:
    """运行测试"""
    # 模拟测试
    test_results = [
        "✅ 单元测试通过",
        "✅ 集成测试通过",
        "⚠️ 性能测试：建议优化"
    ]
    return {"test_results": test_results}

def human_review(state: CodeReviewWorkflow) -> CodeReviewWorkflow:
    """人类审核"""
    review_prompt = f"""
    审核代码：
    
    {state['code']}
    
    测试结果：
    {state['test_results']}
    
    是否批准提交？(approve/reject)
    """
    
    decision = interrupt({"question": review_prompt})
    
    return {"approved": decision.lower() == "approve"}

def finalize(state: CodeReviewWorkflow) -> CodeReviewWorkflow:
    """完成"""
    if state["approved"]:
        return {"messages": [AIMessage(content="✅ 代码已批准并提交")]}
    else:
        return {"messages": [AIMessage(content="❌ 代码未通过审核")]}

# 构建图
workflow = StateGraph(CodeReviewWorkflow)
workflow.add_node("write", write_code)
workflow.add_node("test", run_tests)
workflow.add_node("review", human_review)
workflow.add_node("finalize", finalize)

workflow.add_edge(START, "write")
workflow.add_edge("write", "test")
workflow.add_edge("test", "review")
workflow.add_edge("review", "finalize")
workflow.add_edge("finalize", END)

agent = workflow.compile(checkpointer=MemorySaver())

# 执行
result = agent.invoke(
    {"messages": [HumanMessage(content="实现线程安全的单例模式")]},
    config={"configurable": {"thread_id": "review_001"}}
)
```

### 3.2 敏感操作确认

def sensitive_operation(state: AgentState) -> AgentState:
    """敏感操作需要确认"""
    operation = state["messages"][-1].content
    
    # 检测敏感操作
    sensitive_keywords = ["删除", "delete", "DROP", "发送", "send"]
    is_sensitive = any(kw in operation.lower() for kw in sensitive_keywords)
    
    if is_sensitive:
        # 需要人类确认
        confirm = interrupt({
            "question": "⚠️ 这是敏感操作，确认继续？",
            "operation": operation
        })
        
        if confirm.lower() not in ["yes", "确认", "continue"]:
            return {"messages": [AIMessage(content="操作已取消")]}
    
    # 执行操作
    result = execute_operation(operation)
    return {"messages": [AIMessage(content=f"操作完成：{result}")]}
```

## 4. Spring AI 人机协作实现

### 4.1 人类审批（Interrupt 模式）

对应 Python 的 interrupt() 函数：

```java
import lombok.Data;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 人类审批服务（对应 Python 的 interrupt() 机制）
 * 
 * 注意：Spring AI 没有直接的 interrupt() 函数
 * 需要通过自定义 Advisor 或外部状态机实现
 */
@Service
public class HumanApprovalService {

    private final ChatClient chatClient;
    private final ApprovalStateManager approvalManager;

    public HumanApprovalService(ChatModel chatModel, ApprovalStateManager approvalManager) {
        this.approvalManager = approvalManager;
        
        this.chatClient = ChatClient.builder(chatModel)
            .defaultSystem("""
                你是代码生成助手。
                生成代码后需要人类审批才能继续。
                """)
            .build();
    }

    @Data
    public static class ApprovalState {
        private String threadId;        // 会话 ID
        private String draft;           // 草稿
        private String humanFeedback;   // 人类反馈
        private boolean approved;       // 是否通过
    }

    /**
     * 生成草稿（对应 Python 的 generate_draft）
     */
    public String generateDraft(String userQuery, String threadId) {
        String response = chatClient.prompt()
            .user(userQuery)
            .call()
            .content();
        
        // 保存草稿，等待审批
        ApprovalState state = new ApprovalState();
        state.setThreadId(threadId);
        state.setDraft(response);
        approvalManager.saveState(state);
        
        return response;
    }

    /**
     * 人类审批（对应 Python 的 human_review_node）
     */
    public String submitApproval(String threadId, String feedback) {
        ApprovalState state = approvalManager.getState(threadId);
        
        if (state == null) {
            return "未找到审批请求";
        }
        
        state.setHumanFeedback(feedback);
        state.setApproved("approve".equalsIgnoreCase(feedback));
        approvalManager.saveState(state);
        
        if (state.isApproved()) {
            return "审核通过！\n\n" + state.getDraft();
        } else if ("revise".equalsIgnoreCase(feedback)) {
            // 重新生成
            return chatClient.prompt()
                .user("根据审核意见修改：" + state.getDraft())
                .call()
                .content();
        } else {
            return "草稿被拒绝，需要重新起草";
        }
    }
}

/**
 * 审批状态管理器（模拟 LangGraph 的 checkpointer）
 */
@Service
class ApprovalStateManager {
    
    private final Map<String, HumanApprovalService.ApprovalState> states = new ConcurrentHashMap<>();
    
    public void saveState(HumanApprovalService.ApprovalState state) {
        states.put(state.getThreadId(), state);
    }
    
    public HumanApprovalService.ApprovalState getState(String threadId) {
        return states.get(threadId);
    }
}
```

### 4.2 中断恢复

对应 Python 的中断恢复机制：

```java
import org.springframework.ai.chat.memory.ChatMemory;
import org.springframework.ai.chat.memory.InMemoryChatMemory;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.client.advisor.MessageChatMemoryAdvisor;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

import java.util.List;

/**
 * 中断恢复服务（对应 Python 的中断恢复机制）
 */
@Service
public class InterruptResumeService {

    private final ChatMemory chatMemory;
    private final ChatClient chatClient;

    public InterruptResumeService(ChatModel chatModel) {
        // 使用持久化记忆（对应 Python 的 SqliteSaver）
        this.chatMemory = new InMemoryChatMemory();
        
        this.chatClient = ChatClient.builder(chatModel)
            .defaultAdvisors(
                new MessageChatMemoryAdvisor(chatMemory, "session_123", 20)
            )
            .build();
    }

    /**
     * 执行（可能中断）
     */
    public String execute(String userQuery, String threadId) {
        // 创建带记忆的客户端
        ChatClient sessionClient = ChatClient.builder(chatClient.getChatModel())
            .defaultAdvisors(
                new MessageChatMemoryAdvisor(chatMemory, threadId, 20)
            )
            .build();
        
        // 执行对话（自动保存状态）
        return sessionClient.prompt()
            .user(userQuery)
            .call()
            .content();
    }

    /**
     * 查看当前状态（对应 Python 的 agent.get_state()）
     */
    public String getState(String threadId) {
        var messages = chatMemory.get(threadId, 20);
        
        StringBuilder sb = new StringBuilder();
        sb.append("当前会话状态：\n");
        sb.append("消息数：").append(messages.size()).append("\n");
        
        if (!messages.isEmpty()) {
            sb.append("最新消息：").append(messages.get(messages.size() - 1).getText());
        }
        
        return sb.toString();
    }

    /**
     * 恢复执行（对应 Python 的 agent.invoke(None, config=thread_config)）
     */
    public String resume(String threadId, String userFeedback) {
        // 从上次中断处继续
        ChatClient sessionClient = ChatClient.builder(chatClient.getChatModel())
            .defaultAdvisors(
                new MessageChatMemoryAdvisor(chatMemory, threadId, 20)
            )
            .build();
        
        return sessionClient.prompt()
            .user(userFeedback)
            .call()
            .content();
    }

    /**
     * 更新状态（对应 Python 的 agent.update_state()）
     */
    public void updateState(String threadId, String newDraft) {
        // 清除旧状态，保存新状态
        chatMemory.clear(threadId);
    }
}
```

### 4.3 敏感操作确认

import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

import java.util.List;

/**
 * 敏感操作确认服务
 */
@Service
public class SensitiveOperationService {

    private final ChatClient chatClient;

    public SensitiveOperationService(ChatModel chatModel) {
        this.chatClient = ChatClient.builder(chatModel)
            .build();
    }

    /**
     * 检测并确认敏感操作（对应 Python 的 sensitive_operation）
     */
    public String executeWithConfirmation(String operation, boolean humanConfirm) {
        // 检测敏感操作
        List<String> sensitiveKeywords = List.of("删除", "delete", "DROP", "发送", "send");
        boolean isSensitive = sensitiveKeywords.stream()
            .anyMatch(kw -> operation.toLowerCase().contains(kw.toLowerCase()));
        
        if (isSensitive) {
            // 需要人类确认
            if (!humanConfirm) {
                return "操作已取消";
            }
        }
        
        // 执行操作
        return chatClient.prompt()
            .user("执行操作：" + operation)
            .call()
            .content();
    }
}

### 4.4 完整工作流：代码审查与审批

对应 Python 的代码提交前审批完整工作流：

```java
import lombok.Data;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;

/**
 * 代码审查工作流（对应 Python 的 CodeReviewWorkflow）
 * 
 * 完整流程：生成代码 → 运行测试 → 人类审批 → 提交
 */
@Service
public class CodeReviewWorkflowService {

    private final ChatClient codeClient;
    private final ApprovalStateManager approvalManager;

    public CodeReviewWorkflowService(ChatModel chatModel, ApprovalStateManager approvalManager) {
        this.approvalManager = approvalManager;
        
        this.codeClient = ChatClient.builder(chatModel)
            .defaultSystem("你是代码生成专家")
            .build();
    }

    @Data
    public static class CodeReviewState {
        private String threadId;
        private String code;
        private List<String> testResults;
        private boolean approved;
    }

    /**
     * 编写代码（对应 Python 的 write_code）
     */
    public String writeCode(String userQuery, String threadId) {
        String code = codeClient.prompt()
            .user("实现功能：" + userQuery)
            .call()
            .content();
        
        // 保存状态
        CodeReviewState state = new CodeReviewState();
        state.setThreadId(threadId);
        state.setCode(code);
        state.setTestResults(new ArrayList<>());
        
        return code;
    }

    /**
     * 运行测试（对应 Python 的 run_tests）
     */
    public List<String> runTests(String code) {
        // 模拟测试
        return List.of(
            "✅ 单元测试通过",
            "✅ 集成测试通过",
            "⚠️ 性能测试：建议优化"
        );
    }

    /**
     * 人类审核（对应 Python 的 human_review）
     */
    public String submitReview(String threadId, String decision) {
        CodeReviewState state = approvalManager.getState(threadId);
        
        if (state == null) {
            return "未找到审查请求";
        }
        
        state.setApproved("approve".equalsIgnoreCase(decision));
        
        if (state.isApproved()) {
            return "✅ 代码已批准并提交";
        } else {
            return "❌ 代码未通过审核";
        }
    }

    /**
     * 执行完整工作流
     */
    public String executeWorkflow(String userQuery, String threadId) {
        // 1. 编写代码
        String code = writeCode(userQuery, threadId);
        System.out.println("代码生成完成");
        
        // 2. 运行测试
        List<String> testResults = runTests(code);
        System.out.println("测试完成：" + testResults);
        
        // 3. 等待人类审批（中断）
        System.out.println("等待人类审批...");
        // 实际应通过 API 或 WebSocket 等待
        
        return "代码已生成，等待审批";
    }
}
