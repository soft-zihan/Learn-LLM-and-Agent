## 1. 记忆系统架构

### 1.1 为什么需要记忆

LLM 本身是**无状态**的，每次调用都是独立的。记忆系统让 Agent 能够：

- **跨会话记住用户偏好**
- **追踪长期任务进度**
- **从历史交互中学习**
- **避免重复查询相同信息**

### 1.2 记忆分类

```
记忆系统
├── 短期记忆（Working Memory）
│   ├── 当前会话的对话历史
│   ├── 工具调用结果
│   └── 临时状态信息
│   特点：快速访问，会话结束后消失
│
├── 长期记忆（Long-term Memory）
│   ├── 用户偏好和画像
│   ├── 历史任务的关键信息
│   └── 领域知识和事实
│   特点：持久化存储，跨会话可用
│
└── 情景记忆 vs 语义记忆
    ├── 情景记忆（Episodic）：特定事件的记忆
    │   例："昨天用户问了关于 AI 的问题"
    │
    └── 语义记忆（Semantic）：通用知识
        例："用户是 Python 开发者"
```

## 2. LangGraph 的记忆实现

### 2.1 MemorySaver（内存，开发用）

最简单的记忆实现，适合开发和测试：

```python
from langgraph.checkpoint.memory import MemorySaver

# 创建内存检查点
memory = MemorySaver()

# 编译时指定
agent = workflow.compile(checkpointer=memory)

# 使用 thread_id 区分会话
result1 = agent.invoke(
    {"messages": [HumanMessage(content="你好")]},
    config={"configurable": {"thread_id": "session_1"}}
)

# 同一会话，Agent 会记住历史
result2 = agent.invoke(
    {"messages": [HumanMessage(content="我刚才说了什么？")]},
    config={"configurable": {"thread_id": "session_1"}}
)

# 不同会话，记忆隔离
result3 = agent.invoke(
    {"messages": [HumanMessage(content="你好")]},
    config={"configurable": {"thread_id": "session_2"}}
)
```

## 3. 记忆检索与更新

### 3.1 自定义记忆状态

除了使用内置检查点，可以在 State 中自定义记忆字段：

```python
from typing import TypedDict, Annotated
from langgraph.graph.message import add_messages

class MemoryState(TypedDict):
    """带记忆的状态"""
    messages: Annotated[list, add_messages]
    
    # 长期记忆
    user_preferences: dict  # 用户偏好
    key_facts: list[str]    # 关键事实
    task_history: list[dict]  # 任务历史

def memory_node(state: MemoryState) -> MemoryState:
    """记忆节点：提取和更新记忆"""
    last_message = state["messages"][-1].content
    
    # 从对话中提取用户偏好
    if "我喜欢" in last_message:
        preference = extract_preference(last_message)
        state["user_preferences"].update(preference)
    
    # 提取关键事实
    facts = extract_facts(last_message)
    if facts:
        state["key_facts"].extend(facts)
    
    return {
        "user_preferences": state["user_preferences"],
        "key_facts": state["key_facts"]
    }
```

### 3.2 相似度检索（向量数据库）

对于大量记忆，使用向量数据库进行相似度检索：

```python
from langchain_openai import OpenAIEmbeddings
from langchain_community.vectorstores import Chroma

class VectorMemory:
    """向量记忆系统"""
    
    def __init__(self, persist_directory="./memory_db"):
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = Chroma(
            embedding_function=self.embeddings,
            persist_directory=persist_directory
        )
    
    def add_memory(self, memory: str, metadata: dict = None):
        """添加记忆"""
        self.vectorstore.add_texts(
            texts=[memory],
            metadatas=[metadata or {}]
        )
    
    def retrieve_memories(self, query: str, top_k: int = 5) -> list:
        """检索相关记忆"""
        results = self.vectorstore.similarity_search(
            query,
            k=top_k
        )
        return [doc.page_content for doc in results]
    
    def update_memory(self, old_memory: str, new_memory: str):
        """更新记忆"""
        # 删除旧记忆，添加新记忆
        self.vectorstore.delete(where={"text": old_memory})
        self.add_memory(new_memory)

# 使用
memory = VectorMemory()

# 添加记忆
memory.add_memory("用户是 Python 开发者", {"type": "preference"})
memory.add_memory("用户喜欢简洁的代码示例", {"type": "preference"})

# 检索记忆
relevant = memory.retrieve_memories("用户的技术背景", top_k=3)
print(relevant)
# ["用户是 Python 开发者", "用户喜欢简洁的代码示例"]
```

### 3.3 时间衰减策略

随着时间推移，记忆的重要性会降低：

```python
import time

class TimeDecayMemory:
    """带时间衰减的记忆"""
    
    def __init__(self, half_life: float = 86400 * 7):  # 7 天半衰期
        self.half_life = half_life
        self.memories = []
    
    def add_memory(self, memory: str, importance: float = 1.0):
        """添加记忆"""
        self.memories.append({
            "content": memory,
            "importance": importance,
            "timestamp": time.time()
        })
    
    def get_current_importance(self, memory: dict) -> float:
        """计算当前重要性（时间衰减）"""
        age = time.time() - memory["timestamp"]
        decay = 0.5 ** (age / self.half_life)
        return memory["importance"] * decay
    
    def retrieve_top_memories(self, top_k: int = 5) -> list:
        """检索最重要的记忆"""
        # 按当前重要性排序
        sorted_memories = sorted(
            self.memories,
            key=lambda m: self.get_current_importance(m),
            reverse=True
        )
        
        return sorted_memories[:top_k]
    
    def forget_old_memories(self, threshold: float = 0.1):
        """遗忘不重要的记忆"""
        self.memories = [
            m for m in self.memories
            if self.get_current_importance(m) > threshold
        ]

# 使用
memory = TimeDecayMemory(half_life=86400 * 7)  # 7 天半衰期

# 添加记忆
memory.add_memory("用户是 Python 开发者", importance=0.9)
memory.add_memory("用户今天问了天气", importance=0.3)

# 检索
top_memories = memory.retrieve_top_memories(top_k=3)
```

```java
            .defaultAdvisors(
                new MessageChatMemoryAdvisor(chatMemory, sessionId, 10)
            )
            .build();

            .user(userMessage)
            .call()
            .content();
```

```java
public class PersistentChatService {

                new MessageChatMemoryAdvisor(chatMemory, sessionId, 20)
            )
            .build();
```

### 2.3 Redis ChatMemory（高并发场景）

Redis 适合高并发、需要 TTL 的场景：

```java
import org.springframework.ai.chat.memory.ChatMemory;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.stereotype.Component;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

@Component
public class RedisChatMemory implements ChatMemory {

    private final StringRedisTemplate redisTemplate;
    private final Duration ttl;  // 过期时间

    public RedisChatMemory(StringRedisTemplate redisTemplate, Duration ttl) {
        this.redisTemplate = redisTemplate;
        this.ttl = ttl;  // 例如：24 小时自动清理
    }

    @Override
    public void add(String conversationId, List<org.springframework.ai.chat.messages.Message> messages) {
        String key = "chat_memory:" + conversationId;
        
        // 序列化并保存消息
        for (org.springframework.ai.chat.messages.Message msg : messages) {
            redisTemplate.opsForList().rightPush(key, serializeMessage(msg));
        }
        
        // 设置 TTL
        redisTemplate.expire(key, ttl);
    }

    @Override
    public List<org.springframework.ai.chat.messages.Message> get(String conversationId, int lastN) {
        String key = "chat_memory:" + conversationId;
        
        // 获取最近 N 条消息
        List<String> serializedMessages = redisTemplate.opsForList().range(key, -lastN, -1);
        
        List<org.springframework.ai.chat.messages.Message> messages = new ArrayList<>();
        if (serializedMessages != null) {
            for (String serialized : serializedMessages) {
                messages.add(deserializeMessage(serialized));
            }
        }
        
        return messages;
    }

    @Override
    public void clear(String conversationId) {
        redisTemplate.delete("chat_memory:" + conversationId);
    }

    private String serializeMessage(org.springframework.ai.chat.messages.Message msg) {
        // 简化版序列化，实际应使用 JSON
        return msg.getClass().getSimpleName() + ":" + msg.getText();
    }

    private org.springframework.ai.chat.messages.Message deserializeMessage(String serialized) {
        // 简化版反序列化
        String[] parts = serialized.split(":", 2);
        return new org.springframework.ai.chat.messages.UserMessage(parts[1]);
    }
}
```

## 3. 自定义记忆系统

### 3.1 用户偏好记忆

对应 Python 的自定义记忆状态：

```java
import lombok.Data;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Data
@Component
public class UserMemoryManager {

    // 长期记忆
    private final Map<String, UserMemory> userMemories = new HashMap<>();

    @Data
    public static class UserMemory {
        private Map<String, Object> preferences = new HashMap<>();  // 用户偏好
        private List<String> keyFacts = new ArrayList<>();          // 关键事实
        private List<TaskHistory> taskHistory = new ArrayList<>();  // 任务历史
    }

    @Data
    public static class TaskHistory {
        private String task;
        private String result;
        private long timestamp;
    }

    /**
     * 从对话中提取并更新记忆（对应 Python 的 memory_node）
     */
    public void updateMemory(String userId, String lastMessage) {
        UserMemory memory = userMemories.computeIfAbsent(userId, k -> new UserMemory());
        
        // 提取用户偏好
        if (lastMessage.contains("我喜欢")) {
            String preference = extractPreference(lastMessage);
            memory.getPreferences().put("preference", preference);
        }
        
        // 提取关键事实
        List<String> facts = extractFacts(lastMessage);
        if (!facts.isEmpty()) {
            memory.getKeyFacts().addAll(facts);
        }
    }

    /**
     * 提取偏好（简化版）
     */
    private String extractPreference(String message) {
        // 实际应使用 LLM 或规则提取
        if (message.contains("Python")) return "Python";
        if (message.contains("Java")) return "Java";
        return "unknown";
    }

    /**
     * 提取事实（简化版）
     */
    private List<String> extractFacts(String message) {
        List<String> facts = new ArrayList<>();
        // 实际应使用 LLM 或规则提取
        if (message.contains("开发者")) {
            facts.add("用户是开发者");
        }
        return facts;
    }

    /**
     * 获取用户记忆摘要
     */
    public String getMemorySummary(String userId) {
        UserMemory memory = userMemories.get(userId);
        if (memory == null) return "暂无记忆";
        
        StringBuilder summary = new StringBuilder();
        summary.append("用户偏好：").append(memory.getPreferences()).append("\n");
        summary.append("关键事实：").append(memory.getKeyFacts()).append("\n");
        
        return summary.toString();
    }
}
```

### 3.3 时间衰减记忆

对应 Python 的 TimeDecayMemory：

```java
import lombok.Data;
import org.springframework.stereotype.Component;

import java.time.Instant;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;

@Component
public class TimeDecayMemory {

    private final double halfLife;  // 半衰期（秒）
    private final List<MemoryEntry> memories = new ArrayList<>();

    @Data
    public static class MemoryEntry {
        private String content;
        private double importance;
        private Instant timestamp;

        public MemoryEntry(String content, double importance) {
            this.content = content;
            this.importance = importance;
            this.timestamp = Instant.now();
        }
    }

    public TimeDecayMemory(double halfLifeSeconds) {
        this.halfLife = halfLifeSeconds;
    }

    /**
     * 添加记忆（对应 Python 的 add_memory）
     */
    public void addMemory(String memory, double importance) {
        memories.add(new MemoryEntry(memory, importance));
    }

    /**
     * 计算当前重要性（时间衰减）（对应 Python 的 get_current_importance）
     */
    public double getCurrentImportance(MemoryEntry memory) {
        double age = Instant.now().getEpochSecond() - memory.getTimestamp().getEpochSecond();
        double decay = Math.pow(0.5, age / halfLife);
        return memory.getImportance() * decay;
    }

    /**
     * 检索最重要的记忆（对应 Python 的 retrieve_top_memories）
     */
    public List<MemoryEntry> retrieveTopMemories(int topK) {
        return memories.stream()
            .sorted(Comparator.comparingDouble(this::getCurrentImportance).reversed())
            .limit(topK)
            .toList();
    }

    /**
     * 遗忘不重要的记忆（对应 Python 的 forget_old_memories）
     */
    public void forgetOldMemories(double threshold) {
        memories.removeIf(memory -> getCurrentImportance(memory) <= threshold);
    }
}
```

### 3.4 记忆压缩

```java
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.stream.Collectors;

@Service
public class MemoryCompressor {

    private final ChatClient compressorClient;

    public MemoryCompressor(ChatClient.Builder builder) {
        // 使用便宜模型做压缩
        this.compressorClient = builder.build();
    }

    /**
     * 压缩记忆为摘要（对应 Python 的 compress_memories）
     */
    public String compressMemories(List<String> memories) {
        String memoriesText = memories.stream()
            .map(m -> "- " + m)
            .collect(Collectors.joining("\n"));
        
        String prompt = """
            将以下记忆压缩为简洁的摘要，保留关键信息：
            
            %s
            
            摘要（100字以内）：
            """.formatted(memoriesText);
        
        return compressorClient.prompt()
            .user(prompt)
            .call()
            .content();
    }

    /**
     * 智能记忆管理：超过阈值时自动压缩
     */
    public List<String> manageMemories(List<String> memories, int threshold) {
        if (memories.size() > threshold) {
            // 压缩早期记忆
            String summary = compressMemories(
                memories.subList(0, memories.size() - 10)
            );
            
            // 保留最近记忆
            List<String> recentMemories = memories.subList(
                Math.max(0, memories.size() - 10),
                memories.size()
            );
            
            List<String> result = new java.util.ArrayList<>();
            result.add(summary);
            result.addAll(recentMemories);
            
            return result;
        }
        
        return memories;
    }
}
```

## 4. 实战案例

### 4.1 跨会话记忆

**场景**：记住用户偏好，跨会话使用

```python
from langgraph.graph import StateGraph, START, END
from typing import TypedDict, Annotated
from langgraph.graph.message import add_messages

class UserMemoryState(TypedDict):
    """用户记忆状态"""
    messages: Annotated[list, add_messages]
    user_profile: dict  # 用户画像（持久化）

def extract_preferences(state: UserMemoryState) -> UserMemoryState:
    """从对话中提取用户偏好"""
    last_message = state["messages"][-1].content
    user_profile = state.get("user_profile", {})
    
    # 提取语言偏好
    if "我用 Python" in last_message or "我是 Python开发者" in last_message:
        user_profile["language"] = "Python"
    
    # 提取经验水平
    if "我是新手" in last_message:
        user_profile["level"] = "beginner"
    elif "我是专家" in last_message:
        user_profile["level"] = "expert"
    
    return {"user_profile": user_profile}

def personalized_response(state: UserMemoryState) -> UserMemoryState:
    """基于用户画像生成个性化回复"""
    user_profile = state.get("user_profile", {})
    
    # 根据经验水平调整回答
    level = user_profile.get("level", "intermediate")
    language = user_profile.get("language", "Python")
    
    if level == "beginner":
        explanation_style = "简单易懂，包含详细注释"
    else:
        explanation_style = "专业简洁，直接给出最佳实践"
    
    prompt = f"""
    用户画像：{user_profile}
    解释风格：{explanation_style}
    
    问题：{state['messages'][-1].content}
    """
    
    response = llm.invoke([HumanMessage(content=prompt)])
    
    return {"messages": [AIMessage(content=response.content)]}

# 构建图
workflow = StateGraph(UserMemoryState)
workflow.add_node("extract", extract_preferences)
workflow.add_node("respond", personalized_response)

workflow.add_edge(START, "extract")
workflow.add_edge("extract", "respond")
workflow.add_edge("respond", END)

# 使用 SQLite 保存用户画像
memory = SqliteSaver(conn)
agent = workflow.compile(checkpointer=memory)

# 第一次会话
result1 = agent.invoke(
    {"messages": [HumanMessage(content="我是新手，我用 Python")]},
    config={"configurable": {"thread_id": "user_123"}}
)

# 第二次会话（不同 thread_id，但可以从数据库恢复 user_profile）
result2 = agent.invoke(
    {"messages": [HumanMessage(content="如何写一个函数？")]},
    config={"configurable": {"thread_id": "user_123_session_2"}}
)
# Agent 会根据用户画像（新手、Python）调整回答风格
```

### 4.2 用户偏好学习

**场景**：从交互中自动学习用户偏好

```python
class PreferenceLearningMemory:
    """偏好学习记忆"""
    
    def __init__(self):
        self.preferences = {}
        self.feedback_history = []
    
    def learn_from_feedback(self, user_feedback: str, context: dict):
        """从用户反馈中学习"""
        self.feedback_history.append({
            "feedback": user_feedback,
            "context": context,
            "timestamp": time.time()
        })
        
        # 分析反馈，提取偏好
        if "太长了" in user_feedback or "简洁一点" in user_feedback:
            self.preferences["response_length"] = "short"
        elif "详细一点" in user_feedback:
            self.preferences["response_length"] = "long"
        
        if "用中文" in user_feedback:
            self.preferences["language"] = "chinese"
    
    def get_preferences(self) -> dict:
        """获取学习到的偏好"""
        return self.preferences

# 在 Agent 中使用
def adaptive_agent(state: AgentState) -> AgentState:
    """自适应 Agent：根据用户偏好调整"""
    preferences = memory.get_preferences()
    
    # 根据偏好调整提示词
    if preferences.get("response_length") == "short":
        style_instruction = "回答简洁，控制在100字以内"
    else:
        style_instruction = "回答详细，包含示例和解释"
    
    prompt = f"""
    用户偏好：{preferences}
    风格指令：{style_instruction}
    
    问题：{state['messages'][-1].content}
    """
    
    response = llm.invoke([HumanMessage(content=prompt)])
    return {"messages": [AIMessage(content=response.content)]}
```
