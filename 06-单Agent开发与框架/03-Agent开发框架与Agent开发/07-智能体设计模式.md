## 1. LangGraph 智能体架构

### 1.1 StateGraph 设计原则

StateGraph 是 LangGraph 的核心，设计时遵循以下原则：

**原则 1：状态最小化**

只保存必要的状态，避免冗余：

```python
from typing import TypedDict, Annotated
from langgraph.graph.message import add_messages

# ✅ 推荐：最小状态
class AgentState(TypedDict):
    messages: Annotated[list, add_messages]
    current_step: str
    result: str

# ❌ 不推荐：冗余状态
class BadState(TypedDict):
    messages: Annotated[list, add_messages]
    user_input: str  # 冗余，messages 中已有
    llm_response: str  # 冗余，messages 中已有
    step1_result: str  # 应该合并到 result
    step2_result: str
```

**原则 2：使用 Reducer 管理复杂状态**

```python
from typing import TypedDict, Annotated
import operator

class ComplexState(TypedDict):
    messages: Annotated[list, add_messages]
    
    # 追加模式
    findings: Annotated[list, operator.add]
    
    # 覆盖模式
    current_task: str
    
    # 自定义 Reducer
    best_answer: Annotated[str, lambda old, new: new if new else old]
```

### 1.2 节点拆分策略

**策略 1：按职责拆分**

```python
def research_node(state: AgentState) -> AgentState:

def synthesize_node(state: AgentState) -> AgentState:
```

**策略 2：按复杂度拆分**

复杂逻辑拆分为多个简单节点：

```python
# ❌ 不推荐：一个节点做太多事
def complex_node(state: AgentState) -> AgentState:
    # 1. 搜索
    results = search()
    # 2. 过滤

def search_node(state: AgentState) -> AgentState:
    results = search()
    return {"messages": [AIMessage(content=results)]}

    filtered = filter_results(state["messages"][-1].content)

    analysis = analyze(state["messages"][-1].content)
    return {"messages": [AIMessage(content=analysis)]}
```

## 2. 经典设计模式

### 2.1 ReAct 模式（完整代码）

ReAct（Reasoning + Acting）是最常用的 Agent 模式：

```python
from langchain.agents import create_agent
from langchain_core.tools import tool
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage

# 定义工具
@tool
def search_web(query: str) -> str:
    """搜索网络获取信息"""
    # 实际应调用搜索 API
    return f"搜索结果：{query} 的相关信息..."

@tool
def calculate(expression: str) -> str:
    """执行数学计算"""
    try:
        result = eval(expression)
        return f"计算结果：{result}"
    except Exception as e:
        return f"计算错误：{str(e)}"

# 创建 ReAct Agent
agent = create_agent(
    model=ChatOpenAI(model="gpt-4o"),
    tools=[search_web, calculate],
    system_prompt="""你是一个有用的助手。

使用工具帮助用户解决问题：
1. 需要最新信息时使用 search_web
2. 需要计算时使用 calculate
3. 不确定时先查询再回答
"""
)

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="2024年诺贝尔物理学奖是谁？如果是奇数年获得，计算年份乘以2")]
})

print(result["messages"][-1]["content"])
```

**ReAct 执行流程**：
```
用户问题 
  → LLM 思考（Thought）
  → 选择工具（Action）
  → 执行工具（Observation）
  → LLM 继续思考
  → 最终答案（Final Answer）
```

### 2.2 Plan-and-Execute 模式（完整代码）

先规划，再执行：

```python
from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage
import json

llm = ChatOpenAI(model="gpt-4o")

class PlanState(TypedDict):
    messages: Annotated[list, add_messages]
    plan: List[str]  # 执行计划
    current_step: int
    results: List[str]

def plan_node(state: PlanState) -> PlanState:
    """规划节点：生成执行计划"""
    user_query = state["messages"][-1].content
    
    plan_prompt = f"""
    为以下任务生成详细的执行计划（JSON 数组）：
    
    任务：{user_query}
    
    要求：
    - 分解为 3-5 个步骤
    - 每步明确具体
    - 返回 JSON 数组
    
    示例：["步骤1", "步骤2", "步骤3"]
    """
    
    response = llm.invoke([SystemMessage(content=plan_prompt)])
    
    # 解析计划
    try:
        plan = json.loads(response.content)
    except:
        plan = ["执行研究", "分析数据", "生成报告"]
    
    return {
        "plan": plan,
        "current_step": 0,
        "results": []
    }

def execute_node(state: PlanState) -> PlanState:
    """执行节点：执行当前步骤"""
    plan = state["plan"]
    current_step = state["current_step"]
    
    if current_step >= len(plan):
        return {"current_step": current_step}
    
    step_task = plan[current_step]
    
    # 执行步骤
    execute_prompt = f"""
    执行以下步骤：
    
    步骤：{step_task}
    之前的结果：{state['results']}
    
    请执行并返回结果。
    """
    
    response = llm.invoke([HumanMessage(content=execute_prompt)])
    
    return {
        "results": state["results"] + [response.content],
        "current_step": current_step + 1
    }

def should_continue(state: PlanState) -> str:
    """判断是否继续执行"""
    if state["current_step"] < len(state["plan"]):
        return "continue"
    return "end"

def finalize_node(state: PlanState) -> PlanState:
    """综合节点：生成最终答案"""
    results = state["results"]
    
    finalize_prompt = f"""
    基于以下执行结果，生成最终答案：
    
    计划：{state["plan"]}
    结果：{results}
    
    请生成 comprehensive 的最终答案。
    """
    
    response = llm.invoke([HumanMessage(content=finalize_prompt)])
    
    return {"messages": [AIMessage(content=response.content)]}

# 构建图
workflow = StateGraph(PlanState)
workflow.add_node("plan", plan_node)
workflow.add_node("execute", execute_node)
workflow.add_node("finalize", finalize_node)

workflow.add_edge(START, "plan")
workflow.add_edge("plan", "execute")
workflow.add_conditional_edges(
    "execute",
    should_continue,
    {"continue": "execute", "end": "finalize"}
)
workflow.add_edge("finalize", END)

agent = workflow.compile()

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="分析 AI Agent 的未来发展趋势")]
})

print(result["messages"][-1]["content"])
```

### 2.3 Reflection 模式（完整代码）

自我反思和改进：

```python
from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage

llm = ChatOpenAI(model="gpt-4o")

class ReflectionState(TypedDict):
    messages: Annotated[list, add_messages]
    draft: str  # 草稿
    feedback: str  # 反馈
    iteration: int
    max_iterations: int

def generate_draft(state: ReflectionState) -> ReflectionState:
    """生成草稿"""
    query = state["messages"][-1].content
    
    prompt = f"""
    为以下任务生成初稿：
    
    任务：{query}
    
    请生成详细的初稿。
    """
    
    response = llm.invoke([HumanMessage(content=prompt)])
    
    return {
        "draft": response.content,
        "iteration": state.get("iteration", 0) + 1
    }

def reflect_and_critique(state: ReflectionState) -> ReflectionState:
    """反思和批评"""
    draft = state["draft"]
    
    critique_prompt = f"""
    作为严格的审稿人，批判以下草稿的质量：
    
    草稿：{draft}
    
    评估维度：
    1. 准确性：信息是否准确
    2. 完整性：是否覆盖所有要点
    3. 逻辑性：推理是否严密
    4. 可读性：表达是否清晰
    
    请提供具体的改进建议。
    """
    
    response = llm.invoke([HumanMessage(content=critique_prompt)])
    
    return {"feedback": response.content}

def improve_draft(state: ReflectionState) -> ReflectionState:
    """改进草稿"""
    draft = state["draft"]
    feedback = state["feedback"]
    
    improve_prompt = f"""
    根据以下反馈改进草稿：
    
    原草稿：{draft}
    
    反馈：{feedback}
    
    请生成改进后的版本。
    """
    
    response = llm.invoke([HumanMessage(content=improve_prompt)])
    
    return {"draft": response.content}

def should_improve(state: ReflectionState) -> str:
    """判断是否需要继续改进"""
    if state.get("iteration", 0) < state.get("max_iterations", 3):
        return "improve"
    return "finalize"

def finalize(state: ReflectionState) -> ReflectionState:
    """完成"""
    return {
        "messages": [AIMessage(content=f"最终答案（经过 {state['iteration']} 轮改进）：\n\n{state['draft']}")]
    }

# 构建图
workflow = StateGraph(ReflectionState)
workflow.add_node("generate", generate_draft)
workflow.add_node("critique", reflect_and_critique)
workflow.add_node("improve", improve_draft)
workflow.add_node("finalize", finalize)

workflow.add_edge(START, "generate")
workflow.add_edge("generate", "critique")
workflow.add_edge("critique", "improve")
workflow.add_conditional_edges(
    "improve",
    should_improve,
    {"improve": "critique", "finalize": "finalize"}
)
workflow.add_edge("finalize", END)

agent = workflow.compile()

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="解释量子计算的基本原理")],
    "max_iterations": 3
})

print(result["messages"][-1]["content"])
```

```python
from langchain_core.tools import BaseTool, tool
```

```python
def route_by_task(state: AgentState) -> str:

    query = state["messages"][-1].content.lower()
```

### 3.3 错误处理与重试

```python
from langgraph.types import RetryPolicy
import time

def search_with_retry(state: AgentState) -> AgentState:
    """带重试的搜索"""
    max_retries = 3
    
    for attempt in range(max_retries):
        try:
            result = search_web(state["messages"][-1].content)
            return {"messages": [AIMessage(content=result)]}
        except Exception as e:
            if attempt < max_retries - 1:
                wait_time = 2 ** attempt  # 指数退避
                time.sleep(wait_time)
                print(f"重试 {attempt + 1}/{max_retries}")
            else:
                return {
                    "messages": [AIMessage(content=f"搜索失败：{str(e)}")]
                }
```

## 4. 验证与护栏

### 4.1 输入验证

```python
def validate_input(state: AgentState) -> AgentState:
    """验证用户输入"""
    user_input = state["messages"][-1].content
    
    # 长度检查
    if len(user_input) > 10000:
        return {
            "messages": [AIMessage(content="输入过长，请精简到 10000 字以内")]
        }
    
    # 安全检查
    dangerous_patterns = ["DELETE FROM", "DROP TABLE", "<script>"]
    if any(pattern in user_input.upper() for pattern in dangerous_patterns):
        return {
            "messages": [AIMessage(content="输入包含不安全内容")]
        }
    
    # 通过验证，继续
    return state
```

### 4.2 输出审核

```python
def audit_output(state: AgentState) -> AgentState:
    """审核 Agent 输出"""
    output = state["messages"][-1].content
    
    # 使用 LLM 审核
    audit_prompt = f"""
    审核以下输出是否合适：
    
    输出：{output}
    
    检查：
    1. 是否包含不当内容
    2. 是否泄露敏感信息
    3. 是否有事实错误
    
    如果安全，返回 "PASS"；否则返回 "FAIL: 原因"
    """
    
    audit_result = llm.invoke([HumanMessage(content=audit_prompt)])
    
    if "FAIL" in audit_result.content:
        return {
            "messages": [AIMessage(content="输出审核未通过，需要重新生成")]
        }
    
    return state
```
