## 1. 子智能体概念

### 1.1 主智能体 vs 子智能体

**主智能体**：负责任务规划和调度  
**子智能体**：专注执行特定子任务

```
主智能体（调度者）
├── 理解用户任务
├── 分解为子任务
├── 选择合适的子智能体
├── 调度执行
└── 汇总结果

子智能体 A（专业执行者）
└── 专注网络搜索

子智能体 B（专业执行者）
└── 专注数据库查询

子智能体 C（专业执行者）
└── 专注知识库检索
```

### 1.2 何时需要子智能体

**适合场景**：
- 任务需要多种专业技能
- 需要并行执行独立子任务
- 单个 Agent 上下文不够用

**不适合场景**：
- 简单任务（单个 Agent 即可）
- 子任务强依赖（无法并行）

## 2. 子智能体实现模式

### 2.1 字典式子智能体（完整代码）

使用字典注册和路由子智能体：

```python
from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langchain.agents import create_agent
from langchain_core.tools import tool
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage

llm = ChatOpenAI(model="gpt-4o")

# 定义子智能体的工具
@tool
def search_web(query: str) -> str:
    """搜索网络"""
    return f"网络搜索结果：{query}"

@tool
def query_database(sql: str) -> str:
    """查询数据库"""
    return f"数据库结果：{sql}"

@tool
def search_knowledge(query: str) -> str:
    """搜索知识库"""
    return f"知识库结果：{query}"

# 创建子智能体
search_agent = create_agent(
    model=llm,
    tools=[search_web],
    system_prompt="你是网络搜索专家。只使用 search_web 工具。"
)

db_agent = create_agent(
    model=llm,
    tools=[query_database],
    system_prompt="你是数据库专家。只使用 query_database 工具。"
)

knowledge_agent = create_agent(
    model=llm,
    tools=[search_knowledge],
    system_prompt="你是知识库专家。只使用 search_knowledge 工具。"
)

# 字典式注册
subagents = {
    "search": search_agent,
    "database": db_agent,
    "knowledge": knowledge_agent
}

class SubAgentState(TypedDict):
    messages: Annotated[list, add_messages]
    current_agent: str
    agent_results: dict

def router_node(state: SubAgentState) -> SubAgentState:
    """路由节点：决定使用哪个子智能体"""
    user_query = state["messages"][-1].content
    
    # 使用 LLM 决定路由
    route_prompt = f"""
    根据用户问题，选择最合适的子智能体：
    
    问题：{user_query}
    
    可选：
    - search：需要最新网络信息
    - database：需要查询结构化数据
    - knowledge：需要查询内部知识库
    
    只返回智能体名称（search/database/knowledge）。
    """
    
    response = llm.invoke([HumanMessage(content=route_prompt)])
    agent_name = response.content.strip().lower()
    
    return {"current_agent": agent_name}

def execute_subagent(state: SubAgentState) -> SubAgentState:
    """执行子智能体"""
    agent_name = state["current_agent"]
    subagent = subagents.get(agent_name)
    
    if not subagent:
        return {
            "messages": [AIMessage(content=f"未知的子智能体：{agent_name}")]
        }
    
    # 执行子智能体
    result = subagent.invoke({
        "messages": [state["messages"][-1]]
    })
    
    # 保存结果
    agent_results = state.get("agent_results", {})
    agent_results[agent_name] = result["messages"][-1]["content"]
    
    return {
        "messages": [AIMessage(content=result["messages"][-1]["content"])],
        "agent_results": agent_results
    }

def should_continue(state: SubAgentState) -> str:
    """判断是否需要更多子智能体"""
    user_query = state["messages"][0]["content"]
    
    # 简单判断：如果需要多个来源，继续
    if any(kw in user_query for kw in ["和", "以及", "both", "and"]):
        return "continue"
    return "finalize"

def finalize_node(state: SubAgentState) -> SubAgentState:
    """汇总结果"""
    all_results = state.get("agent_results", {})
    results_text = "\n\n".join([f"{k}：{v}" for k, v in all_results.items()])
    
    summary_prompt = f"""
    汇总以下子智能体的结果：
    
    {results_text}
    
    生成综合答案。
    """
    
    response = llm.invoke([HumanMessage(content=summary_prompt)])
    
    return {"messages": [AIMessage(content=response.content)]}

# 构建图
workflow = StateGraph(SubAgentState)
workflow.add_node("route", router_node)
workflow.add_node("execute", execute_subagent)
workflow.add_node("finalize", finalize_node)

workflow.add_edge(START, "route")
workflow.add_edge("route", "execute")
workflow.add_conditional_edges(
    "execute",
    should_continue,
    {
        "continue": "route",
        "finalize": "finalize"
    }
)
workflow.add_edge("finalize", END)

agent = workflow.compile()

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="搜索 AI 最新进展和数据库中的相关论文")]
})

print(result["messages"][-1]["content"])
```

```python
import asyncio

    return f"搜索结果：{query}"

@tool
def analyze_data(data: str) -> str:
    """数据分析"""
    import time
    time.sleep(1)  # 模拟计算时间
    return f"分析结果：{data}"

# 创建子智能体
search_agent = create_agent(
    model=llm,
    tools=[search_web],
    system_prompt="网络搜索专家"

    model=llm,
    tools=[analyze_data],
    system_prompt="数据分析专家"
)

async def run_subagents_parallel(query: str):
    """并行运行多个子智能体"""
    # 创建异步任务
    tasks = [
        asyncio.create_task(
            search_agent.ainvoke({"messages": [HumanMessage(content=f"搜索：{query}")]})
        ),
        asyncio.create_task(
            analysis_agent.ainvoke({"messages": [HumanMessage(content=f"分析：{query}")]})
        )

    # 等待所有任务完成
    results = await asyncio.gather(*tasks)
    
    # 汇总结果
    search_result = results[0]["messages"][-1]["content"]
    analysis_result = results[1]["messages"][-1]["content"]
    
    return {
        "search": search_result,
        "analysis": analysis_result
    }
```

### 2.3 子智能体嵌套限制

**问题**：子智能体再调用子智能体会导致无限嵌套

**解决方案**：设置最大深度

```python
class SubAgentRouter:
    """子智能体路由器（带深度限制）"""
    
    def __init__(self, max_depth: int = 3):
        self.max_depth = max_depth
    
    def can_dispatch(self, current_depth: int) -> bool:
        """检查是否可以继续分发"""
        return current_depth < self.max_depth
    
    def dispatch(self, task: str, current_depth: int = 0):
        """分发任务"""
        if not self.can_dispatch(current_depth):
            raise Exception(f"达到最大嵌套深度 {self.max_depth}")
        
        # 执行子智能体
        result = execute_subagent(task)
        
        # 如果子任务需要进一步分解
        if needs_further_decomposition(result):
            return self.dispatch(result, current_depth + 1)
        
        return result
```

## 3. 路由模式

### 3.1 基于任务类型路由（完整代码）

```python
from typing import Literal
from langgraph.graph import StateGraph, START, END

class RouterState(TypedDict):
    messages: Annotated[list, add_messages]
    task_type: str

def classify_task(state: RouterState) -> RouterState:
    """分类任务类型"""
    user_query = state["messages"][-1].content
    
    classification_prompt = f"""
    分类以下任务类型：
    
    任务：{user_query}
    
    类型：
    - research：需要研究和搜索
    - calculation：需要数学计算
    - coding：需要编写代码
    - writing：需要撰写文档
    
    只返回类型名称。
    """
    
    response = llm.invoke([HumanMessage(content=classification_prompt)])
    task_type = response.content.strip().lower()
    
    return {"task_type": task_type}

def route_by_task(state: RouterState) -> Literal["research", "calculation", "coding", "writing"]:
    """根据任务类型路由"""
    return state["task_type"]

# 各专业节点
def research_node(state: RouterState) -> RouterState:
    """研究节点"""
    result = llm.invoke([
        HumanMessage(content=f"深入研究：{state['messages'][-1].content}")
    ])
    return {"messages": [AIMessage(content=result.content)]}

def calculation_node(state: RouterState) -> RouterState:
    """计算节点"""
    result = llm.invoke([
        HumanMessage(content=f"执行计算：{state['messages'][-1].content}")
    ])
    return {"messages": [AIMessage(content=result.content)]}

def coding_node(state: RouterState) -> RouterState:
    """编程节点"""
    result = llm.invoke([
        HumanMessage(content=f"编写代码：{state['messages'][-1].content}")
    ])
    return {"messages": [AIMessage(content=result.content)]}

def writing_node(state: RouterState) -> RouterState:
    """写作节点"""
    result = llm.invoke([
        HumanMessage(content=f"撰写内容：{state['messages'][-1].content}")
    ])
    return {"messages": [AIMessage(content=result.content)]}

# 构建路由图
workflow = StateGraph(RouterState)
workflow.add_node("classify", classify_task)
workflow.add_node("research", research_node)
workflow.add_node("calculation", calculation_node)
workflow.add_node("coding", coding_node)
workflow.add_node("writing", writing_node)

workflow.add_edge(START, "classify")
workflow.add_conditional_edges(
    "classify",
    route_by_task,
    {
        "research": "research",
        "calculation": "calculation",
        "coding": "coding",
        "writing": "writing"
    }
)

# 所有节点都到 END
for node in ["research", "calculation", "coding", "writing"]:
    workflow.add_edge(node, END)

agent = workflow.compile()

# 执行
result = agent.invoke({
    "messages": [HumanMessage(content="写一个快速排序算法")]
})

print(result["messages"][-1]["content"])
```

### 3.2 LangGraph 中的子图路由（完整代码）

使用嵌套图实现子智能体：

```python
from langgraph.graph import StateGraph, START, END

# 子图：搜索智能体
def search_subgraph(state: RouterState) -> RouterState:
    """搜索子图"""
    # 可以包含多个节点
    step1 = llm.invoke([HumanMessage(content=f"生成搜索查询：{state['messages'][-1].content}")])
    step2 = llm.invoke([HumanMessage(content=f"搜索并总结：{step1.content}")])
    
    return {"messages": [AIMessage(content=step2.content)]}

# 子图：分析智能体
def analysis_subgraph(state: RouterState) -> RouterState:
    """分析子图"""
    step1 = llm.invoke([HumanMessage(content=f"分析数据：{state['messages'][-1].content}")])
    step2 = llm.invoke([HumanMessage(content=f"生成报告：{step1.content}")])
    
    return {"messages": [AIMessage(content=step2.content)]}

# 主图
workflow = StateGraph(RouterState)

# 添加子图作为节点
workflow.add_node("search_agent", search_subgraph)
workflow.add_node("analysis_agent", analysis_subgraph)

def route_to_subgraph(state: RouterState) -> str:
    """路由到子图"""
    if "搜索" in state["messages"][-1].content:
        return "search_agent"
    return "analysis_agent"

workflow.add_edge(START, "classify")
workflow.add_conditional_edges(
    "classify",
    route_to_subgraph,
    {
        "search_agent": "search_agent",
        "analysis_agent": "analysis_agent"
    }
)
workflow.add_edge("search_agent", END)
workflow.add_edge("analysis_agent", END)

agent = workflow.compile()
```

## 4. Spring AI 子智能体与路由实现

### 4.1 字典式子智能体

对应 Python 的字典式子智能体实现：

```java
import lombok.Data;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.ai.tool.annotation.Tool;
import org.springframework.ai.tool.annotation.ToolParam;
import org.springframework.stereotype.Component;
import org.springframework.stereotype.Service;

import java.util.Map;

/**
 * 字典式子智能体（对应 Python 的 subagents 字典）
 */
@Service
public class SubAgentService {

    private final ChatClient routerClient;  // 路由决策
    private final Map<String, ChatClient> subAgents;  // 子智能体字典

    public SubAgentService(ChatModel chatModel) {
        // 1. 创建路由客户端
        this.routerClient = ChatClient.builder(chatModel)
            .defaultSystem("""
                你是任务路由器，根据用户问题选择最合适的子智能体：
                - search: 需要最新网络信息
                - database: 需要查询结构化数据
                - knowledge: 需要查询内部知识库
                - code: 需要代码分析
                
                只返回智能体名称（search/database/knowledge/code）。
                """)
            .build();
        
        // 2. 创建子智能体字典
        this.subAgents = Map.of(
            "search", createSubAgent(chatModel, "search", "你是网络搜索专家。只使用 searchWeb 工具。"),
            "database", createSubAgent(chatModel, "database", "你是数据库专家。只使用 queryDatabase 工具。"),
            "knowledge", createSubAgent(chatModel, "knowledge", "你是知识库专家。只使用 searchKnowledge 工具。"),
            "code", createSubAgent(chatModel, "code", "你是代码分析专家。只使用 analyzeCode 工具。")
        );
    }

    /**
     * 创建子智能体
     */
    private ChatClient createSubAgent(ChatModel chatModel, String name, String systemPrompt) {
        return ChatClient.builder(chatModel)
            .defaultSystem(systemPrompt)
            .defaultTools(new SubAgentTools(name))  // 注入对应工具
            .build();
    }

    /**
     * 路由节点（对应 Python 的 router_node）
     */
    public String route(String userQuery) {
        String response = routerClient.prompt()
            .user(userQuery)
            .call()
            .content();
        
        return response.strip().toLowerCase();
    }

    /**
     * 执行子智能体（对应 Python 的 execute_subagent）
     */
    public String executeSubAgent(String agentName, String userQuery) {
        ChatClient subAgent = subAgents.get(agentName);
        
        if (subAgent == null) {
            return "未知的子智能体：" + agentName;
        }
        
        return subAgent.prompt()
            .user(userQuery)
            .call()
            .content();
    }

    /**
     * 完整流程：路由 + 执行
     */
    public String executeWithRouting(String userQuery) {
        // 1. 路由
        String agentName = route(userQuery);
        System.out.println("路由到：" + agentName);
        
        // 2. 执行
        return executeSubAgent(agentName, userQuery);
    }
}

/**
 * 子智能体工具类
 */

    }

        }
        return "数据库结果：" + sql;
    }
    
    @Tool(description = "搜索知识库")
    public String searchKnowledge(@ToolParam(description = "查询内容") String query) {
        if (!"knowledge".equals(agentType)) {
            return "无权使用此工具";
        }
        return "知识库结果：" + query;
    }
    
    @Tool(description = "分析代码")
    public String analyzeCode(@ToolParam(description = "代码内容") String code) {
        if (!"code".equals(agentType)) {
```

### 4.2 基于任务类型路由

对应 Python 的 classify_task + route_by_task：

```java
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

/**
 * 基于任务类型路由（对应 Python 的 classify_task 和 route_by_task）
 */
@Service
public class TaskRouterService {

    private final ChatClient classifierClient;
    private final ChatClient researchClient;
    private final ChatClient calculationClient;
    private final ChatClient codingClient;
    private final ChatClient writingClient;

    public TaskRouterService(ChatModel chatModel) {
        // 分类器
        this.classifierClient = ChatClient.builder(chatModel)
            .defaultSystem("""
                分类以下任务类型：
                - research：需要研究和搜索
                - calculation：需要数学计算
                - coding：需要编写代码
                - writing：需要撰写文档
                
                只返回类型名称。
                """)
            .build();
        
        // 专业节点
        this.researchClient = ChatClient.builder(chatModel)
            .defaultSystem("你是研究专家，擅长深入分析和搜索信息")
            .build();
        
        this.calculationClient = ChatClient.builder(chatModel)
            .defaultSystem("你是数学计算专家")
            .build();
        
        this.codingClient = ChatClient.builder(chatModel)
            .defaultSystem("你是编程专家，擅长编写高质量代码")
            .build();
        
        this.writingClient = ChatClient.builder(chatModel)
            .defaultSystem("你是写作专家，擅长撰写文档和内容")
            .build();
    }

    /**
     * 分类任务类型（对应 Python 的 classify_task）
     */
    public String classifyTask(String userQuery) {
        String response = classifierClient.prompt()
            .user(userQuery)
            .call()
            .content();
        
        return response.strip().toLowerCase();
    }

    /**
     * 根据任务类型路由（对应 Python 的 route_by_task）
     */
    public String routeByTask(String taskType, String userQuery) {
        return switch (taskType) {
            case "research" -> researchClient.prompt()
                .user("深入研究：" + userQuery)
                .call()
                .content();
            case "calculation" -> calculationClient.prompt()
                .user("执行计算：" + userQuery)
                .call()
                .content();
            case "coding" -> codingClient.prompt()
                .user("编写代码：" + userQuery)
                .call()
                .content();
            case "writing" -> writingClient.prompt()
                .user("撰写内容：" + userQuery)
                .call()
                .content();
            default -> "未知任务类型：" + taskType;
        };
    }

    /**
     * 完整流程：分类 + 路由
     */
    public String executeWithRouting(String userQuery) {
        // 1. 分类
        String taskType = classifyTask(userQuery);
        System.out.println("任务类型：" + taskType);
        
        // 2. 路由
        return routeByTask(taskType, userQuery);
    }
}
```

### 4.3 并行子智能体

对应 Python 的异步子智能体并行执行：

```java
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;
import org.springframework.stereotype.Service;

import java.util.concurrent.CompletableFuture;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

/**
 * 并行子智能体（对应 Python 的 run_subagents_parallel）
 */
@Service
public class ParallelSubAgentService {

    private final ChatClient searchClient;
    private final ChatClient analysisClient;
    private final ExecutorService executor = Executors.newFixedThreadPool(10);

    public ParallelSubAgentService(ChatModel chatModel) {
        this.searchClient = ChatClient.builder(chatModel)
            .defaultSystem("你是网络搜索专家")
            .build();
        
        this.analysisClient = ChatClient.builder(chatModel)
            .defaultSystem("你是数据分析专家")
            .build();
    }

    /**
     * 并行运行多个子智能体（对应 Python 的 asyncio.gather）
     */
    public ParallelResult runSubAgentsParallel(String query) {
        // 创建异步任务
        CompletableFuture<String> searchTask = CompletableFuture.supplyAsync(
            () -> searchClient.prompt()
                .user("搜索：" + query)
                .call()
                .content(),
            executor
        );
        
        CompletableFuture<String> analysisTask = CompletableFuture.supplyAsync(
            () -> analysisClient.prompt()
                .user("分析：" + query)
                .call()
                .content(),
            executor
        );
        
        // 等待所有任务完成（对应 Python 的 await asyncio.gather）
        CompletableFuture.allOf(searchTask, analysisTask).join();
        
        // 获取结果
        try {
            String searchResult = searchTask.get();
            String analysisResult = analysisTask.get();
            
            return new ParallelResult(searchResult, analysisResult);
        } catch (Exception e) {
            throw new RuntimeException("并行执行失败", e);
        }
    }

    public record ParallelResult(String searchResult, String analysisResult) {
```
